   299→// ---- Interactive Mode ----
   300→async function startInteractive(options: any) {
   301→  const config = initConfig(options.apiKey);
   302→  if (options.model) config.defaultModel = options.model;
   303→  if (options.theme) config.ui.theme = options.theme as 'dark' | 'light' | 'dracula' | 'nord';
   304→  if (options.noStreaming) config.ui.streaming = false;
   305→  if (options.permissionMode) config.permissionMode = options.permissionMode;
   306→  if (options.diffPreview === true) config.diffPreview = true;
   307→  if (options.diffPreview === false) config.diffPreview = false;
   308→  if (options.cache === true) config.promptCache.enabled = true;
   309→  if (options.cache === false) config.promptCache.enabled = false;
   310→  if (options.spendingLimit) config.spendingLimit = options.spendingLimit;
   311→  if (options.autoApprove) {
   312→    config.tools.autoApprove = [...config.tools.autoApprove, ...config.tools.requireApproval];
   313→    config.tools.requireApproval = [];
   314→    config.permissionMode = 'yolo';
   315→  }
   316→
   317→  // Check API key
   318→  if (!config.apiKey && !options.ollama) {
   319→    console.log(chalk.red('\nOpenRouter API key not configured!'));
   320→    console.log(chalk.yellow('\nSet it with:'));
   321→    console.log(`  ${chalk.cyan('neuro config --set-key YOUR_API_KEY')}`);
   322→    console.log(`  ${chalk.cyan('export OPENROUTER_API_KEY=YOUR_API_KEY')}`);
   323→    console.log(`  ${chalk.cyan('neuro -k YOUR_API_KEY')}`);
   324→    console.log(chalk.gray('\nGet your key at: https://openrouter.ai/keys\n'));
   325→    process.exit(1);
   326→  }
   327→
   328→  // Initialize engine
   329→  const engine = new NeuroEngine(config);
   330→
   331→  // Set effort level if specified
   332→  if (options.effort) {
   333→    engine.modelRouter.setEffort(options.effort);
   334→  }
   335→
   336→  // Set output style if specified
   337→  if (options.style) {
   338→    engine.styleManager.setStyle(options.style);
   339→  }
   340→
   341→  // Enable thinking if specified
   342→  if (options.thinking) {
   343→    engine.extendedThinking.setMode('full');
   344→    engine.extendedThinking.toggleDisplay();
   345→  }
   346→
   347→  // Enable sandbox if specified
   348→  if (options.sandbox) {
   349→    engine.sandbox.enable();
   350→  }
   351→
   352→  // Enable prompt cache if specified
   353→  if (options.cache) {
   354→    config.promptCache.enabled = true;
   355→  }
   356→
   357→  // Resume session if requested
   358→  if (options.resume) {
   359→    const session = engine.sessionManager.load(options.resume);
   360→    if (session) {
   361→      engine.ui.success(`Resumed session: ${session.id.slice(0, 20)}...`);
   362→      if (options.fork) {
   363→        const forked = engine.sessionManager.create(process.cwd(), config.defaultModel);
   364→        forked.messages = [...session.messages];
   365→        engine.sessionManager.save();
   366→        engine.ui.success(`Forked to new session: ${forked.id.slice(0, 20)}...`);
   367→      }
   368→    } else {
   369→      engine.ui.error(`Session not found: ${options.resume}`);
   370→    }
   371→  } else if (options.continue) {
   372→    const sessions = engine.sessionManager.list();
   373→    if (sessions.length > 0) {
   374→      const session = engine.sessionManager.load(sessions[0].id);
   375→      if (session) {
   376→        engine.ui.success(`Continued session: ${session.id.slice(0, 20)}...`);
   377→      }
   378→    } else {
   379→      engine.ui.warning('No sessions found to continue');
   380→    }
   381→  }
   382→
   383→  // Initialize auto-updater and check for updates in background
   384→  const updater = new AutoUpdater({ currentVersion: VERSION });
   385→  const updateCheck = updater.checkOnStartup(); // Fire and forget — don't block startup
   386→
   387→  // Print banner
   388→  engine.ui.banner();
   389→  const theme = engine.ui.theme;
   390→  const permMode = engine.approval.getMode();
   391→  const permIcon = engine.approval.getModeIcon();
   392→  const activeStyle = engine.styleManager.getStyle();
   393→  console.log(theme.muted(`  Model: ${MODELS[config.defaultModel]?.name || config.defaultModel}`));
   394→  console.log(theme.muted(`  Permission: ${permIcon} ${permMode}`));
   395→  console.log(theme.muted(`  Style: ${activeStyle.name}`));
   396→  console.log(theme.muted(`  Thinking: ${engine.extendedThinking.getMode()}`));
   397→  console.log(theme.muted(`  Cache: ${config.promptCache.enabled ? 'on' : 'off'}`));
   398→  console.log(theme.muted(`  Working Dir: ${process.cwd()}`));
   399→  console.log(theme.muted(`  Type /help for commands, Tab for completion, Ctrl+C to exit\n`));
   400→
   401→  // Show update notification if available (after banner)
   402→  updateCheck.then((result) => {
   403→    if (result && result.hasUpdate) {
   404→      updater.showUpdateNotification(result);
   405→    }
   406→  }).catch(() => { /* silently ignore */ });
   407→
   408→  // Create readline with tab completion
   409→  const completionEngine = new CompletionEngine(process.cwd());
   410→  completionEngine.setAgentNames(Array.from(engine.agents.keys()));
   411→
   412→  const rl = createInterface({
   413→    input: process.stdin,
   414→    output: process.stdout,
   415→    prompt: chalk.cyan('❯ '),
   416→    historySize: 100,
   417→    completer: completionEngine.complete,
   418→  });
   419→
   420→  rl.prompt();
   421→
   422→  // Wire the main readline to the approval system to prevent dual-readline bug
   423→  // (double character input when approval prompt and main rl both read stdin)
   424→  engine.approval.setMainReadline(rl);
   425→
   426→  let currentMode: 'auto' | 'agent' | 'direct' = 'auto';
   427→  let currentAgent: string | undefined;
   428→  let isProcessing = false;
   429→
   430→  rl.on('line', async (line) => {
   431→    const input = line.trim();
   432→    if (!input) { rl.prompt(); return; }
   433→
   434→    // Prevent input while processing (approval prompt may be active)
   435→    if (isProcessing) {
   436→      return;
   437→    }
   438→
   439→    // Add to history
   440→    completionEngine.addHistory(input);
   441→
   442→    // Handle slash commands
   443→    if (input.startsWith('/')) {
   444→      const [cmd, ...args] = input.slice(1).split(' ');
   445→
   446→      switch (cmd) {
   447→        case 'help':
   448→          printHelp(engine);
   449→          break;
   450→
   451→        case 'model':
   452→          if (args[0]) {
   453→            engine.switchModel(args[0]);
   454→            engine.fallback.addFallbackModel(config.defaultModel);
   455→          } else {
   456→            engine.ui.modelList(config.defaultModel);
   457→          }
   458→          break;
   459→
   460→        case 'agent':
   461→          if (args[0]) {
   462→            const agentNames = Array.from(engine.agents.keys());
   463→            const found = agentNames.find(n => n.toLowerCase() === args[0].toLowerCase());
   464→            if (found) { currentAgent = found; currentMode = 'direct'; engine.ui.success(`Switched to agent: ${found}`); }
   465→            else { engine.ui.error(`Agent not found. Available: ${agentNames.join(', ')}`); }
   466→          } else {
   467→            const agents = Array.from(engine.agents.entries()).map(([name]) => ({ name, description: '', model: config.defaultModel }));
   468→            engine.ui.agentList(agents);
   469→          }
   470→          break;
   471→
   472→        case 'auto':
   473→          currentMode = 'auto'; currentAgent = undefined;
   474→          engine.ui.success('Mode: Auto (smart orchestration)');
   475→          break;
   476→
   477→        case 'orchestrate':
   478→        case 'plan':
   479→          currentMode = 'agent'; currentAgent = undefined;
   480→          engine.ui.success('Mode: Multi-agent orchestration');
   481→          break;
   482→
   483→        case 'direct':
   484→          currentMode = 'direct';
   485→          engine.ui.success('Mode: Direct (single agent)');
   486→          break;
   487→
   488→        case 'plan-mode':
   489→          engine.approval.setMode('plan');
   490→          engine.ui.success(`${engine.approval.getModeIcon()} Permission: plan (read-only)`);
   491→          break;
   492→
   493→        case 'stats':
   494→          const stats = engine.getSessionStats();
   495→          engine.ui.sessionStats(stats.inputTokens, stats.outputTokens, stats.cost);

   925→              const prResult = execSync(`gh pr create --title "${commitMsg}" --body "Auto-generated by NeuroCLI"`, { cwd: process.cwd(), encoding: 'utf-8' });
   926→              engine.ui.success(`PR created: ${prResult.trim()}`);
   927→            } catch {
   928→              engine.ui.info('Could not create PR (gh CLI not available or not a GitHub repo)');
   929→            }
   930→          } catch (e) { engine.ui.error(`Git operation failed: ${e instanceof Error ? e.message : String(e)}`); }
   931→          break;
   932→
   933→        case 'code-review':
   934→          engine.ui.info('Starting multi-agent code review...');
   935→          try {
   936→            const reviewResult = await engine.processMessage(
   937→              'Perform a thorough code review of all recent changes in this repository. Check for: bugs, security issues, performance problems, code style, test coverage. Provide findings with severity levels (CRITICAL, WARNING, INFO).',
   938→              'agent',
   939→            );
   940→          } catch (e) { engine.ui.error(`Code review failed: ${e instanceof Error ? e.message : String(e)}`); }
   941→          break;
   942→
   943→        case 'feedback':
   944→          console.log(chalk.bold('\nFeedback:\n'));
   945→          console.log('  Report issues: https://github.com/neuro-cli/neuro/issues');
   946→          console.log('  Discussions:   https://github.com/neuro-cli/neuro/discussions');
   947→          console.log(chalk.gray('\n  Your feedback helps make NeuroCLI better!'));
   948→          console.log();
   949→          break;
   950→
   951→        case 'cost':
   952→          engine.spendingMonitor.printReport();
   953→          if (engine.promptCache) {
   954→            console.log(chalk.bold('\nCache Savings:'));
   955→            engine.promptCache.printStats();
   956→          }
   957→          break;
   958→
   959→        case 'update':
   960→        case 'upgrade':
   961→          const updateSub = args[0];
   962→          if (updateSub === 'now') {
   963→            engine.ui.info('Updating NeuroCLI...');
   964→            const updateResult = await updater.performUpdate();
   965→            if (updateResult.success) {
   966→              engine.ui.success(updateResult.message);
   967→              console.log(chalk.yellow('  Please restart NeuroCLI to use the new version.'));
   968→            } else {
   969→              engine.ui.error(updateResult.message);
   970→            }
   971→          } else if (updateSub === 'check') {
   972→            engine.ui.info('Checking for updates...');
   973→            const checkResult = await updater.checkForUpdate(true);
   974→            if (checkResult.hasUpdate) {
   975→              updater.showUpdateDetails(checkResult);
   976→              updater.showUpdateNotification(checkResult);
   977→            } else {
   978→              updater.showUpToDate();
   979→            }
   980→          } else if (updateSub === 'dismiss') {
   981→            const checkResult = await updater.checkForUpdate(true);
   982→            if (checkResult.hasUpdate) {
   983→              updater.dismissVersion(checkResult.latestVersion);
   984→              engine.ui.success(`Dismissed update notification for v${checkResult.latestVersion}`);
   985→            } else {
   986→              engine.ui.info('No update to dismiss');
   987→            }
   988→          } else if (updateSub === 'auto') {
   989→            const enableAuto = args[1] !== 'off';
   990→            updater.setAutoUpdate(enableAuto);
   991→            engine.ui.success(`Auto-update: ${enableAuto ? 'enabled' : 'disabled'}`);
   992→          } else if (updateSub === 'interval') {
   993→            const hours = args[1] ? parseFloat(args[1]) : 24;
   994→            if (isNaN(hours) || hours < 1) {
   995→              engine.ui.error('Interval must be at least 1 hour');
   996→            } else {
   997→              updater.setCheckInterval(hours);
   998→              engine.ui.success(`Update check interval set to ${hours} hours`);
   999→            }
  1000→          } else if (updateSub === 'reset') {
  1001→            updater.resetDismissed();
  1002→            updater.forceNextCheck();
  1003→            engine.ui.success('Update preferences reset');
  1004→          } else {
  1005→            // Default: interactive update flow
  1006→            await updater.interactiveUpdate();
  1007→          }
  1008→          break;
  1009→
  1010→        case 'exit':
  1011→        case 'quit':
  1012→        case 'q':
  1013→          engine.mcpClient.disconnectAll().catch(() => {});
  1014→          engine.approval.close();
  1015→          engine.ui.info('Goodbye!');
  1016→          process.exit(0);
  1017→          break;
  1018→
  1019→        default:
  1020→          engine.ui.error(`Unknown command: /${cmd}. Type /help for available commands.`);
  1021→      }
  1022→
  1023→      rl.prompt();
  1024→      return;
  1025→    }
  1026→
  1027→    // Process message with the engine
  1028→    try {
  1029→      isProcessing = true;
  1030→      await engine.processMessage(input, currentMode, currentAgent);
  1031→    } catch (error) {
  1032→      engine.ui.error(error instanceof Error ? error.message : String(error));
  1033→    } finally {
  1034→      isProcessing = false;
  1035→    }
  1036→
  1037→    rl.prompt();
  1038→  });
  1039→
  1040→  rl.on('close', () => {
  1041→    engine.mcpClient.disconnectAll().catch(() => {});
  1042→    engine.approval.close();
  1043→    engine.ui.info('Goodbye!');
  1044→    process.exit(0);
  1045→  });
  1046→}
