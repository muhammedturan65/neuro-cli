   522тЖТ   522тЖТ  /**
   523тЖТ   523тЖТ   * Process a user message
   524тЖТ   524тЖТ   */
   525тЖТ   525тЖТ  async processMessage(
   526тЖТ   526тЖТ    message: string,
   527тЖТ   527тЖТ    mode: 'auto' | 'agent' | 'direct' = 'auto',
   528тЖТ   528тЖТ    targetAgent?: string,
   529тЖТ   529тЖТ  ): Promise<{ content: string; usage: TokenUsage; execution?: AgentExecution }> {
   530тЖТ   530тЖТ    // Check spending limit via spending monitor
   531тЖТ   531тЖТ    const limitCheck = this.spendingMonitor.checkLimit();
   532тЖТ   532тЖТ    if (!limitCheck.allowed) {
   533тЖТ   533тЖТ      this.ui.error(`Spending limit reached (${limitCheck.limitReached}). Use /spending to check or /config to adjust.`);
   534тЖТ   534тЖТ      return { content: 'Spending limit reached.', usage: { inputTokens: 0, outputTokens: 0, cost: 0 } };
   535тЖТ   535тЖТ    }
   536тЖТ   536тЖТ
   537тЖТ   537тЖТ    // Auto-activate skills based on prompt
   538тЖТ   538тЖТ    const activatedSkills = this.skillSystem.autoActivate(message);
   539тЖТ   539тЖТ    if (activatedSkills.length > 0) {
   540тЖТ   540тЖТ      for (const skill of activatedSkills) {
   541тЖТ   541тЖТ        this.ui.info(`Skill activated: ${skill.skill.name} (${skill.activatedBy})`);
   542тЖТ   542тЖТ      }
   543тЖТ   543тЖТ    }
   544тЖТ   544тЖТ
   545тЖТ   545тЖТ    // Model routing (if auto mode)
   546тЖТ   546тЖТ    let routeDecision: RoutingDecision | null = null;
   547тЖТ   547тЖТ    if (mode === 'auto') {
   548тЖТ   548тЖТ      routeDecision = this.modelRouter.route(message);
   549тЖТ   549тЖТ      if (routeDecision.model !== this.config.defaultModel) {
   550тЖТ   550тЖТ        this.ui.info(`Model router: ${routeDecision.complexity} task -> ${MODELS[routeDecision.model]?.name || routeDecision.model}`);
   551тЖТ   551тЖТ      }
   552тЖТ   552тЖТ    }
   553тЖТ   553тЖТ
   554тЖТ   554тЖТ    // Check prompt cache
   555тЖТ   555тЖТ    if (this.promptCache && this.config.promptCache.enabled) {
   556тЖТ   556тЖТ      const session = this.sessionManager.getCurrent();
   557тЖТ   557тЖТ      if (session) {
   558тЖТ   558тЖТ        const cached = this.promptCache.get(
   559тЖТ   559тЖТ          routeDecision?.model || this.config.defaultModel,
   560тЖТ   560тЖТ          session.messages,
   561тЖТ   561тЖТ        );
   562тЖТ   562тЖТ        if (cached) {
   563тЖТ   563тЖТ          this.ui.info('Cache hit - using cached response');
   564тЖТ   564тЖТ          return {
   565тЖТ   565тЖТ            content: cached.response,
   566тЖТ   566тЖТ            usage: { inputTokens: cached.inputTokens, outputTokens: cached.outputTokens, cost: 0 },
   567тЖТ   567тЖТ          };
   568тЖТ   568тЖТ        }
   569тЖТ   569тЖТ      }
   570тЖТ   570тЖТ    }
   571тЖТ   571тЖТ
   572тЖТ   572тЖТ    // Start or get session
   573тЖТ   573тЖТ    let session = this.sessionManager.getCurrent();
   574тЖТ   574тЖТ    if (!session) {
   575тЖТ   575тЖТ      session = this.sessionManager.create(process.cwd(), this.config.defaultModel);
   576тЖТ   576тЖТ    }
   577тЖТ   577тЖТ
   578тЖТ   578тЖТ    // Add user message to session
   579тЖТ   579тЖТ    this.sessionManager.addMessage({
   580тЖТ   580тЖТ      role: 'user',
   581тЖТ   581тЖТ      content: message,
   582тЖТ   582тЖТ      timestamp: Date.now(),
   583тЖТ   583тЖТ    });
   584тЖТ   584тЖТ
   585тЖТ   585тЖТ    // Build system prompt additions from skills and styles
   586тЖТ   586тЖТ    const skillAdditions = this.skillSystem.getSystemPromptAdditions();
   587тЖТ   587тЖТ    const styleAddition = this.styleManager.getSystemPromptAddition();
   588тЖТ   588тЖТ    const thinkingAddition = this.extendedThinking.getSystemPromptAddition();
   589тЖТ   589тЖТ
   590тЖТ   590тЖТ    // Build UI callbacks
   591тЖТ   591тЖТ    const callbacks: AgentCallbacks = {
   592тЖТ   592тЖТ      onThinking: (thinking) => this.ui.thinking(thinking),
   593тЖТ   593тЖТ      onToken: (token) => this.ui.streamingToken(token),
   594тЖТ   594тЖТ      onToolCall: (name, args) => {
   595тЖТ   595тЖТ        // Check .neuroignore for file paths
   596тЖТ   596тЖТ        if (args.path && typeof args.path === 'string' && this.neuroIgnore.isIgnored(args.path)) {
   597тЖТ   597тЖТ          this.ui.warning(`Ignored path: ${args.path} (matches .neuroignore rule)`);
   598тЖТ   598тЖТ          return;
   599тЖТ   599тЖТ        }
   600тЖТ   600тЖТ        // Sandbox check before tool execution
   601тЖТ   601тЖТ        if (this.sandbox.isEnabled() && !this.checkSandboxForTool(name, args)) {
   602тЖТ   602тЖТ          return;
   603тЖТ   603тЖТ        }
   604тЖТ   604тЖТ        this.ui.toolCall(name, args);
   605тЖТ   605тЖТ
   606тЖТ   606тЖТ        // Record in undo/redo for file modification tools
   607тЖТ   607тЖТ        if (['write_file', 'edit_file', 'apply_diff', 'delete_file'].includes(name)) {
   608тЖТ   608тЖТ          // The undo/redo push will happen in the tool execution result handler
   609тЖТ   609тЖТ        }
   610тЖТ   610тЖТ      },
   611тЖТ   611тЖТ      onToolResult: (name, result, isError) => this.ui.toolResult(name, result, isError),
   612тЖТ   612тЖТ      onApprovalNeeded: async (name, args, risk) => {
   613тЖТ   613тЖТ        return this.handleApproval(name, args, risk as 'low' | 'medium' | 'high');
   614тЖТ   614тЖТ      },
   615тЖТ   615тЖТ      onIteration: (i, max) => {
   616тЖТ   616тЖТ        this.ui.info(`Iteration ${i}/${max}`);
   617тЖТ   617тЖТ      },
   618тЖТ   618тЖТ    };
   619тЖТ   619тЖТ
   620тЖТ   620тЖТ    let result;
   621тЖТ   621тЖТ    const activeModel = routeDecision?.model || this.config.defaultModel;
   622тЖТ   622тЖТ
   623тЖТ   623тЖТ    if (mode === 'direct' && targetAgent) {
   624тЖТ   624тЖТ      const agent = this.agents.get(targetAgent);
   625тЖТ   625тЖТ      if (!agent) {
   626тЖТ   626тЖТ        this.ui.error(`Agent not found: ${targetAgent}`);
   627тЖТ   627тЖТ        return { content: '', usage: { inputTokens: 0, outputTokens: 0, cost: 0 } };
   628тЖТ   628тЖТ      }
   629тЖТ   629тЖТ      this.ui.startStreaming();
   630тЖТ   630тЖТ      result = await agent.run(message, callbacks);
   631тЖТ   631тЖТ      this.ui.endStreaming();
   632тЖТ   632тЖТ    } else if (mode === 'agent') {
   633тЖТ   633тЖТ      const orchestrateResult = await this.orchestrator.orchestrate(message, callbacks);
   634тЖТ   634тЖТ      result = {
   635тЖТ   635тЖТ        content: orchestrateResult.content,
   636тЖТ   636тЖТ        toolCallsMade: 0,
   637тЖТ   637тЖТ        iterations: orchestrateResult.execution.iterations,
   638тЖТ   638тЖТ        usage: orchestrateResult.totalUsage,
   639тЖТ   639тЖТ        execution: orchestrateResult.execution,
   640тЖТ   640тЖТ      };
   641тЖТ   641тЖТ    } else {
   642тЖТ   642тЖТ      const complexity = routeDecision?.complexity || this.assessComplexity(message);
   643тЖТ   643тЖТ      if (complexity === 'simple') {
   644тЖТ   644тЖТ        const agent = this.agents.get('Coder');
   645тЖТ   645тЖТ        if (agent) {
   646тЖТ   646тЖТ          this.ui.startStreaming();
   647тЖТ   647тЖТ          result = await agent.run(message, callbacks);
   648тЖТ   648тЖТ          this.ui.endStreaming();
   649тЖТ   649тЖТ        } else {
   650тЖТ   650тЖТ          throw new Error('Coder agent not initialized');
   651тЖТ   651тЖТ        }
   652тЖТ   652тЖТ      } else {
   653тЖТ   653тЖТ        this.ui.thinking('Analyzing task complexity... Using multi-agent orchestration');
   654тЖТ   654тЖТ        const orchestrateResult = await this.orchestrator.orchestrate(message, callbacks);
   655тЖТ   655тЖТ        result = {
   656тЖТ   656тЖТ          content: orchestrateResult.content,
   657тЖТ   657тЖТ          toolCallsMade: 0,
   658тЖТ   658тЖТ          iterations: orchestrateResult.execution.iterations,
   659тЖТ   659тЖТ          usage: orchestrateResult.totalUsage,
   660тЖТ   660тЖТ          execution: orchestrateResult.execution,
   661тЖТ   661тЖТ        };
   662тЖТ   662тЖТ      }
   663тЖТ   663тЖТ    }
   664тЖТ   664тЖТ
   665тЖТ   665тЖТ    // Parse extended thinking blocks from response
   666тЖТ   666тЖТ    const thinkingResult = this.extendedThinking.parseResponse(result.content);
   667тЖТ   667тЖТ    if (thinkingResult.hadThinking && this.extendedThinking.isDisplayEnabled()) {
   668тЖТ   668тЖТ      // Thinking blocks were already displayed during streaming
   669тЖТ   669тЖТ    }
   670тЖТ   670тЖТ    // Use cleaned response (without thinking blocks) if thinking is hidden
   671тЖТ   671тЖТ    if (thinkingResult.hadThinking && !this.extendedThinking.isDisplayEnabled()) {
   672тЖТ   672тЖТ      result.content = thinkingResult.cleanedResponse;
   673тЖТ   673тЖТ    }
   674тЖТ   674тЖТ
   675тЖТ   675тЖТ    // Record spending
   676тЖТ   676тЖТ    const spendResult = this.spendingMonitor.record({
   677тЖТ   677тЖТ      model: activeModel,
   678тЖТ   678тЖТ      inputTokens: result.usage.inputTokens,
   679тЖТ   679тЖТ      outputTokens: result.usage.outputTokens,
   680тЖТ   680тЖТ      cost: result.usage.cost,
   681тЖТ   681тЖТ      sessionId: this.sessionManager.getCurrent()?.id || 'unknown',
   682тЖТ   682тЖТ    });
   683тЖТ   683тЖТ    if (spendResult.warning) {
   684тЖТ   684тЖТ      this.ui.warning(spendResult.warning);
   685тЖТ   685тЖТ    }
   686тЖТ   686тЖТ    if (!spendResult.allowed) {
   687тЖТ   687тЖТ      this.ui.error('Spending limit reached. Operation completed but further requests may be blocked.');
   688тЖТ   688тЖТ    }
   689тЖТ   689тЖТ
   690тЖТ   690тЖТ    // Cache the response if caching is enabled
   691тЖТ   691тЖТ    if (this.promptCache && this.config.promptCache.enabled && session) {
   692тЖТ   692тЖТ      this.promptCache.set(activeModel, session.messages, result.content, result.usage);
   693тЖТ   693тЖТ    }
   694тЖТ   694тЖТ
   695тЖТ   695тЖТ    // Print usage
   696тЖТ   696тЖТ    this.ui.tokenUsage(result.usage, activeModel);
   697тЖТ   697тЖТ
   698тЖТ   698тЖТ    // Update session
   699тЖТ   699тЖТ    this.sessionManager.addMessage({
   700тЖТ   700тЖТ      role: 'assistant',
   701тЖТ   701тЖТ      content: result.content,
   702тЖТ   702тЖТ      timestamp: Date.now(),
   703тЖТ   703тЖТ    });
   704тЖТ   704тЖТ    this.sessionManager.updateUsage(
   705тЖТ   705тЖТ      result.usage.inputTokens,

   718тЖТ   718тЖТ  /**
   719тЖТ   719тЖТ   * Check sandbox permissions for a tool call
   720тЖТ   720тЖТ   */
   721тЖТ   721тЖТ  private checkSandboxForTool(toolName: string, args: Record<string, unknown>): boolean {
   722тЖТ   722тЖТ    // File operations
   723тЖТ   723тЖТ    if (['write_file', 'edit_file', 'apply_diff'].includes(toolName)) {
   724тЖТ   724тЖТ      const path = args.path as string;
   725тЖТ   725тЖТ      if (path && !this.sandbox.canWrite(path)) {
   726тЖТ   726тЖТ        this.ui.warning(`Sandbox: Write access denied for ${path}`);
   727тЖТ   727тЖТ        return false;
   728тЖТ   728тЖТ      }
   729тЖТ   729тЖТ      // Check neuroignore
   730тЖТ   730тЖТ      if (path && this.neuroIgnore.isIgnored(path)) {
   731тЖТ   731тЖТ        this.ui.warning(`Ignored: ${path} is in .neuroignore`);
   732тЖТ   732тЖТ        return false;
   733тЖТ   733тЖТ      }
   734тЖТ   734тЖТ      // Backup file before modification
   735тЖТ   735тЖТ      if (path) this.sandbox.backupFile(path);
   736тЖТ   736тЖТ    }
   737тЖТ   737тЖТ
   738тЖТ   738тЖТ    if (['delete_file'].includes(toolName)) {
   739тЖТ   739тЖТ      const path = args.path as string;
   740тЖТ   740тЖТ      if (path && !this.sandbox.canDelete(path)) {
   741тЖТ   741тЖТ        this.ui.warning(`Sandbox: Delete access denied for ${path}`);
   742тЖТ   742тЖТ        return false;
   743тЖТ   743тЖТ      }
   744тЖТ   744тЖТ      if (path && this.neuroIgnore.isIgnored(path)) {
   745тЖТ   745тЖТ        this.ui.warning(`Ignored: ${path} is in .neuroignore`);
   746тЖТ   746тЖТ        return false;
   747тЖТ   747тЖТ      }
   748тЖТ   748тЖТ    }
   749тЖТ   749тЖТ
   750тЖТ   750тЖТ    if (['read_file', 'search_files', 'list_directory'].includes(toolName)) {
   751тЖТ   751тЖТ      const path = (args.path || args.directory) as string;
   752тЖТ   752тЖТ      if (path && !this.sandbox.canRead(path)) {
   753тЖТ   753тЖТ        this.ui.warning(`Sandbox: Read access denied for ${path}`);
   754тЖТ   754тЖТ        return false;
   755тЖТ   755тЖТ      }
   756тЖТ   756тЖТ      if (path && this.neuroIgnore.isIgnored(path)) {
   757тЖТ   757тЖТ        this.ui.warning(`Ignored: ${path} is in .neuroignore`);
   758тЖТ   758тЖТ        return false;
   759тЖТ   759тЖТ      }
   760тЖТ   760тЖТ    }
   761тЖТ   761тЖТ
   762тЖТ   762тЖТ    // Command execution
   763тЖТ   763тЖТ    if (['run_command', 'bash'].includes(toolName)) {
   764тЖТ   764тЖТ      const command = args.command as string;
   765тЖТ   765тЖТ      if (command && !this.sandbox.canRunCommand(command)) {
   766тЖТ   766тЖТ        this.ui.warning('Sandbox: Command execution denied');
   767тЖТ   767тЖТ        return false;
   768тЖТ   768тЖТ      }
   769тЖТ   769тЖТ    }
   770тЖТ   770тЖТ
   771тЖТ   771тЖТ    // Network access
   772тЖТ   772тЖТ    if (['web_search', 'web_fetch'].includes(toolName)) {
   773тЖТ   773тЖТ      if (!this.sandbox.canAccessNetwork()) {
   774тЖТ   774тЖТ        this.ui.warning('Sandbox: Network access denied');
   775тЖТ   775тЖТ        return false;
   776тЖТ   776тЖТ      }
   777тЖТ   777тЖТ    }
   778тЖТ   778тЖТ
   779тЖТ   779тЖТ    return true;
   780тЖТ   780тЖТ  }
