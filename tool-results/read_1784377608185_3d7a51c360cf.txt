     1→// ============================================================
     2→// NeuroCLI - Background Session Manager
     3→// Manages long-running sessions that execute independently in
     4→// the background — inspired by Claude Code background sessions
     5→// and Agent View. Supports persistence, output capture, desktop
     6→// notifications, pause/resume, and attach/detach workflows.
     7→// Uses only Node.js built-in modules.
     8→// ============================================================
     9→

    68→export interface BackgroundSessionHistoryEntry {
    69→  timestamp: number;
    70→  type: 'info' | 'iteration' | 'output' | 'error' | 'pause' | 'resume' | 'stop' | 'complete';

    75→// ---- Internal Types ----
    76→
    77→interface SessionRecord {
    78→  config: BackgroundSessionConfig;
    79→  state: BackgroundSessionState;
    80→  history: BackgroundSessionHistoryEntry[];
    81→  abortController: AbortController;
    82→  outputBuffer: string[];
    83→  attached: boolean;
    84→  /** Resolves when the session completes. */
    85→  completionPromise: Promise<void>;
    86→  completionResolve: (() => void) | null;
    87→  completionReject: ((error: Error) => void) | null;
    88→  /** Pause handle — resolves when unpaused. */
    89→  pausePromise: Promise<void> | null;
    90→  pauseResolve: (() => void) | null;

    99→const DEFAULT_MAX_COST = 5.0; // USD

   107→export class BackgroundSessionManager extends EventEmitter {
   108→  private sessions: Map<string, SessionRecord> = new Map();
   109→  private currentlyAttached: string | null = null;
   110→
   111→  constructor() {
   112→    super();
   113→    this.ensureDir(SESSION_DIR);
   114→    this.ensureDir(OUTPUT_DIR);
   115→    this.ensureDir(HISTORY_DIR);
   116→
   117→    // Restore persisted sessions on startup
   118→    this.restoreSessions();
   119→  }

   121→  // ---- Public API ----
   122→
   123→  /**
   124→   * Create a new background session. The session starts in "created" status.
   125→   * Call `startSession(sessionId)` to begin execution.
   126→   */
   127→  createSession(config: BackgroundSessionConfig): string {
   128→    if (this.sessions.has(config.id)) {
   129→      throw new Error(`Session with id "${config.id}" already exists`);
   130→    }
   131→
   132→    const outputFilePath = config.saveOutput
   133→      ? join(OUTPUT_DIR, `${config.id}.log`)
   134→      : undefined;
   135→
   136→    const state: BackgroundSessionState = {
   137→      id: config.id,
   138→      name: config.name,
   139→      status: 'created',
   140→      createdAt: Date.now(),
   141→      iterations: 0,
   142→      tokensUsed: 0,
   143→      cost: 0,
   144→      outputFilePath,
   145→      tags: config.tags ?? [],
   146→    };
   147→
   148→    let completionResolve: (() => void) | null = null;
   149→    let completionReject: ((error: Error) => void) | null = null;
   150→    const completionPromise = new Promise<void>((resolve, reject) => {
   151→      completionResolve = resolve;
   152→      completionReject = reject;
   153→    });
   154→
   155→    const record: SessionRecord = {
   156→      config,
   157→      state,
   158→      history: [],
   159→      abortController: new AbortController(),
   160→      outputBuffer: [],
   161→      attached: false,
   162→      completionPromise,
   163→      completionResolve,
   164→      completionReject,
   165→      pausePromise: null,
   166→      pauseResolve: null,
   167→    };
   168→
   169→    this.sessions.set(config.id, record);
   170→    this.addHistoryEntry(config.id, 'info', `Session "${config.name}" created`);
   171→
   172→    if (outputFilePath) {
   173→      this.initializeOutputFile(outputFilePath, config);
   174→    }
   175→
   176→    this.persistSession(config.id);
   177→    this.emit('session:created', config.id, state);
   178→
   179→    return config.id;
   180→  }
   181→
   182→  /**
   183→   * Start running a session in the background.
   184→   */
   185→  startSession(sessionId: string): boolean {
   186→    const record = this.sessions.get(sessionId);
   187→    if (!record) return false;
   188→
   189→    if (record.state.status !== 'created' && record.state.status !== 'paused') {
   190→      return false; // Can only start from 'created' or 'paused'
   191→    }
   192→
   193→    if (record.state.status === 'paused' && record.pauseResolve) {
   194→      // Unpause by resolving the pause promise
   195→      record.pauseResolve();
   196→      record.pausePromise = null;
   197→      record.pauseResolve = null;
   198→      record.state.status = 'running';
   199→      this.addHistoryEntry(sessionId, 'resume', `Session "${record.config.name}" resumed`);
   200→      this.persistSession(sessionId);
   201→      this.emit('session:resumed', sessionId, record.state);
   202→      return true;
   203→    }
   204→
   205→    // Fresh start
   206→    record.abortController = new AbortController();
   207→    record.state.status = 'running';
   208→    record.state.startedAt = Date.now();
   209→
   210→    this.addHistoryEntry(sessionId, 'info', `Session "${record.config.name}" started`);
   211→    this.persistSession(sessionId);
   212→    this.emit('session:started', sessionId, record.state);
   213→
   214→    this.runSessionLoop(record)
   215→      .then(() => {
   216→        record.state.status = 'completed';
   217→        record.state.completedAt = Date.now();
   218→        record.completionResolve?.();
   219→        this.addHistoryEntry(sessionId, 'complete', `Session completed after ${record.state.iterations} iterations`);
   220→        this.persistSession(sessionId);
   221→
   222→        if (record.config.notifyOnComplete) {
   223→          this.sendDesktopNotification(
   224→            'NeuroCLI Session Complete',
   225→            `"${record.config.name}" finished successfully (${record.state.iterations} iterations, $${record.state.cost.toFixed(4)}).`
   226→          );
   227→        }
   228→
   229→        this.emit('session:completed', sessionId, record.state);
   230→      })
   231→      .catch((err: unknown) => {
   232→        if (record.state.status === 'stopped' || record.state.status === 'paused') {
   233→          return; // Handled by stop/pause
   234→        }
   235→        const errorMsg = err instanceof Error ? err.message : String(err);
   236→        record.state.status = 'failed';
   237→        record.state.completedAt = Date.now();
   238→        record.state.lastOutput = errorMsg;
   239→        record.completionReject?.(err instanceof Error ? err : new Error(errorMsg));
   240→        this.addHistoryEntry(sessionId, 'error', `Session failed: ${errorMsg}`);
   241→        this.persistSession(sessionId);
   242→
   243→        if (record.config.notifyOnComplete) {
   244→          this.sendDesktopNotification(
   245→            'NeuroCLI Session Failed',
   246→            `"${record.config.name}" encountered an error: ${errorMsg}`
   247→          );
   248→        }
   249→
   250→        this.emit('session:failed', sessionId, errorMsg, record.state);
   251→      });
   252→
   253→    return true;
   254→  }
   255→
   256→  /**
   257→   * Stop a running session. Cannot be resumed.
   258→   */
   259→  stopSession(sessionId: string): boolean {
   260→    const record = this.sessions.get(sessionId);
   261→    if (!record) return false;
   262→
   263→    if (record.state.status !== 'running') return false;
   264→
   265→    record.abortController.abort();
   266→    record.state.status = 'stopped';
   267→    record.state.completedAt = Date.now();
   268→
   269→    // If paused, also resolve the pause so the loop can exit
   270→    if (record.pauseResolve) {
   271→      record.pauseResolve();
   272→      record.pausePromise = null;
   273→      record.pauseResolve = null;
   274→    }
   275→
   276→    this.addHistoryEntry(sessionId, 'stop', `Session "${record.config.name}" stopped by user`);
   277→    this.persistSession(sessionId);
   278→    this.emit('session:stopped', sessionId, record.state);
   279→
   280→    return true;
   281→  }
   282→
   283→  /**
   284→   * Pause a running session. It can be resumed later.
   285→   */
   286→  pauseSession(sessionId: string): boolean {
   287→    const record = this.sessions.get(sessionId);
   288→    if (!record) return false;
   289→
   290→    if (record.state.status !== 'running') return false;
   291→
   292→    // Set up a pause barrier that the loop will wait on
   293→    let pauseResolve: (() => void) | null = null;
   294→    const pausePromise = new Promise<void>((resolve) => {
   295→      pauseResolve = resolve;
   296→    });
   297→
   298→    record.pausePromise = pausePromise;
   299→    record.pauseResolve = pauseResolve;
   300→    record.state.status = 'paused';
   301→
   302→    this.addHistoryEntry(sessionId, 'pause', `Session "${record.config.name}" paused by user`);
   303→    this.persistSession(sessionId);
   304→    this.emit('session:paused', sessionId, record.state);
   305→
   306→    return true;
   307→  }
   308→
   309→  /**
   310→   * Resume a paused session.
   311→   */
   312→  resumeSession(sessionId: string): boolean {
   313→    const record = this.sessions.get(sessionId);
   314→    if (!record) return false;
   315→    if (record.state.status !== 'paused') return false;
   316→
   317→    return this.startSession(sessionId); // startSession handles unpause
   318→  }

   323→  getSessionStatus(sessionId: string): BackgroundSessionState | null {
   324→    const record = this.sessions.get(sessionId);
   325→    if (!record) return null;
   326→    return { ...record.state };

   329→  /**
   330→   * List all background sessions.
   331→   */
   332→  listSessions(): BackgroundSessionState[] {
   333→    const result: BackgroundSessionState[] = [];
   334→    for (const record of this.sessions.values()) {
   335→      result.push({ ...record.state });
   336→    }
   337→    return result.sort((a, b) => b.createdAt - a.createdAt);
   338→  }

   340→  /**
   341→   * Get recent output from a session.
   342→   * @param lines Number of recent lines to return (default 50).
   343→   */
   344→  getSessionOutput(sessionId: string, lines: number = 50): BackgroundSessionOutput | null {
   345→    const record = this.sessions.get(sessionId);
   346→    if (!record) return null;
   347→
   348→    // If we have an output file, read from it
   349→    if (record.state.outputFilePath && existsSync(record.state.outputFilePath)) {
   350→      try {
   351→        const content = readFileSync(record.state.outputFilePath, 'utf-8');
   352→        const allLines = content.split('\n').filter(Boolean);
   353→        const fetchedLines = allLines.slice(-lines);
   354→        return {
   355→          sessionId,
   356→          lines: fetchedLines,
   357→          totalLines: allLines.length,
   358→          fetchedAt: Date.now(),
   359→        };
   360→      } catch {
   361→        // Fall through to buffer
   362→      }
   363→    }
   364→
   365→    // Fall back to in-memory buffer
   366→    const fetchedLines = record.outputBuffer.slice(-lines);
   367→    return {
   368→      sessionId,
   369→      lines: fetchedLines,
   370→      totalLines: record.outputBuffer.length,
   371→      fetchedAt: Date.now(),
   372→    };
   373→  }

   375→  /**
   376→   * Attach to a background session for interactive monitoring.
   377→   * Only one session can be attached at a time.
   378→   */
   379→  attachSession(sessionId: string): boolean {
   380→    const record = this.sessions.get(sessionId);
   381→    if (!record) return false;
   382→
   383→    // Detach from any currently attached session
   384→    if (this.currentlyAttached && this.currentlyAttached !== sessionId) {
   385→      this.detachSession();
   386→    }
   387→
   388→    record.attached = true;
   389→    this.currentlyAttached = sessionId;
   390→    this.addHistoryEntry(sessionId, 'info', `User attached to session`);
   391→    this.emit('session:attached', sessionId);
   392→
   393→    return true;
   394→  }
   395→
   396→  /**
   397→   * Detach from the currently attached session.
   398→   */
   399→  detachSession(): boolean {
   400→    if (!this.currentlyAttached) return false;
   401→
   402→    const record = this.sessions.get(this.currentlyAttached);
   403→    if (record) {
   404→      record.attached = false;
   405→      this.addHistoryEntry(this.currentlyAttached, 'info', `User detached from session`);
   406→    }
   407→
   408→    this.emit('session:detached', this.currentlyAttached);
   409→    this.currentlyAttached = null;
   410→    return true;
   411→  }

   413→  /**
   414→   * Wait for a session to complete.
   415→   */
   416→  async waitForSession(sessionId: string): Promise<void> {
   417→    const record = this.sessions.get(sessionId);
   418→    if (!record) {
   419→      throw new Error(`Session "${sessionId}" not found`);
   420→    }
   421→
   422→    if (record.state.status === 'completed') return;
   423→    if (record.state.status === 'failed') {
   424→      throw new Error(record.state.lastOutput ?? `Session "${sessionId}" failed`);
   425→    }
   426→    if (record.state.status === 'stopped') {
   427→      throw new Error(`Session "${sessionId}" was stopped`);
   428→    }
   429→
   430→    return record.completionPromise;
   431→  }

   434→   * Get the full execution history of a session.

   436→  getSessionHistory(sessionId: string): BackgroundSessionHistoryEntry[] | null {
   437→    const record = this.sessions.get(sessionId);
   438→    if (!record) return null;
   439→    return [...record.history];
   440→  }

   442→  /**
   443→   * Clean up old completed/failed/stopped sessions older than
   444→   * the specified age (default 7 days).
   445→   */
   446→  cleanup(maxAgeMs: number = CLEANUP_AGE_MS): number {
   447→    const cutoff = Date.now() - maxAgeMs;
   448→    let removed = 0;
   449→
   450→    const toRemove: string[] = [];
   451→    for (const [id, record] of this.sessions) {
   452→      const isTerminal =
   453→        record.state.status === 'completed' ||
   454→        record.state.status === 'failed' ||
   455→        record.state.status === 'stopped';
   456→
   457→      if (!isTerminal) continue;
   458→
   459→      const endTime = record.state.completedAt ?? record.state.createdAt;
   460→      if (endTime < cutoff) {
   461→        toRemove.push(id);
   462→      }
   463→    }
   464→
   465→    for (const id of toRemove) {
   466→      this.removeSession(id);
   467→      removed++;
   468→    }
   469→
   470→    return removed;
   471→  }

   493→  // ---- Private: Session Execution Loop ----
   494→
   495→  /**
   496→   * The main execution loop for a background session.
   497→   *
   498→   * In production, this would call an LLM, execute tools, handle
   499→   * approvals, etc. Here we simulate with iteration tracking,
   500→   * cost accounting, pause/resume, abort, and output capture.
   501→   */
   502→  private async runSessionLoop(record: SessionRecord): Promise<void> {
   503→    const maxIterations = record.config.maxIterations || DEFAULT_MAX_ITERATIONS;
   504→    const maxCost = record.config.maxCost || DEFAULT_MAX_COST;
   505→    const abortSignal = record.abortController.signal;
   506→
   507→    for (let i = 0; i < maxIterations; i++) {
   508→      // ---- Abort check ----
   509→      if (abortSignal.aborted) {
   510→        throw new Error('Session aborted');
   511→      }
   512→
   513→      // ---- Pause check ----
   514→      if (record.pausePromise) {
   515→        this.addHistoryEntry(record.config.id, 'pause', `Paused at iteration ${i}`);
   516→        await record.pausePromise;
   517→        // After resuming, check abort again
   518→        if (abortSignal.aborted) {
   519→          throw new Error('Session aborted after resume');
   520→        }
   521→      }
   522→
   523→      // ---- Simulate iteration ----
   524→      const iterationTokens = 50 + Math.floor(Math.random() * 200);
   525→      const iterationCost = (iterationTokens / 1_000_000) * 3.0; // simulated $3/1M tokens
   526→
   527→      record.state.iterations = i + 1;
   528→      record.state.tokensUsed += iterationTokens;
   529→      record.state.cost += iterationCost;
   530→
   531→      const outputLine = `[iter ${i + 1}] Processing... (tokens: ${record.state.tokensUsed}, cost: $${record.state.cost.toFixed(4)})`;
   532→      record.state.lastOutput = outputLine;
   533→
   534→      // Capture output
   535→      this.captureOutput(record, outputLine);
   536→
   537→      // Add history entry every 10 iterations to avoid bloat
   538→      if ((i + 1) % 10 === 0 || i === 0) {
   539→        this.addHistoryEntry(
   540→          record.config.id,
   541→          'iteration',
   542→          `Iteration ${i + 1}/${maxIterations}`,
   543→          { tokensUsed: record.state.tokensUsed, cost: record.state.cost }
   544→        );
   545→      }
   546→
   547→      // ---- Cost limit check ----
   548→      if (record.state.cost >= maxCost) {
   549→        const msg = `Cost limit ($${maxCost}) reached at iteration ${i + 1}`;
   550→        this.captureOutput(record, msg);
   551→        this.addHistoryEntry(record.config.id, 'info', msg);
   552→        break;
   553→      }
   554→
   555→      // Persist state periodically (every 5 iterations)
   556→      if ((i + 1) % 5 === 0) {
   557→        this.persistSession(record.config.id);
   558→      }
   559→
   560→      // Yield control for cooperative multitasking
   561→      await this.yieldControl();
   562→    }
   563→
   564→    // Final persist
   565→    this.persistSession(record.config.id);
   566→  }
   567→
   568→  /**
   569→   * Yield control to the event loop.
   570→   */
   571→  private yieldControl(): Promise<void> {
   572→    return new Promise((resolve) => setImmediate(resolve));
   573→  }

   575→  // ---- Private: Output Management ----
   576→
   577→  /**
   578→   * Capture an output line to both the in-memory buffer and
   579→   * the output file (if configured).
   580→   */
   581→  private captureOutput(record: SessionRecord, line: string): void {
   582→    // In-memory buffer (bounded)
   583→    record.outputBuffer.push(line);
   584→    if (record.outputBuffer.length > OUTPUT_BUFFER_MAX) {
   585→      record.outputBuffer.shift();
   586→    }
   587→
   588→    // File output
   589→    if (record.state.outputFilePath) {
   590→      try {
   591→        appendFileSync(record.state.outputFilePath, line + '\n', 'utf-8');
   592→      } catch {
   593→        // Non-critical — best effort file I/O
   594→      }
   595→    }
   596→  }

   598→  /**
   599→   * Initialize an output file with a header.
   600→   */
   601→  private initializeOutputFile(filePath: string, config: BackgroundSessionConfig): void {
   602→    try {
   603→      const header = [
   604→        `=== NeuroCLI Background Session Output ===`,
   605→        `Session: ${config.name} (${config.id})`,
   606→        `Model: ${config.model ?? 'default'}`,
   607→        `Working Directory: ${config.workingDirectory}`,
   608→        `Max Iterations: ${config.maxIterations}`,
   609→        `Max Cost: $${config.maxCost}`,
   610→        `Started: ${new Date().toISOString()}`,
   611→        `==========================================\n`,
   612→      ].join('\n');
   613→
   614→      writeFileSync(filePath, header, 'utf-8');
   615→    } catch {
   616→      // Non-critical
   617→    }
   618→  }

   625→  private addHistoryEntry(
   626→    sessionId: string,
   627→    type: BackgroundSessionHistoryEntry['type'],
   628→    message: string,
   629→    data?: Record<string, unknown>
   630→  ): void {

   634→    record.history.push({
   635→      timestamp: Date.now(),
   636→      type,
   637→      message,
   638→      data,

   643→      timestamp: Date.now(),
   644→      type,
   645→      message,
   646→      data,
   647→    });
   648→  }

   650→  // ---- Private: Persistence ----
   651→
   652→  /**
   653→   * Persist session state to disk so it survives app restarts.
   654→   */
   655→  private persistSession(sessionId: string): void {
   656→    const record = this.sessions.get(sessionId);
   657→    if (!record) return;
   658→
   659→    const filePath = join(SESSION_DIR, `${sessionId}.json`);
   660→    try {
   661→      const data = {
   662→        config: record.config,
   663→        state: record.state,
   664→        // Don't persist runtime objects like AbortController or promises
   665→        persistedAt: Date.now(),
   666→      };
   667→      writeFileSync(filePath, JSON.stringify(data, null, 2), 'utf-8');
   668→    } catch {
   669→      // Non-critical — persistence is best-effort
   670→    }
   671→  }
   672→
   673→  /**
   674→   * Persist a single history entry to the session's history file.
   675→   */
   676→  private persistHistory(sessionId: string, entry: BackgroundSessionHistoryEntry): void {
   677→    const filePath = join(HISTORY_DIR, `${sessionId}.jsonl`);
   678→    try {
   679→      appendFileSync(filePath, JSON.stringify(entry) + '\n', 'utf-8');
   680→    } catch {
   681→      // Non-critical
   682→    }
   683→  }
   684→
   685→  /**
   686→   * Restore sessions from disk on startup.
   687→   * Only restores terminal-state sessions; running sessions
   688→   * from a previous process are marked as 'stopped'.
   689→   */
   690→  private restoreSessions(): void {
   691→    if (!existsSync(SESSION_DIR)) return;
   692→
   693→    try {
   694→      for (const file of readdirSync(SESSION_DIR)) {
   695→        if (!file.endsWith('.json')) continue;
   696→
   697→        const filePath = join(SESSION_DIR, file);
   698→        try {
   699→          const content = readFileSync(filePath, 'utf-8');
   700→          const data = JSON.parse(content) as {
   701→            config: BackgroundSessionConfig;
   702→            state: BackgroundSessionState;
   703→          };
   704→
   705→          // If the session was running when the app exited, mark it stopped
   706→          if (data.state.status === 'running' || data.state.status === 'paused') {
   707→            data.state.status = 'stopped';
   708→            data.state.completedAt = data.state.completedAt ?? Date.now();
   709→          }
   710→
   711→          // Re-create a minimal record (no running loop)
   712→          let completionResolve: (() => void) | null = null;
   713→          let completionReject: ((error: Error) => void) | null = null;
   714→          const completionPromise = new Promise<void>((resolve, reject) => {
   715→            completionResolve = resolve;
   716→            completionReject = reject;
   717→          });
   718→
   719→          // Resolve immediately for terminal-state sessions
   720→          if (
   721→            data.state.status === 'completed' ||
   722→            data.state.status === 'failed' ||
   723→            data.state.status === 'stopped'
   724→          ) {
   725→            // Pre-resolved
   726→            completionPromise.catch(() => {}); // swallow
   727→          }
   728→
   729→          // Restore history from disk if available
   730→          const history = this.loadHistoryFromDisk(data.config.id);
   731→
   732→          const record: SessionRecord = {
   733→            config: data.config,
   734→            state: data.state,
   735→            history,
   736→            abortController: new AbortController(),
   737→            outputBuffer: [],
   738→            attached: false,
   739→            completionPromise,
   740→            completionResolve,
   741→            completionReject,
   742→            pausePromise: null,
   743→            pauseResolve: null,
   744→          };
   745→
   746→          this.sessions.set(data.config.id, record);
   747→        } catch {
   748→          // Corrupted file — skip
   749→        }
   750→      }
   751→    } catch {
   752→      // Session directory read failure — skip
   753→    }
   754→  }
   755→
   756→  /**
   757→   * Load history entries from the JSONL file on disk.
   758→   */
   759→  private loadHistoryFromDisk(sessionId: string): BackgroundSessionHistoryEntry[] {
   760→    const filePath = join(HISTORY_DIR, `${sessionId}.jsonl`);
   761→    if (!existsSync(filePath)) return [];
   762→
   763→    const entries: BackgroundSessionHistoryEntry[] = [];
   764→    try {
   765→      const content = readFileSync(filePath, 'utf-8');
   766→      for (const line of content.split('\n')) {
   767→        if (!line.trim()) continue;
   768→        try {
   769→          entries.push(JSON.parse(line));
   770→        } catch {
   771→          // Skip malformed lines
   772→        }
   773→      }
   774→    } catch {
   775→      // Read failure — return what we have
   776→    }
   777→
   778→    return entries;
   779→  }
   780→
   781→  /**
   782→   * Remove a session and its associated files.
   783→   */
   784→  private removeSession(sessionId: string): void {
   785→    const record = this.sessions.get(sessionId);
   786→    if (!record) return;
   787→
   788→    // Remove session file
   789→    const sessionFile = join(SESSION_DIR, `${sessionId}.json`);
   790→    try { if (existsSync(sessionFile)) unlinkSync(sessionFile); } catch { /* ignore */ }
   791→
   792→    // Remove history file
   793→    const historyFile = join(HISTORY_DIR, `${sessionId}.jsonl`);
   794→    try { if (existsSync(historyFile)) unlinkSync(historyFile); } catch { /* ignore */ }
   795→
   796→    // Remove output file
   797→    if (record.state.outputFilePath) {
   798→      try { if (existsSync(record.state.outputFilePath)) unlinkSync(record.state.outputFilePath); } catch { /* ignore */ }
   799→    }
   800→
   801→    // Detach if currently attached
   802→    if (this.currentlyAttached === sessionId) {
   803→      this.currentlyAttached = null;
   804→    }
   805→
   806→    this.sessions.delete(sessionId);
   807→    this.emit('session:removed', sessionId);
   808→  }

   812→  /**
   813→   * Send a desktop notification using the operating system's
   814→   * native notification mechanism.
   815→   */
   816→  private sendDesktopNotification(title: string, body: string): void {
   817→    try {
   818→      const platform = process.platform;
   819→
   820→      if (platform === 'darwin') {
   821→        // macOS: use osascript to display a notification
   822→        const escapedTitle = title.replace(/"/g, '\\"');
   823→        const escapedBody = body.replace(/"/g, '\\"');
   824→        spawn('osascript', [
   825→          '-e',
   826→          `display notification "${escapedBody}" with title "${escapedTitle}"`,
   827→        ]).unref();
   828→      } else if (platform === 'linux') {
   829→        // Linux: try notify-send
   830→        spawn('notify-send', [title, body]).unref();
   831→      } else if (platform === 'win32') {
   832→        // Windows: use PowerShell toast
   833→        const escapedTitle = title.replace(/'/g, "''");
   834→        const escapedBody = body.replace(/'/g, "''");
   835→        const psCmd = `
   836→          Add-Type -AssemblyName System.Windows.Forms
   837→          $notify = New-Object System.Windows.Forms.NotifyIcon
   838→          $notify.Icon = [System.Drawing.SystemIcons]::Information
   839→          $notify.Visible = $true
   840→          $notify.ShowBalloonTip(5000, '${escapedTitle}', '${escapedBody}', [System.Windows.Forms.ToolTipIcon]::Info)
   841→        `.trim();
   842→        spawn('powershell', ['-NoProfile', '-Command', psCmd], {
   843→          windowsHide: true,
   844→        }).unref();
   845→      }
   846→      // If platform is unsupported, silently skip notification
   847→    } catch {
   848→      // Notification failure must never crash the manager
   849→    }
   850→  }
