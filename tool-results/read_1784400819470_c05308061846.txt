   760→   760→  /**
   761→   761→   * Process a user message
   762→   762→   */
   763→   763→  async processMessage(
   764→   764→    message: string,
   765→   765→    mode: 'auto' | 'agent' | 'direct' = 'auto',
   766→   766→    targetAgent?: string,
   767→   767→  ): Promise<{ content: string; usage: TokenUsage; execution?: AgentExecution }> {
   768→   768→    // Check spending limit via spending monitor
   769→   769→    const limitCheck = this.spendingMonitor.checkLimit();
   770→   770→    if (!limitCheck.allowed) {
   771→   771→      this.ui.error(`Spending limit reached (${limitCheck.limitReached}). Use /spending to check or /config to adjust.`);
   772→   772→      return { content: 'Spending limit reached.', usage: { inputTokens: 0, outputTokens: 0, cost: 0 } };
   773→   773→    }
   774→   774→
   775→   775→    // Auto-activate skills based on prompt
   776→   776→    const activatedSkills = this.skillSystem.autoActivate(message);
   777→   777→    if (activatedSkills.length > 0) {
   778→   778→      for (const skill of activatedSkills) {
   779→   779→        this.ui.info(`Skill activated: ${skill.skill.name} (${skill.activatedBy})`);
   780→   780→      }
   781→   781→    }
   782→   782→
   783→   783→    // Model routing (if auto mode)
   784→   784→    let routeDecision: RoutingDecision | null = null;
   785→   785→    if (mode === 'auto') {
   786→   786→      routeDecision = this.modelRouter.route(message);
   787→   787→      if (routeDecision.model !== this.config.defaultModel) {
   788→   788→        this.ui.info(`Model router: ${routeDecision.complexity} task -> ${MODELS[routeDecision.model]?.name || routeDecision.model}`);
   789→   789→      }
   790→   790→    }
   791→   791→
   792→   792→    // Check prompt cache
   793→   793→    if (this.promptCache && this.config.promptCache.enabled) {
   794→   794→      const session = this.sessionManager.getCurrent();
   795→   795→      if (session) {
   796→   796→        const cached = this.promptCache.get(
   797→   797→          routeDecision?.model || this.config.defaultModel,
   798→   798→          session.messages,
   799→   799→        );
   800→   800→        if (cached) {
   801→   801→          this.ui.info('Cache hit - using cached response');
   802→   802→          return {
   803→   803→            content: cached.response,
   804→   804→            usage: { inputTokens: cached.inputTokens, outputTokens: cached.outputTokens, cost: 0 },
   805→   805→          };
   806→   806→        }
   807→   807→      }
   808→   808→    }
   809→   809→
   810→   810→    // Start or get session
   811→   811→    let session = this.sessionManager.getCurrent();
   812→   812→    if (!session) {
   813→   813→      session = this.sessionManager.create(process.cwd(), this.config.defaultModel);
   814→   814→    }
   815→   815→
   816→   816→    // Add user message to session
   817→   817→    this.sessionManager.addMessage({
   818→   818→      role: 'user',
   819→   819→      content: message,
   820→   820→      timestamp: Date.now(),
   821→   821→    });
   822→   822→
   823→   823→    // Build system prompt additions from skills and styles
   824→   824→    const skillAdditions = this.skillSystem.getSystemPromptAdditions();
   825→   825→    const styleAddition = this.styleManager.getSystemPromptAddition();
   826→   826→    const thinkingAddition = this.extendedThinking.getSystemPromptAddition();
   827→   827→
   828→   828→    // Build UI callbacks
   829→   829→    const callbacks: AgentCallbacks = {
   830→   830→      onThinking: (thinking) => this.ui.thinking(thinking),
   831→   831→      onToken: (token) => this.ui.streamingToken(token),
   832→   832→      onToolCall: (name, args) => {
   833→   833→        // Check .neuroignore for file paths
   834→   834→        if (args.path && typeof args.path === 'string' && this.neuroIgnore.isIgnored(args.path)) {
   835→   835→          this.ui.warning(`Ignored path: ${args.path} (matches .neuroignore rule)`);
   836→   836→          return;
   837→   837→        }
   838→   838→        // Sandbox check before tool execution
   839→   839→        if (this.sandbox.isEnabled() && !this.checkSandboxForTool(name, args)) {
   840→   840→          return;
   841→   841→        }
   842→   842→        this.ui.toolCall(name, args);
   843→   843→
   844→   844→        // Record in undo/redo for file modification tools
   845→   845→        if (['write_file', 'edit_file', 'apply_diff', 'delete_file'].includes(name)) {
   846→   846→          // The undo/redo push will happen in the tool execution result handler
   847→   847→        }
   848→   848→      },
   849→   849→      onToolResult: (name, result, isError) => this.ui.toolResult(name, result, isError),
   850→   850→      onApprovalNeeded: async (name, args, risk) => {
   851→   851→        return this.handleApproval(name, args, risk as 'low' | 'medium' | 'high');
   852→   852→      },
   853→   853→      onIteration: (i, max) => {
   854→   854→        this.ui.info(`Iteration ${i}/${max}`);
   855→   855→      },
   856→   856→    };
   857→   857→
   858→   858→    let result;
   859→   859→    const activeModel = routeDecision?.model || this.config.defaultModel;
   860→   860→
   861→   861→    if (mode === 'direct' && targetAgent) {
   862→   862→      const agent = this.agents.get(targetAgent);
   863→   863→      if (!agent) {
   864→   864→        this.ui.error(`Agent not found: ${targetAgent}`);
   865→   865→        return { content: '', usage: { inputTokens: 0, outputTokens: 0, cost: 0 } };
   866→   866→      }
   867→   867→      // Override agent model if defaultModel differs from agent's model
   868→   868→      const originalModel = agent.configModel;
   869→   869→      if (this.config.defaultModel !== originalModel) {
   870→   870→        agent.configModel = this.config.defaultModel;
   871→   871→      }
   872→   872→      this.ui.startStreaming();
   873→   873→      try {
   874→   874→        result = await agent.run(message, callbacks);
   875→   875→      } finally {
   876→   876→        this.ui.endStreaming();
   877→   877→      }
   878→   878→      // Restore original model
   879→   879→      agent.configModel = originalModel;
   880→   880→    } else if (mode === 'agent') {
   881→   881→      const orchestrateResult = await this.orchestrator.orchestrate(message, callbacks);
   882→   882→      result = {
   883→   883→        content: orchestrateResult.content,
   884→   884→        toolCallsMade: 0,
   885→   885→        iterations: orchestrateResult.execution.iterations,
   886→   886→        usage: orchestrateResult.totalUsage,
   887→   887→        execution: orchestrateResult.execution,
   888→   888→      };
   889→   889→    } else {
   890→   890→      const complexity = routeDecision?.complexity || this.assessComplexity(message);
   891→   891→      const category = routeDecision?.category || this.modelRouter.getCategory(message);
   892→   892→
   893→   893→      if (complexity === 'simple') {
   894→   894→        // Route to the best agent based on task category
   895→   895→        const targetAgentName = this.selectAgentForCategory(category);
   896→   896→        const agent = this.agents.get(targetAgentName);
   897→   897→        if (agent) {
   898→   898→          this.ui.info(`Using ${targetAgentName} agent (${category} task)`);
   899→   899→          this.ui.startStreaming();
   900→   900→          try {
   901→   901→            result = await agent.run(message, callbacks);
   902→   902→          } finally {
   903→   903→            this.ui.endStreaming();
   904→   904→          }
   905→   905→        } else {
   906→   906→          // Fallback to Coder
   907→   907→          const coderAgent = this.agents.get('Coder');
   908→   908→          if (coderAgent) {
   909→   909→            this.ui.startStreaming();
   910→   910→            try {
   911→   911→              result = await coderAgent.run(message, callbacks);
   912→   912→            } finally {
   913→   913→              this.ui.endStreaming();
   914→   914→            }
   915→   915→          } else {
   916→   916→            this.ui.warning('No agents initialized, using orchestration mode');
   917→   917→            const orchestrateResult = await this.orchestrator.orchestrate(message, callbacks);
   918→   918→            result = {
   919→   919→              content: orchestrateResult.content,
   920→   920→              toolCallsMade: 0,
   921→   921→              iterations: orchestrateResult.execution.iterations,
   922→   922→              usage: orchestrateResult.totalUsage,
   923→   923→              execution: orchestrateResult.execution,
   924→   924→            };
   925→   925→          }
   926→   926→        }
   927→   927→      } else {
   928→   928→        this.ui.thinking('Analyzing task complexity... Using multi-agent orchestration');
   929→   929→        const orchestrateResult = await this.orchestrator.orchestrate(message, callbacks);
   930→   930→        result = {
   931→   931→          content: orchestrateResult.content,
   932→   932→          toolCallsMade: 0,
   933→   933→          iterations: orchestrateResult.execution.iterations,
   934→   934→          usage: orchestrateResult.totalUsage,
   935→   935→          execution: orchestrateResult.execution,
   936→   936→        };
   937→   937→      }
   938→   938→    }
   939→   939→
   940→   940→    // Parse extended thinking blocks from response
   941→   941→    const thinkingResult = this.extendedThinking.parseResponse(result.content);
   942→   942→    if (thinkingResult.hadThinking && this.extendedThinking.isDisplayEnabled()) {
   943→   943→      // Thinking blocks were already displayed during streaming
   944→   944→    }
   945→   945→    // Use cleaned response (without thinking blocks) if thinking is hidden
   946→   946→    if (thinkingResult.hadThinking && !this.extendedThinking.isDisplayEnabled()) {
   947→   947→      result.content = thinkingResult.cleanedResponse;
   948→   948→    }
   949→   949→
   950→   950→    // Record spending
   951→   951→    const spendResult = this.spendingMonitor.record({
   952→   952→      model: activeModel,
   953→   953→      inputTokens: result.usage.inputTokens,
   954→   954→      outputTokens: result.usage.outputTokens,
   955→   955→      cost: result.usage.cost,
   956→   956→      sessionId: this.sessionManager.getCurrent()?.id || 'unknown',
   957→   957→    });
   958→   958→    if (spendResult.warning) {
   959→   959→      this.ui.warning(spendResult.warning);
   960→   960→    }
   961→   961→    if (!spendResult.allowed) {
   962→   962→      this.ui.error('Spending limit reached. Operation completed but further requests may be blocked.');
   963→   963→    }
   964→   964→
   965→   965→    // Cache the response if caching is enabled
   966→   966→    if (this.promptCache && this.config.promptCache.enabled && session) {
   967→   967→      this.promptCache.set(activeModel, session.messages, result.content, result.usage);
   968→   968→    }
   969→   969→
   970→   970→    // Print usage
   971→   971→    this.ui.tokenUsage(result.usage, activeModel);
   972→   972→
   973→   973→    // Update session
   974→   974→    this.sessionManager.addMessage({
   975→   975→      role: 'assistant',
   976→   976→      content: result.content,
   977→   977→      timestamp: Date.now(),
   978→   978→    });
   979→   979→    this.sessionManager.updateUsage(
   980→   980→      result.usage.inputTokens,
   981→   981→      result.usage.outputTokens,
   982→   982→      result.usage.cost,
   983→   983→    );
   984→   984→    this.sessionManager.save();
   985→   985→
   986→   986→    return {
   987→   987→      content: result.content,
   988→   988→      usage: result.usage,
   989→   989→      execution: result.execution,
   990→   990→    };
   991→   991→  }

   993→   993→  /**
   994→   994→   * Check sandbox permissions for a tool call
   995→   995→   */
   996→   996→  private checkSandboxForTool(toolName: string, args: Record<string, unknown>): boolean {
   997→   997→    // File operations
   998→   998→    if (['write_file', 'edit_file', 'apply_diff'].includes(toolName)) {
   999→   999→      const path = args.path as string;
  1000→  1000→      if (path && !this.sandbox.canWrite(path)) {
  1001→  1001→        this.ui.warning(`Sandbox: Write access denied for ${path}`);
  1002→  1002→        return false;
  1003→  1003→      }
  1004→  1004→      // Check neuroignore
  1005→  1005→      if (path && this.neuroIgnore.isIgnored(path)) {
  1006→  1006→        this.ui.warning(`Ignored: ${path} is in .neuroignore`);
  1007→  1007→        return false;
  1008→  1008→      }
  1009→  1009→      // Backup file before modification
  1010→  1010→      if (path) this.sandbox.backupFile(path);
  1011→  1011→    }
  1012→  1012→
  1013→  1013→    if (['delete_file'].includes(toolName)) {
  1014→  1014→      const path = args.path as string;
  1015→  1015→      if (path && !this.sandbox.canDelete(path)) {
  1016→  1016→        this.ui.warning(`Sandbox: Delete access denied for ${path}`);
  1017→  1017→        return false;
  1018→  1018→      }
  1019→  1019→      if (path && this.neuroIgnore.isIgnored(path)) {
  1020→  1020→        this.ui.warning(`Ignored: ${path} is in .neuroignore`);
  1021→  1021→        return false;
  1022→  1022→      }
  1023→  1023→    }
  1024→  1024→
  1025→  1025→    if (['read_file', 'search_files', 'list_directory'].includes(toolName)) {
  1026→  1026→      const path = (args.path || args.directory) as string;
  1027→  1027→      if (path && !this.sandbox.canRead(path)) {
  1028→  1028→        this.ui.warning(`Sandbox: Read access denied for ${path}`);
  1029→  1029→        return false;
  1030→  1030→      }
  1031→  1031→      if (path && this.neuroIgnore.isIgnored(path)) {
  1032→  1032→        this.ui.warning(`Ignored: ${path} is in .neuroignore`);
  1033→  1033→        return false;
  1034→  1034→      }
  1035→  1035→    }
  1036→  1036→
  1037→  1037→    // Command execution
  1038→  1038→    if (['run_command', 'bash'].includes(toolName)) {
  1039→  1039→      const command = args.command as string;
  1040→  1040→      if (command && !this.sandbox.canRunCommand(command)) {
  1041→  1041→        this.ui.warning('Sandbox: Command execution denied');
  1042→  1042→        return false;
  1043→  1043→      }
  1044→  1044→    }
  1045→  1045→
  1046→  1046→    // Network access
  1047→  1047→    if (['web_search', 'web_fetch'].includes(toolName)) {
  1048→  1048→      if (!this.sandbox.canAccessNetwork()) {
  1049→  1049→        this.ui.warning('Sandbox: Network access denied');
  1050→  1050→        return false;
  1051→  1051→      }
  1052→  1052→    }
  1053→  1053→
  1054→  1054→    return true;
  1055→  1055→  }
