   257→// ---- Interactive Mode ----
   258→async function startInteractive(options: any) {
   259→  const config = initConfig(options.apiKey);
   260→  if (options.model) config.defaultModel = options.model;
   261→  if (options.theme) config.ui.theme = options.theme as 'dark' | 'light' | 'dracula' | 'nord';
   262→  if (options.noStreaming) config.ui.streaming = false;
   263→  if (options.permissionMode) config.permissionMode = options.permissionMode;
   264→  if (options.diffPreview === true) config.diffPreview = true;
   265→  if (options.diffPreview === false) config.diffPreview = false;
   266→  if (options.cache === true) config.promptCache.enabled = true;
   267→  if (options.cache === false) config.promptCache.enabled = false;
   268→  if (options.spendingLimit) config.spendingLimit = options.spendingLimit;
   269→  if (options.autoApprove) {
   270→    config.tools.autoApprove = [...config.tools.autoApprove, ...config.tools.requireApproval];
   271→    config.tools.requireApproval = [];
   272→    config.permissionMode = 'yolo';
   273→  }
   274→
   275→  // Check API key
   276→  if (!config.apiKey && !options.ollama) {
   277→    console.log(chalk.red('\nOpenRouter API key not configured!'));
   278→    console.log(chalk.yellow('\nSet it with:'));
   279→    console.log(`  ${chalk.cyan('neuro config --set-key YOUR_API_KEY')}`);
   280→    console.log(`  ${chalk.cyan('export OPENROUTER_API_KEY=YOUR_API_KEY')}`);
   281→    console.log(`  ${chalk.cyan('neuro -k YOUR_API_KEY')}`);
   282→    console.log(chalk.gray('\nGet your key at: https://openrouter.ai/keys\n'));
   283→    process.exit(1);
   284→  }
   285→
   286→  // Initialize engine
   287→  const engine = new NeuroEngine(config);
   288→
   289→  // Set effort level if specified
   290→  if (options.effort) {
   291→    engine.modelRouter.setEffort(options.effort);
   292→  }
   293→
   294→  // Set output style if specified
   295→  if (options.style) {
   296→    engine.styleManager.setStyle(options.style);
   297→  }
   298→
   299→  // Enable thinking if specified
   300→  if (options.thinking) {
   301→    engine.extendedThinking.setMode('full');
   302→    engine.extendedThinking.toggleDisplay();
   303→  }
   304→
   305→  // Enable sandbox if specified
   306→  if (options.sandbox) {
   307→    engine.sandbox.enable();
   308→  }
   309→
   310→  // Enable prompt cache if specified
   311→  if (options.cache) {
   312→    config.promptCache.enabled = true;
   313→  }
   314→
   315→  // Resume session if requested
   316→  if (options.resume) {
   317→    const session = engine.sessionManager.load(options.resume);
   318→    if (session) {
   319→      engine.ui.success(`Resumed session: ${session.id.slice(0, 20)}...`);
   320→      if (options.fork) {
   321→        const forked = engine.sessionManager.create(process.cwd(), config.defaultModel);
   322→        forked.messages = [...session.messages];
   323→        engine.sessionManager.save();
   324→        engine.ui.success(`Forked to new session: ${forked.id.slice(0, 20)}...`);
   325→      }
   326→    } else {
   327→      engine.ui.error(`Session not found: ${options.resume}`);
   328→    }
   329→  } else if (options.continue) {
   330→    const sessions = engine.sessionManager.list();
   331→    if (sessions.length > 0) {
   332→      const session = engine.sessionManager.load(sessions[0].id);
   333→      if (session) {
   334→        engine.ui.success(`Continued session: ${session.id.slice(0, 20)}...`);
   335→      }
   336→    } else {
   337→      engine.ui.warning('No sessions found to continue');
   338→    }
   339→  }
   340→
   341→  // Print banner
   342→  engine.ui.banner();
   343→  const theme = engine.ui.theme;
   344→  const permMode = engine.approval.getMode();
   345→  const permIcon = engine.approval.getModeIcon();
   346→  const activeStyle = engine.styleManager.getStyle();
   347→  console.log(theme.muted(`  Model: ${MODELS[config.defaultModel]?.name || config.defaultModel}`));
   348→  console.log(theme.muted(`  Permission: ${permIcon} ${permMode}`));
   349→  console.log(theme.muted(`  Style: ${activeStyle.name}`));
   350→  console.log(theme.muted(`  Thinking: ${engine.extendedThinking.getMode()}`));
   351→  console.log(theme.muted(`  Cache: ${config.promptCache.enabled ? 'on' : 'off'}`));
   352→  console.log(theme.muted(`  Working Dir: ${process.cwd()}`));
   353→  console.log(theme.muted(`  Type /help for commands, Tab for completion, Ctrl+C to exit\n`));
   354→
   355→  // Create readline with tab completion
   356→  const completionEngine = new CompletionEngine(process.cwd());
   357→  completionEngine.setAgentNames(Array.from(engine.agents.keys()));
   358→
   359→  const rl = createInterface({
   360→    input: process.stdin,
   361→    output: process.stdout,
   362→    prompt: chalk.cyan('❯ '),
   363→    historySize: 100,
   364→    completer: completionEngine.complete,
   365→  });
   366→
   367→  rl.prompt();
   368→
   369→  let currentMode: 'auto' | 'agent' | 'direct' = 'auto';
   370→  let currentAgent: string | undefined;
   371→
   372→  rl.on('line', async (line) => {
   373→    const input = line.trim();
   374→    if (!input) { rl.prompt(); return; }
   375→
   376→    // Add to history
   377→    completionEngine.addHistory(input);
   378→
   379→    // Handle slash commands
   380→    if (input.startsWith('/')) {
   381→      const [cmd, ...args] = input.slice(1).split(' ');
   382→
   383→      switch (cmd) {
   384→        case 'help':
   385→          printHelp(engine);
   386→          break;
   387→
   388→        case 'model':
   389→          if (args[0]) {
   390→            engine.switchModel(args[0]);
   391→            engine.fallback.addFallbackModel(config.defaultModel);
   392→          } else {
   393→            engine.ui.modelList(config.defaultModel);
   394→          }
   395→          break;
   396→
   397→        case 'agent':
   398→          if (args[0]) {
   399→            const agentNames = Array.from(engine.agents.keys());
   400→            const found = agentNames.find(n => n.toLowerCase() === args[0].toLowerCase());
   401→            if (found) { currentAgent = found; currentMode = 'direct'; engine.ui.success(`Switched to agent: ${found}`); }
   402→            else { engine.ui.error(`Agent not found. Available: ${agentNames.join(', ')}`); }
   403→          } else {
   404→            const agents = Array.from(engine.agents.entries()).map(([name]) => ({ name, description: '', model: config.defaultModel }));
   405→            engine.ui.agentList(agents);
   406→          }
   407→          break;
   408→
   409→        case 'auto':
   410→          currentMode = 'auto'; currentAgent = undefined;
   411→          engine.ui.success('Mode: Auto (smart orchestration)');
   412→          break;
   413→
   414→        case 'orchestrate':
   415→        case 'plan':
   416→          currentMode = 'agent'; currentAgent = undefined;
   417→          engine.ui.success('Mode: Multi-agent orchestration');
   418→          break;
   419→
   420→        case 'direct':
   421→          currentMode = 'direct';
   422→          engine.ui.success('Mode: Direct (single agent)');
   423→          break;
   424→
   425→        case 'plan-mode':
   426→          engine.approval.setMode('plan');
   427→          engine.ui.success(`${engine.approval.getModeIcon()} Permission: plan (read-only)`);
   428→          break;
   429→
   430→        case 'stats':
   431→          const stats = engine.getSessionStats();
   432→          engine.ui.sessionStats(stats.inputTokens, stats.outputTokens, stats.cost);
   433→          break;
   434→
   435→        case 'clear':
   436→          console.clear(); engine.ui.banner();
   437→          break;
   438→
   439→        case 'theme':
   440→          if (args[0]) {
   441→            config.ui.theme = args[0] as 'dark' | 'light' | 'dracula' | 'nord';
   442→            engine.ui = new TerminalUI(config.ui.theme, config.ui.showTokenCount, config.ui.showCost);
   443→            engine.ui.success(`Theme: ${args[0]}`);
   444→          } else { console.log('Available themes: dracula, dark, nord, light'); }
   445→          break;
   446→
   447→        case 'resume':
   448→          if (args[0]) {
   449→            const sessionId = args[0] === 'latest' ? engine.sessionManager.list()[0]?.id : args[0];
   450→            if (sessionId) {
   451→              const session = engine.sessionManager.load(sessionId);
   452→              if (session) engine.ui.success(`Resumed session: ${sessionId.slice(0, 20)}...`);
   453→              else engine.ui.error(`Session not found: ${sessionId}`);
   454→            } else { engine.ui.error('No sessions found'); }
   455→          } else {
   456→            const sessions = engine.sessionManager.list();
   457→            if (sessions.length === 0) { engine.ui.info('No sessions found'); break; }
   458→            console.log(chalk.bold('\nRecent sessions:\n'));
   459→            for (const s of sessions.slice(0, 10)) {
   460→              const date = new Date(s.createdAt).toLocaleString();
   461→              console.log(`  ${chalk.cyan(s.id.slice(0, 20))}  ${chalk.gray(date)}  ${chalk.gray(`${s.messageCount} msgs`)}`);
   462→            }
   463→            console.log(chalk.gray('\nUse /resume <id> to resume a session'));
   464→          }
   465→          break;
   466→
   467→        case 'compact':
   468→          engine.ui.info('Compacting conversation context...');
   469→          engine.contextManager.manage(engine.sessionManager.getCurrent()?.messages || []);
   470→          engine.ui.success('Context compacted');
   471→          break;
   472→
   473→        case 'undo':
   474→          engine.ui.info('Undoing last change...');
   475→          const undoAction = engine.undoRedo.undo();
   476→          if (undoAction) {
   477→            engine.ui.success(`Undone: ${undoAction.description}`);
   478→          } else {
   479→            engine.ui.warning('Nothing to undo');
   480→          }
   481→          break;
   482→
   483→        case 'redo':
   484→          engine.ui.info('Redoing...');
   485→          const redoAction = engine.undoRedo.redo();
   486→          if (redoAction) {
   487→            engine.ui.success(`Redone: ${redoAction.description}`);

   883→          break;
   884→
   885→        case 'exit':
   886→        case 'quit':
   887→        case 'q':
   888→          engine.mcpClient.disconnectAll().catch(() => {});
   889→          engine.approval.close();
   890→          engine.ui.info('Goodbye!');
   891→          process.exit(0);
   892→          break;
   893→
   894→        default:
   895→          engine.ui.error(`Unknown command: /${cmd}. Type /help for available commands.`);
   896→      }
   897→
   898→      rl.prompt();
   899→      return;
   900→    }
   901→
   902→    // Process message with the engine
   903→    try {
   904→      await engine.processMessage(input, currentMode, currentAgent);
   905→    } catch (error) {
   906→      engine.ui.error(error instanceof Error ? error.message : String(error));
   907→    }
   908→
   909→    rl.prompt();
   910→  });
   911→
   912→  rl.on('close', () => {
   913→    engine.mcpClient.disconnectAll().catch(() => {});
   914→    engine.approval.close();
   915→    engine.ui.info('Goodbye!');
   916→    process.exit(0);
   917→  });
   918→}
