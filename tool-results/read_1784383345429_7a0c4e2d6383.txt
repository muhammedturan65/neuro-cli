   203→// ScopedToolRegistry
   204→// ============================================================

   206→/**
   207→ * A filtered view of a ToolRegistry that only exposes tools
   208→ * permitted by the sub-agent's allowlist/denylist, and wraps
   209→ * file tools with path validation for file scope enforcement.
   210→ */
   211→class ScopedToolRegistry extends ToolRegistry {
   212→  private allowedSet: Set<string>;
   213→  private deniedSet: Set<string>;
   214→  private fileScopeAllowed: Set<string>;
   215→  private fileScopeDenied: Set<string>;
   216→  private fileScopeEnabled: boolean;
   217→  private workingDir: string;
   218→
   219→  constructor(
   220→    parentRegistry: ToolRegistry,
   221→    allowedTools: string[],
   222→    deniedTools: string[],
   223→    fileScope: SubAgentConfig['fileScope'],
   224→    workingDir: string,
   225→  ) {
   226→    super();
   227→
   228→    this.allowedSet = new Set(allowedTools);
   229→    this.deniedSet = new Set(deniedTools);
   230→    this.workingDir = workingDir;
   231→
   232→    // Resolve file scope paths to absolute
   233→    this.fileScopeAllowed = new Set(
   234→      (fileScope?.allowedPaths ?? []).map((p) =>
   235→        isAbsolute(p) ? p : resolve(workingDir, p),
   236→      ),
   237→    );
   238→    this.fileScopeDenied = new Set(
   239→      (fileScope?.deniedPaths ?? []).map((p) =>
   240→        isAbsolute(p) ? p : resolve(workingDir, p),
   241→      ),
   242→    );
   243→    this.fileScopeEnabled = this.fileScopeAllowed.size > 0 || this.fileScopeDenied.size > 0;
   244→
   245→    // Copy permitted tools from parent, wrapping file tools if needed
   246→    for (const tool of parentRegistry.getAll()) {
   247→      if (!this.allowedSet.has(tool.name)) continue;
   248→      if (this.deniedSet.has(tool.name)) continue;
   249→
   250→      if (this.fileScopeEnabled && FILE_TOOL_NAMES.has(tool.name)) {
   251→        this.register(this.wrapFileTool(tool));
   252→      } else {
   253→        this.register(tool);
   254→      }
   255→    }
   256→  }
   257→
   258→  /**
   259→   * Wrap a file tool executor with path validation.
   260→   * If a file path argument is outside the allowed scope or
   261→   * inside the denied scope, the tool returns an error instead
   262→   * of executing.
   263→   */
   264→  private wrapFileTool(tool: ToolExecutor): ToolExecutor {
   265→    const originalExecute = tool.execute.bind(tool);
   266→    const scopeAllowed = this.fileScopeAllowed;
   267→    const scopeDenied = this.fileScopeDenied;
   268→    const workingDir = this.workingDir;
   269→
   270→    const wrappedExecute = async (
   271→      args: Record<string, unknown>,
   272→      context: ToolContext,
   273→    ): Promise<string> => {
   274→      // Find the file path argument — tools use various arg names
   275→      const filePath = this.extractFilePath(args);
   276→      if (filePath !== null) {
   277→        const absolutePath = isAbsolute(filePath) ? filePath : resolve(workingDir, filePath);
   278→
   279→        // Check denied paths first (denied takes priority)
   280→        for (const denied of scopeDenied) {
   281→          if (absolutePath.startsWith(denied) || absolutePath === denied) {
   282→            return `ERROR: Access denied — path "${filePath}" is outside the allowed file scope.`;
   283→          }
   284→        }
   285→
   286→        // If allowed paths are specified, the path must be under one of them
   287→        if (scopeAllowed.size > 0) {
   288→          let isAllowed = false;
   289→          for (const allowed of scopeAllowed) {
   290→            if (absolutePath.startsWith(allowed) || absolutePath === allowed) {
   291→              isAllowed = true;
   292→              break;
   293→            }
   294→          }
   295→          if (!isAllowed) {
   296→            return `ERROR: Access denied — path "${filePath}" is outside the allowed file scope.`;
   297→          }
   298→        }
   299→      }
   300→
   301→      return originalExecute(args, context);
   302→    };
   303→
   304→    return {
   305→      name: tool.name,
   306→      definition: tool.definition,
   307→      description: tool.description,
   308→      parameters: tool.parameters,
   309→      execute: wrappedExecute,
   310→      getApprovalRequest: tool.getApprovalRequest?.bind(tool),
   311→      risk: tool.risk,
   312→    };
   313→  }
   314→
   315→  /**
   316→   * Extract a file path from tool arguments.
   317→   * Different tools use different argument names for paths.
   318→   */
   319→  private extractFilePath(args: Record<string, unknown>): string | null {
   320→    const pathKeys = ['path', 'filePath', 'file_path', 'source', 'directory', 'dir'];
   321→    for (const key of pathKeys) {
   322→      if (typeof args[key] === 'string' && (args[key] as string).length > 0) {
   323→        return args[key] as string;
   324→      }
   325→    }
   326→    // Check for nested destinations (e.g., move_file has source + destination)
   327→    if (typeof args['destination'] === 'string') {
   328→      return args['destination'] as string;
   329→    }
   330→    return null;
   331→  }
   332→}

   336→// ============================================================
   337→
   338→/**
   339→ * Simple file snapshot utility that records file content hashes
   340→ * before sub-agent execution and compares afterwards to detect
   341→ * what files were modified.
   342→ */
   343→class FileSnapshot {
   344→  private workingDir: string;
   345→  private snapshot: Map<string, string> = new Map();
   346→  private scopedPaths: string[];
   347→
   348→  constructor(workingDir: string, fileScope?: SubAgentConfig['fileScope']) {
   349→    this.workingDir = workingDir;
   350→
   351→    if (fileScope && fileScope.allowedPaths.length > 0) {
   352→      this.scopedPaths = fileScope.allowedPaths.map((p) =>
   353→        isAbsolute(p) ? p : resolve(workingDir, p),
   354→      );
   355→    } else {
   356→      this.scopedPaths = [workingDir];
   357→    }
   358→  }
   359→
   360→  /**
   361→   * Take a snapshot of all files in the scoped paths.
   362→   * Uses a fast hash (simple content length + first/last bytes) for
   363→   * comparison purposes. Does not follow symlinks or descend into
   364→   * node_modules / .git directories.
   365→   */
   366→  take(): Map<string, string> {
   367→    this.snapshot = new Map();
   368→    for (const scopedPath of this.scopedPaths) {
   369→      this.walkDir(scopedPath);
   370→    }
   371→    return new Map(this.snapshot);
   372→  }
   373→
   374→  /**
   375→   * Compare the current file state against the snapshot and
   376→   * return a list of detected changes.
   377→   */
   378→  diff(): FileChange[] {
   379→    const changes: FileChange[] = [];
   380→    const currentSnapshot = new Map<string, string>();
   381→
   382→    // Re-scan the same directories
   383→    for (const scopedPath of this.scopedPaths) {
   384→      this.walkDirInto(scopedPath, currentSnapshot);
   385→    }
   386→
   387→    // Find modified and deleted files (present in original snapshot)
   388→    for (const [filePath, originalHash] of this.snapshot) {
   389→      const currentHash = currentSnapshot.get(filePath);
   390→      if (currentHash === undefined) {
   391→        changes.push({ path: filePath, type: 'delete' });
   392→      } else if (currentHash !== originalHash) {
   393→        changes.push({ path: filePath, type: 'modify' });
   394→      }
   395→    }
   396→
   397→    // Find newly created files (present in current but not original)
   398→    for (const [filePath] of currentSnapshot) {
   399→      if (!this.snapshot.has(filePath)) {
   400→        changes.push({ path: filePath, type: 'create' });
   401→      }
   402→    }
   403→
   404→    return changes;
   405→  }
   406→
   407→  /**
   408→   * Recursively walk a directory and record file hashes.
   409→   */
   410→  private walkDir(dirPath: string): void {
   411→    this.walkDirInto(dirPath, this.snapshot);
   412→  }
   413→
   414→  private walkDirInto(dirPath: string, target: Map<string, string>): void {
   415→    if (!existsSync(dirPath)) return;
   416→
   417→    let entries;
   418→    try {
   419→      entries = readdirSync(dirPath, { withFileTypes: true });
   420→    } catch {
   421→      return; // Permission denied or similar
   422→    }
   423→
   424→    for (const entry of entries) {
   425→      // Skip common non-essential directories
   426→      if (entry.isDirectory()) {
   427→        if (
   428→          entry.name === 'node_modules' ||
   429→          entry.name === '.git' ||
   430→          entry.name === '.next' ||
   431→          entry.name === 'dist' ||
   432→          entry.name === '.turbo'
   433→        ) {
   434→          continue;
   435→        }
   436→        this.walkDirInto(resolve(dirPath, entry.name), target);
   437→      } else if (entry.isFile()) {
   438→        const fullPath = resolve(dirPath, entry.name);
   439→        try {
   440→          const hash = this.quickHash(fullPath);
   441→          target.set(fullPath, hash);
   442→        } catch {
   443→          // File may be unreadable; skip
   444→        }
   445→      }
   446→    }
   447→  }
   448→
   449→  /**
   450→   * Quick content hash: uses file size + first 64 bytes + last 64 bytes.
   451→   * Fast and sufficient for detecting modifications without reading
   452→   * entire large files.
   453→   */
   454→  private quickHash(filePath: string): string {
   455→    const stat = statSync(filePath);
   456→    const size = stat.size;
   457→
   458→    if (size === 0) return `0:empty`;
   459→
   460→    const fd = [] as number[];
   461→    const buf = Buffer.alloc(Math.min(size, 128));
   462→    const fileHandle = readFileSync(filePath);
   463→
   464→    // Use first 64 bytes and last 64 bytes
   465→    const firstChunk = fileHandle.subarray(0, Math.min(64, size));
   466→    const lastChunk = size > 64 ? fileHandle.subarray(Math.max(0, size - 64)) : Buffer.alloc(0);
   467→
   468→    return `${size}:${firstChunk.toString('hex')}:${lastChunk.toString('hex')}`;
   469→  }
   470→}

   472→// ============================================================
   473→// SubAgentManager
   474→// ============================================================
   475→
   476→/**
   477→ * Manages the lifecycle of sub-agents spawned by a parent agent.
   478→ *
   479→ * A parent agent creates a SubAgentManager with its own OpenRouterClient
   480→ * and ToolRegistry. When it wants to delegate work, it calls spawn()
   481→ * or spawnAsync() with a configuration that defines the child's
   482→ * permissions, tools, and resource limits.
   483→ *
   484→ * The SubAgentManager:
   485→ * - Creates a temporary BaseAgent with a scoped tool registry
   486→ * - Enforces resource limits (tokens, turns, timeout)
   487→ * - Tracks file modifications via before/after snapshots
   488→ * - Supports hierarchical delegation (sub-agents can spawn their own children)
   489→ * - Emits events for status changes
   490→ *
   491→ * Usage:
   492→ * ```typescript
   493→ * const manager = new SubAgentManager(client, registry, '/project', 'session-1');
   494→ *
   495→ * // Sequential: block until done
   496→ * const result = await manager.spawn({
   497→ *   name: 'code-searcher',
   498→ *   systemPrompt: 'You are a code search specialist.',
   499→ *   allowedTools: ['search_files', 'read_file'],
   500→ *   resourceLimits: { maxTokens: 50_000, maxTurns: 10, timeoutMs: 60_000 },
   501→ *   maxDepth: 2,
   502→ *   mode: 'sequential',
   503→ * }, 'Find all uses of the deprecated API');
   504→ *
   505→ * // Async: fire and forget, get a task ID
   506→ * const taskId = await manager.spawnAsync({ ... }, 'Refactor the utils');
   507→ * // Later...
   508→ * const status = manager.getStatus(taskId);
   509→ * const result = await manager.waitFor(taskId);
   510→ * ```
   511→ */
   512→export class SubAgentManager extends EventEmitter {
   513→  private client: OpenRouterClient;
   514→  private parentRegistry: ToolRegistry;
   515→  private workingDirectory: string;
   516→  private sessionId: string;
   517→  private defaultModel: string;
   518→  private currentDepth: number;
   519→  private maxGlobalDepth: number;
   520→
   521→  /** Active and completed sub-agent records */
   522→  private records: Map<string, SubAgentRecord> = new Map();
   523→
   524→  /** Monotonic counter for generating unique task IDs */
   525→  private taskCounter: number = 0;
   526→
   527→  /**
   528→   * @param client        The OpenRouterClient shared with the parent agent
   529→   * @param registry      The parent's ToolRegistry (tools will be filtered for children)
   530→   * @param workingDir    The working directory for sub-agents
   531→   * @param sessionId     The session ID to inherit
   532→   * @param defaultModel  The default model to use (inherited from parent)
   533→   * @param currentDepth  Current nesting depth (0 for top-level manager)
   534→   * @param maxGlobalDepth Maximum allowed nesting depth across the hierarchy
   535→   */
   536→  constructor(
   537→    client: OpenRouterClient,
   538→    registry: ToolRegistry,
   539→    workingDir: string,
   540→    sessionId: string,
   541→    defaultModel: string = 'qwen/qwen3-coder:free',
   542→    currentDepth: number = 0,
   543→    maxGlobalDepth: number = DEFAULT_MAX_DEPTH,
   544→  ) {
   545→    super();
   546→    this.client = client;
   547→    this.parentRegistry = registry;
   548→    this.workingDirectory = workingDir;
   549→    this.sessionId = sessionId;
   550→    this.defaultModel = defaultModel;
   551→    this.currentDepth = currentDepth;
   552→    this.maxGlobalDepth = maxGlobalDepth;
   553→  }
   554→
   555→  // ----------------------------------------------------------
   556→  // Public API — Spawn
   557→  // ----------------------------------------------------------
   558→
   559→  /**
   560→   * Spawn a sub-agent in sequential mode and wait for its result.
   561→   * The parent agent blocks until the child completes (or fails/times out).
   562→   *
   563→   * @param config  Configuration for the sub-agent
   564→   * @param prompt  The task prompt to send to the sub-agent
   565→   * @returns       Structured result from the sub-agent
   566→   * @throws        Error if the sub-agent fails or is cancelled
   567→   */
   568→  async spawn(config: SubAgentConfig, prompt: string): Promise<SubAgentResult> {
   569→    const effectiveConfig = this.applyDefaults(config);
   570→    this.validateConfig(effectiveConfig);
   571→
   572→    const taskId = this.generateTaskId(effectiveConfig.name);
   573→    const record = this.createRecord(taskId, effectiveConfig, prompt);
   574→
   575→    this.emit('subagent:spawning', taskId, effectiveConfig.name);
   576→
   577→    try {
   578→      const result = await this.executeSubAgent(record, effectiveConfig, prompt);
   579→      return result;
   580→    } catch (err) {
   581→      const error = err instanceof Error ? err : new Error(String(err));
   582→      const failResult: SubAgentResult = {
   583→        success: false,
   584→        output: '',
   585→        filesModified: [],
   586→        tokensUsed: 0,
   587→        turnsCompleted: 0,
   588→        error: error.message,
   589→        duration: Date.now() - record.startedAt,
   590→      };
   591→      record.status = 'failed';
   592→      record.result = failResult;
   593→      record.completedAt = Date.now();
   594→      record.completionReject?.(error);
   595→      this.emit('subagent:failed', taskId, error.message);
   596→      return failResult;
   597→    } finally {
   598→      this.records.delete(taskId);
   599→    }
   600→  }
   601→
   602→  /**
   603→   * Spawn a sub-agent in async (fire-and-forget) mode.
   604→   * Returns a task ID immediately; the sub-agent runs in the background.
   605→   * Use getStatus() to check progress and waitFor() to collect results.
   606→   *
   607→   * @param config  Configuration for the sub-agent
   608→   * @param prompt  The task prompt to send to the sub-agent
   609→   * @returns       Unique task ID for tracking this sub-agent
   610→   */
   611→  async spawnAsync(config: SubAgentConfig, prompt: string): Promise<string> {
   612→    const effectiveConfig = this.applyDefaults(config);
   613→    this.validateConfig(effectiveConfig);
   614→
   615→    const taskId = this.generateTaskId(effectiveConfig.name);
   616→    const record = this.createRecord(taskId, effectiveConfig, prompt);
   617→
   618→    this.emit('subagent:spawning', taskId, effectiveConfig.name);
   619→
   620→    // Execute in the background (non-blocking)
   621→    this.executeSubAgent(record, effectiveConfig, prompt)
   622→      .then((result) => {
   623→        record.status = 'completed';
   624→        record.result = result;
   625→        record.completedAt = Date.now();
   626→        record.completionResolve?.(result);
   627→        this.emit('subagent:completed', taskId, result);
   628→      })
   629→      .catch((err: unknown) => {
   630→        if (record.status === 'cancelled') return; // Already handled
   631→        const error = err instanceof Error ? err : new Error(String(err));
   632→        record.status = 'failed';
   633→        record.result = {
   634→          success: false,
   635→          output: '',
   636→          filesModified: [],
   637→          tokensUsed: 0,
   638→          turnsCompleted: 0,
   639→          error: error.message,
   640→          duration: Date.now() - record.startedAt,
   641→        };
   642→        record.completedAt = Date.now();
   643→        record.completionReject?.(error);
   644→        this.emit('subagent:failed', taskId, error.message);
   645→      });
   646→
   647→    return taskId;
   648→  }

   672→  getInfo(taskId: string): SubAgentInfo | undefined {
   673→    const record = this.records.get(taskId);

   677→      taskId: record.taskId,

   689→  /**
   690→   * Cancel a running sub-agent.
   691→   * The sub-agent's abort controller is triggered, causing it to
   692→   * stop at the next iteration boundary.
   693→   *
   694→   * @param taskId  The task ID to cancel
   695→   * @returns       True if the agent was successfully cancelled
   696→   */
   697→  cancel(taskId: string): boolean {
   698→    const record = this.records.get(taskId);
   699→    if (!record) return false;
   700→
   701→    if (
   702→      record.status === 'completed' ||
   703→      record.status === 'failed' ||
   704→      record.status === 'cancelled'
   705→    ) {
   706→      return false; // Already terminal
   707→    }
   708→
   709→    record.abortController.abort();
   710→    record.status = 'cancelled';
   711→    record.completedAt = Date.now();
   712→
   713→    if (!record.result) {
   714→      record.result = {
   715→        success: false,
   716→        output: '',
   717→        filesModified: [],
   718→        tokensUsed: 0,
   719→        turnsCompleted: 0,
   720→        error: 'Cancelled by parent',
   721→        duration: Date.now() - record.startedAt,
   722→      };
   723→    }
   724→
   725→    record.completionReject?.(new Error(`Sub-agent "${taskId}" was cancelled`));
   726→    this.emit('subagent:cancelled', taskId);
   727→    return true;
   728→  }

   730→  /**
   731→   * List all active sub-agents (pending, running, or timed_out).
   732→   *
   733→   * @returns Array of sub-agent info objects
   734→   */
   735→  list(): SubAgentInfo[] {
   736→    const infos: SubAgentInfo[] = [];
   737→    for (const record of this.records.values()) {
   738→      if (
   739→        record.status === 'pending' ||
   740→        record.status === 'running' ||
   741→        record.status === 'timed_out'
   742→      ) {
   743→        infos.push({
   744→          taskId: record.taskId,
   745→          name: record.config.name,
   746→          status: record.status,
   747→          depth: record.depth,
   748→          turnsCompleted: record.result?.turnsCompleted ?? 0,
   749→          tokensUsed: record.result?.tokensUsed ?? 0,
   750→          elapsedMs: Date.now() - record.startedAt,
   751→          parentTaskId: record.parentTaskId,
   752→          startedAt: record.startedAt,
   753→        });
   754→      }
   755→    }
   756→    return infos.sort((a, b) => a.startedAt - b.startedAt);
   757→  }

   759→  /**
   760→   * Wait for a specific async sub-agent to complete and return its result.
   761→   *
   762→   * @param taskId  The task ID to wait for
   763→   * @returns       The sub-agent's result
   764→   * @throws        Error if the task doesn't exist or was cancelled
   765→   */
   766→  async waitFor(taskId: string): Promise<SubAgentResult> {
   767→    const record = this.records.get(taskId);
   768→    if (!record) {
   769→      throw new Error(`Sub-agent task "${taskId}" not found`);
   770→    }
   771→
   772→    // If already completed, return immediately
   773→    if (record.status === 'completed' && record.result) {
   774→      return record.result;
   775→    }
   776→    // If already failed/cancelled, throw
   777→    if (
   778→      (record.status === 'failed' || record.status === 'cancelled') &&
   779→      record.result
   780→    ) {
   781→      if (record.result.error) {
   782→        throw new Error(record.result.error);
   783→      }
   784→      return record.result;
   785→    }
   786→
   787→    return record.completionPromise;
   788→  }
   789→
   790→  /**
   791→   * Wait for all currently running async sub-agents to complete.
   792→   * Returns results for all tasks that were active at the time of the call.
   793→   *
   794→   * @returns Array of results from all awaited sub-agents
   795→   */
   796→  async waitForAll(): Promise<SubAgentResult[]> {
   797→    const activeIds: string[] = [];
   798→    for (const [id, record] of this.records) {
   799→      if (
   800→        record.status === 'pending' ||
   801→        record.status === 'running'
   802→      ) {
   803→        activeIds.push(id);
   804→      }
   805→    }
   806→
   807→    const results = await Promise.allSettled(
   808→      activeIds.map((id) => this.waitFor(id)),
   809→    );
   810→
   811→    return results.map((r, i) => {
   812→      if (r.status === 'fulfilled') return r.value;
   813→      // Rejected — return a failure result
   814→      return {
   815→        success: false,
   816→        output: '',
   817→        filesModified: [],
   818→        tokensUsed: 0,
   819→        turnsCompleted: 0,
   820→        error: r.reason?.message ?? String(r.reason),
   821→        duration: 0,
   822→      } as SubAgentResult;
   823→    });
   824→  }

   827→   * Get the result of a completed sub-agent.
   828→   *
   829→   * @param taskId  The task ID
   830→   * @returns       The result, or undefined if not completed
   831→   */
   832→  getResult(taskId: string): SubAgentResult | undefined {
   833→    const record = this.records.get(taskId);
   834→    if (!record || record.status !== 'completed') return undefined;
   835→    return record.result;

   891→  /**
   892→   * Validate a SubAgentConfig, throwing if it's invalid.
   893→   */
   894→  private validateConfig(config: SubAgentConfig): void {
   895→    if (!config.name || config.name.trim().length === 0) {
   896→      throw new Error('Sub-agent config must have a non-empty name');
   897→    }
   898→    if (!config.systemPrompt || config.systemPrompt.trim().length === 0) {
   899→      throw new Error('Sub-agent config must have a non-empty systemPrompt');
   900→    }
   901→    if (!config.allowedTools || config.allowedTools.length === 0) {
   902→      throw new Error('Sub-agent config must specify at least one allowed tool');
   903→    }
   904→    if (config.resourceLimits.maxTokens <= 0) {
   905→      throw new Error('Sub-agent resourceLimits.maxTokens must be > 0');
   906→    }
   907→    if (config.resourceLimits.maxTurns <= 0) {
   908→      throw new Error('Sub-agent resourceLimits.maxTurns must be > 0');
   909→    }
   910→    if (config.resourceLimits.timeoutMs <= 0) {
   911→      throw new Error('Sub-agent resourceLimits.timeoutMs must be > 0');
   912→    }
   913→
   914→    // Check nesting depth
   915→    const nextDepth = this.currentDepth + 1;
   916→    if (nextDepth > this.maxGlobalDepth) {
   917→      throw new Error(
   918→        `Cannot spawn sub-agent: nesting depth ${nextDepth} exceeds maximum ${this.maxGlobalDepth}`,
   919→      );
   920→    }
   921→    if (config.maxDepth > this.maxGlobalDepth) {
   922→      throw new Error(
   923→        `Sub-agent maxDepth ${config.maxDepth} exceeds global maximum ${this.maxGlobalDepth}`,
   924→      );
   925→    }
   926→
   927→    // Validate that allowed tools exist in the parent registry
   928→    for (const toolName of config.allowedTools) {
   929→      if (!this.parentRegistry.has(toolName)) {
   930→        throw new Error(
   931→          `Allowed tool "${toolName}" does not exist in the parent tool registry`,
   932→        );
   933→      }
   934→    }
   935→  }

   937→  /**
   938→   * Create a SubAgentRecord for tracking the sub-agent lifecycle.

   940→  private createRecord(
   941→    taskId: string,
   942→    config: SubAgentConfig,
   943→    prompt: string,
   944→  ): SubAgentRecord {
   945→    let completionResolve: ((result: SubAgentResult) => void) | null = null;
   946→    let completionReject: ((error: Error) => void) | null = null;
   947→    const completionPromise = new Promise<SubAgentResult>((resolve, reject) => {
   948→      completionResolve = resolve;
   949→      completionReject = reject;
   950→    });
   951→
   952→    const record: SubAgentRecord = {
   953→      taskId,
   954→      config,
   955→      prompt,
   956→      status: 'pending',
   957→      depth: this.currentDepth + 1,
   958→      workingDirectory: this.workingDirectory,
   959→      sessionId: this.sessionId,
   960→      abortController: new AbortController(),
   961→      startedAt: Date.now(),
   962→      fileSnapshot: new Map(),
   963→      completionPromise,
   964→      completionResolve,
   965→      completionReject,
   966→    };
   967→
   968→    this.records.set(taskId, record);
   969→    return record;
   970→  }

   976→  /**
   977→   * Execute a sub-agent: create a scoped BaseAgent, run it with
   978→   * resource limit enforcement, and collect results.
   979→   */
   980→  private async executeSubAgent(
   981→    record: SubAgentRecord,
   982→    config: SubAgentConfig,
   983→    prompt: string,
   984→  ): Promise<SubAgentResult> {
   985→    record.status = 'running';
   986→    this.emit('subagent:started', record.taskId, config.name);
   987→
   988→    const startTime = Date.now();
   989→    const abortSignal = record.abortController.signal;
   990→
   991→    // Take a file snapshot before execution
   992→    const snapshot = new FileSnapshot(this.workingDirectory, config.fileScope);
   993→    record.fileSnapshot = snapshot.take();
   994→
   995→    // Build the scoped tool registry
   996→    const scopedRegistry = new ScopedToolRegistry(
   997→      this.parentRegistry,
   998→      config.allowedTools,
   999→      config.deniedTools ?? [],
  1000→      config.fileScope,
  1001→      this.workingDirectory,
  1002→    );
  1003→
  1004→    // Build the AgentConfig for the sub-agent's BaseAgent
  1005→    const agentConfig: AgentConfig = {
  1006→      name: config.name,
  1007→      description: `Sub-agent: ${config.name}`,
  1008→      systemPrompt: this.buildSubAgentSystemPrompt(config),
  1009→      model: config.model,
  1010→      maxTokens: config.resourceLimits.maxTokens,
  1011→      maxIterations: config.resourceLimits.maxTurns,
  1012→      tools: config.allowedTools.filter((t) => !(config.deniedTools ?? []).includes(t)),
  1013→      autoApprove: true, // Sub-agents auto-approve; the parent already gated access
  1014→      temperature: 0.5,
  1015→    };
  1016→
  1017→    // Create the temporary BaseAgent
  1018→    const agent = new BaseAgent(
  1019→      agentConfig,
  1020→      this.client,
  1021→      scopedRegistry,
  1022→      this.workingDirectory,
  1023→      `${this.sessionId}-sub-${record.taskId}`,
  1024→    );
  1025→
  1026→    // Set up timeout
  1027→    let timeoutHandle: ReturnType<typeof setTimeout> | null = null;
  1028→    let timedOut = false;
  1029→
  1030→    if (config.resourceLimits.timeoutMs > 0) {
  1031→      timeoutHandle = setTimeout(() => {
  1032→        timedOut = true;
  1033→        record.abortController.abort();
  1034→      }, config.resourceLimits.timeoutMs);
  1035→    }
  1036→
  1037→    // Resource-tracking state
  1038→    let totalTokensUsed = 0;
  1039→    let turnsCompleted = 0;
  1040→
  1041→    // Callbacks that enforce resource limits
  1042→    const resourceCallbacks: AgentCallbacks = {
  1043→      onIteration: (iteration: number, _maxIterations: number) => {
  1044→        turnsCompleted = iteration;
  1045→
  1046→        // Check for abort (timeout or explicit cancel)
  1047→        if (abortSignal.aborted) {
  1048→          throw new Error(
  1049→            timedOut
  1050→              ? `Sub-agent "${config.name}" timed out after ${config.resourceLimits.timeoutMs}ms`
  1051→              : `Sub-agent "${config.name}" was cancelled`,
  1052→          );
  1053→        }
  1054→
  1055→        // Check token budget
  1056→        if (totalTokensUsed >= config.resourceLimits.maxTokens) {
  1057→          throw new Error(
  1058→            `Sub-agent "${config.name}" exceeded token budget of ${config.resourceLimits.maxTokens}`,
  1059→          );
  1060→        }
  1061→      },
  1062→
  1063→      onToolCall: (_toolName: string, _args: Record<string, unknown>) => {
  1064→        // Check for abort before each tool call
  1065→        if (abortSignal.aborted) {
  1066→          throw new Error(
  1067→            timedOut
  1068→              ? `Sub-agent "${config.name}" timed out`
  1069→              : `Sub-agent "${config.name}" was cancelled`,
  1070→          );
  1071→        }
  1072→      },
  1073→    };
  1074→
  1075→    try {
  1076→      // Run the sub-agent
  1077→      const runResult: AgentRunResult = await agent.run(prompt, resourceCallbacks);
  1078→
  1079→      // Clear timeout
  1080→      if (timeoutHandle !== null) clearTimeout(timeoutHandle);
  1081→
  1082→      // Account for token usage
  1083→      totalTokensUsed = runResult.usage.inputTokens + runResult.usage.outputTokens;
  1084→
  1085→      // Detect file changes
  1086→      const fileChanges = snapshot.diff();
  1087→      const filesModified = fileChanges.map((fc) =>
  1088→        relative(this.workingDirectory, fc.path) || fc.path,
  1089→      );
  1090→
  1091→      const result: SubAgentResult = {
  1092→        success: true,
  1093→        output: runResult.content || '',
  1094→        filesModified,
  1095→        tokensUsed: totalTokensUsed,
  1096→        turnsCompleted: runResult.iterations,
  1097→        duration: Date.now() - startTime,
  1098→      };
  1099→
  1100→      record.status = 'completed';
  1101→      record.result = result;
  1102→      record.completedAt = Date.now();
  1103→      record.completionResolve?.(result);
  1104→      this.emit('subagent:completed', record.taskId, result);
  1105→
  1106→      return result;
  1107→    } catch (err) {
  1108→      if (timeoutHandle !== null) clearTimeout(timeoutHandle);
  1109→
  1110→      // If timed out, set the appropriate status
  1111→      if (timedOut) {
  1112→        record.status = 'timed_out';
  1113→      }
  1114→
  1115→      const error = err instanceof Error ? err : new Error(String(err));
  1116→
  1117→      // Detect file changes even on failure (partial modifications are important)
  1118→      const fileChanges = snapshot.diff();
  1119→      const filesModified = fileChanges.map((fc) =>
  1120→        relative(this.workingDirectory, fc.path) || fc.path,
  1121→      );
  1122→
  1123→      const result: SubAgentResult = {
  1124→        success: false,
  1125→        output: '',
  1126→        filesModified,
  1127→        tokensUsed: totalTokensUsed,
  1128→        turnsCompleted,
  1129→        error: error.message,
  1130→        duration: Date.now() - startTime,
  1131→      };
  1132→
  1133→      record.result = result;
  1134→      record.completedAt = Date.now();
  1135→
  1136→      if (record.status === 'running') {
  1137→        record.status = 'failed';
  1138→      }
  1139→
  1140→      this.emit('subagent:failed', record.taskId, error.message);
  1141→
  1142→      // For sequential mode, re-throw so the caller knows it failed
  1143→      if (config.mode === 'sequential') {
  1144→        throw error;
  1145→      }
  1146→
  1147→      // For async mode, the result is stored and accessible via getStatus/getResult
  1148→      return result;
  1149→    }
  1150→  }

  1152→  /**
  1153→   * Build the system prompt for a sub-agent, combining the user-provided

  1156→  private buildSubAgentSystemPrompt(config: SubAgentConfig): string {

  1162→
  1163→    if (config.fileScope) {
  1164→      const allowedPaths = config.fileScope.allowedPaths.join(', ') || '(none)';
  1165→      const deniedPaths = config.fileScope.deniedPaths.join(', ') || '(none)';
  1166→      prompt += `\n- Allowed file paths: ${allowedPaths}`;
  1167→      prompt += `\n- Denied file paths: ${deniedPaths}`;

  1186→- If you encounter scope violations, report them rather than retrying`;
  1187→
  1188→    return prompt;
  1189→  }

  1231→ * const subAgentTool = new SubAgentTool(manager);
  1232→ * registry.register(subAgentTool);
  1233→ * ```
  1234→ */
  1235→export class SubAgentTool implements ToolExecutor {
  1236→  name = 'spawn_sub_agent';
  1237→  description =
  1238→    'Spawn a sub-agent to perform a specific task. The sub-agent runs with restricted tools and resources. Use this to delegate focused work to a specialist agent.';
  1239→  risk: 'low' | 'medium' | 'high' = 'medium';

  1284→  async execute(
  1285→    args: Record<string, unknown>,
  1286→    _context: ToolContext,
  1287→  ): Promise<string> {
  1288→    const name = args.name as string;
  1289→    const prompt = args.prompt as string;
  1290→    const allowedToolsStr = args.allowedTools as string;
  1291→    const maxTokens = (args.maxTokens as number) ?? 50_000;
  1292→    const maxTurns = (args.maxTurns as number) ?? 10;
  1293→    const timeoutSeconds = (args.timeoutSeconds as number) ?? 60;
  1294→
  1295→    if (!name || !prompt || !allowedToolsStr) {
  1296→      return 'ERROR: Missing required parameters: name, prompt, allowedTools';
  1297→    }
  1298→
  1299→    const allowedTools = allowedToolsStr
  1300→      .split(',')
  1301→      .map((t) => t.trim())
  1302→      .filter((t) => t.length > 0);
  1303→
  1304→    if (allowedTools.length === 0) {
  1305→      return 'ERROR: At least one tool must be specified in allowedTools';
  1306→    }
  1307→
  1308→    // Check if the manager can still spawn
  1309→    if (!this.manager.canSpawn) {
  1310→      return 'ERROR: Maximum sub-agent nesting depth reached. Cannot spawn further sub-agents.';
  1311→    }
  1312→
  1313→    const config: SubAgentConfig = {
  1314→      name,
  1315→      systemPrompt: `You are a specialized sub-agent named "${name}". Focus on completing the specific task you are assigned efficiently.`,
  1316→      allowedTools,
  1317→      resourceLimits: {
  1318→        maxTokens,
  1319→        maxTurns,
  1320→        timeoutMs: timeoutSeconds * 1000,
  1321→      },
  1322→      maxDepth: DEFAULT_MAX_DEPTH,
  1323→      mode: 'sequential',
  1324→    };
  1325→
  1326→    try {
  1327→      const result = await this.manager.spawn(config, prompt);
  1328→
  1329→      if (result.success) {
  1330→        let output = `Sub-agent "${name}" completed successfully.\n`;
  1331→        output += `Duration: ${Math.round(result.duration / 1000)}s | Tokens: ${result.tokensUsed.toLocaleString()} | Turns: ${result.turnsCompleted}\n`;
  1332→        if (result.filesModified.length > 0) {
  1333→          output += `Files modified: ${result.filesModified.join(', ')}\n`;
  1334→        }
  1335→        output += `\nResult:\n${result.output}`;
  1336→        return output;
  1337→      } else {
  1338→        return `Sub-agent "${name}" failed: ${result.error}\nTokens used: ${result.tokensUsed.toLocaleString()} | Turns: ${result.turnsCompleted}`;
  1339→      }
  1340→    } catch (err) {
  1341→      const error = err instanceof Error ? err.message : String(err);
  1342→      return `ERROR: Sub-agent "${name}" encountered an error: ${error}`;
  1343→    }
  1344→  }
  1345→
  1346→  getApprovalRequest?(args: Record<string, unknown>): import('../core/types.js').ApprovalRequest {
  1347→    // Sub-agent spawning is medium risk — it could modify files
  1348→    // but is already scoped by the config
  1349→    return {
  1350→      toolName: this.name,
  1351→      args,
  1352→      risk: 'medium',
  1353→      description: `Spawn sub-agent "${args.name}" with tools: ${args.allowedTools}`,
  1354→    };
  1355→  }
