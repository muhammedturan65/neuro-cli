   181→export class ModelRouter {
   182→  private config: RouterConfig;
   183→  private availableModels: Record<string, ModelInfo>;
   184→  private currentEffort: EffortLevel = 'medium';
   185→  private forcedModel: string | null = null;
   186→
   187→  // -----------------------------------------------------------------------
   188→  // Construction
   189→  // -----------------------------------------------------------------------
   190→
   191→  constructor(
   192→    config: RouterConfig,
   193→    availableModels: Record<string, ModelInfo>,
   194→  ) {
   195→    this.config = config;
   196→    this.availableModels = availableModels;
   197→    this.validateConfig();
   198→  }
   199→
   200→  // -----------------------------------------------------------------------
   201→  // Public API
   202→  // -----------------------------------------------------------------------
   203→
   204→  /**
   205→   * Route a prompt to the best available model.
   206→   */
   207→  route(prompt: string, effort?: EffortLevel): RoutingDecision {
   208→    const effectiveEffort = effort ?? this.currentEffort;
   209→    const analysis = this.analyzePrompt(prompt);
   210→    const estimatedTokens = this.estimateTokens(prompt);
   211→    const model = this.selectModel(analysis.complexity, analysis.category, effectiveEffort);
   212→    const alternatives = this.getAlternatives(analysis.complexity, analysis.category);
   213→    const reasoning = this.buildReasoning(analysis, effectiveEffort, model);
   214→
   215→    return {
   216→      model,
   217→      complexity: analysis.complexity,
   218→      category: analysis.category,
   219→      effort: effectiveEffort,
   220→      estimatedTokens,
   221→      reasoning,
   222→      alternatives,
   223→    };
   224→  }
   225→
   226→  /**
   227→   * Set the default effort level for subsequent routing calls.
   228→   */
   229→  setEffort(level: EffortLevel): void {
   230→    this.currentEffort = level;
   231→  }
   232→
   233→  /**
   234→   * Get the current effort level.
   235→   */
   236→  getEffort(): EffortLevel {
   237→    return this.currentEffort;
   238→  }
   239→
   240→  /**
   241→   * Force all routing decisions to use a specific model.
   242→   */
   243→  overrideModel(modelId: string): void {
   244→    if (!this.availableModels[modelId]) {
   245→      throw new Error(
   246→        `Cannot override to unknown model "${modelId}". Available: ${Object.keys(this.availableModels).join(', ')}`,
   247→      );
   248→    }
   249→    this.forcedModel = modelId;
   250→  }
   251→
   252→  /**
   253→   * Remove any forced model override.
   254→   */
   255→  clearOverride(): void {
   256→    this.forcedModel = null;
   257→  }
   258→
   259→  /**
   260→   * Classify the task category of a prompt.
   261→   */
   262→  getCategory(prompt: string): TaskCategory {
   263→    return this.analyzePrompt(prompt).category;
   264→  }
   265→
   266→  /**
   267→   * Classify the complexity of a prompt.
   268→   */
   269→  getComplexity(prompt: string): Complexity {
   270→    return this.analyzePrompt(prompt).complexity;
   271→  }
   272→
   273→  /**
   274→   * Estimate the number of tokens a prompt will consume.
   275→   * Uses a character-based heuristic (~4 chars per token for English,
   276→   * with adjustments for code-heavy content).
   277→   */
   278→  estimateTokens(prompt: string): number {
   279→    if (prompt.length === 0) return 0;
   280→
   281→    // Base estimate: ~4 characters per token for typical English text
   282→    let charPerToken = 4;
   283→
   284→    // Code-heavy prompts tend to have higher token density
   285→    const codeIndicators = [
   286→      /```[\s\S]*?```/,       // fenced code blocks
   287→      /\bfunction\b/,          // function keyword
   288→      /\bconst\b|\blet\b|\bvar\b/,  // variable declarations
   289→      /[{}()\[\];]/,           // braces and brackets
   290→      /=>/,                    // arrow functions
   291→      /import\s+.+\s+from/,   // imports
   292→    ];
   293→
   294→    const codeSignalCount = codeIndicators.filter(p => p.test(prompt)).length;
   295→    if (codeSignalCount >= 3) {
   296→      charPerToken = 3.2; // code is denser
   297→    } else if (codeSignalCount >= 1) {
   298→      charPerToken = 3.6;
   299→    }
   300→
   301→    // Whitespace-heavy prompts have lower density
   302→    const whitespaceRatio = (prompt.match(/\s/g) ?? []).length / prompt.length;
   303→    if (whitespaceRatio > 0.4) {
   304→      charPerToken = 4.5;
   305→    }
   306→
   307→    const baseTokens = Math.ceil(prompt.length / charPerToken);
   308→
   309→    // Add overhead for special tokens (system, role markers, etc.)
   310→    const overhead = 8;
   311→
   312→    return baseTokens + overhead;
   313→  }
   314→
   315→  /**
   316→   * Print a human-readable routing decision to stdout.
   317→   */
   318→  printDecision(decision: RoutingDecision): void {
   319→    const border = '-'.repeat(60);
   320→    const lines: string[] = [
   321→      border,
   322→      '  MODEL ROUTING DECISION',
   323→      border,
   324→      `  Model       : ${decision.model}`,
   325→      `  Complexity  : ${decision.complexity}`,
   326→      `  Category    : ${decision.category}`,
   327→      `  Effort      : ${decision.effort}`,
   328→      `  Est. Tokens : ${decision.estimatedTokens.toLocaleString()}`,
   329→      `  Reasoning   : ${decision.reasoning}`,
   330→    ];
   331→
   332→    if (decision.alternatives.length > 0) {
   333→      lines.push(`  Alternatives: ${decision.alternatives.join(', ')}`);
   334→    }
   335→
   336→    lines.push(border);
   337→
   338→    for (const line of lines) {
   339→      console.log(line);
   340→    }
   341→  }

   343→  // -----------------------------------------------------------------------
   344→  // Private: prompt analysis
   345→  // -----------------------------------------------------------------------
   346→
   347→  private analyzePrompt(prompt: string): PromptAnalysis {
   348→    const indicators: string[] = [];
   349→
   350→    // --- Category detection (scored) ---
   351→    const categoryScores = this.scoreCategories(prompt);
   352→    const category = this.pickTopCategory(categoryScores, indicators);
   353→
   354→    // --- Complexity detection (scored) ---
   355→    const complexity = this.scoreComplexity(prompt, indicators);
   356→
   357→    return { complexity, category, indicators };
   358→  }
   359→
   360→  /**
   361→   * Score each category based on how many pattern groups match.
   362→   * Each group in CATEGORY_PATTERNS is worth 1 point when at least one
   363→   * pattern in the group matches.
   364→   */
   365→  private scoreCategories(prompt: string): Map<TaskCategory, number> {
   366→    const scores = new Map<TaskCategory, number>();
   367→
   368→    for (const [cat, patternGroups] of Object.entries(CATEGORY_PATTERNS) as [TaskCategory, RegExp[]][]) {
   369→      let score = 0;
   370→      for (const pattern of patternGroups) {
   371→        if (pattern.test(prompt)) {
   372→          score += 1;
   373→        }
   374→      }
   375→      scores.set(cat, score);
   376→    }
   377→
   378→    return scores;
   379→  }
   380→
   381→  /**
   382→   * Determine the winning category and record indicators.
   383→   */
   384→  private pickTopCategory(
   385→    scores: Map<TaskCategory, number>,
   386→    indicators: string[],
   387→  ): TaskCategory {
   388→    let bestCategory: TaskCategory = 'conversation';
   389→    let bestScore = 0;
   390→
   391→    for (const [cat, score] of scores) {
   392→      if (score > bestScore) {
   393→        bestScore = score;
   394→        bestCategory = cat;
   395→      }
   396→    }
   397→
   398→    // If nothing matched at all, default to conversation
   399→    if (bestScore === 0) {
   400→      indicators.push('no strong category signals -- defaulting to conversation');
   401→      return 'conversation';
   402→    }
   403→
   404→    indicators.push(`category "${bestCategory}" matched with score ${bestScore}`);
   405→    return bestCategory;
   406→  }
   407→
   408→  /**
   409→   * Score complexity using a weighted combination of:
   410→   *   1. Pattern match signals
   411→   *   2. Prompt length
   412→   *   3. Multi-step indicators
   413→   */
   414→  private scoreComplexity(prompt: string, indicators: string[]): Complexity {
   415→    let simpleScore = 0;
   416→    let moderateScore = 0;
   417→    let complexScore = 0;
   418→
   419→    // --- Pattern-based scoring ---
   420→
   421→    for (const pattern of COMPLEXITY_SIGNALS.simple.patterns) {
   422→      if (pattern.test(prompt)) {
   423→        simpleScore += 2;
   424→        indicators.push(`simple signal: "${this.truncateMatch(prompt, pattern)}"`);
   425→      }
   426→    }
   427→
   428→    for (const pattern of COMPLEXITY_SIGNALS.moderate.patterns) {
   429→      if (pattern.test(prompt)) {
   430→        moderateScore += 2;
   431→        indicators.push(`moderate signal: "${this.truncateMatch(prompt, pattern)}"`);
   432→      }
   433→    }
   434→
   435→    for (const pattern of COMPLEXITY_SIGNALS.complex.patterns) {
   436→      if (pattern.test(prompt)) {
   437→        complexScore += 3;
   438→        indicators.push(`complex signal: "${this.truncateMatch(prompt, pattern)}"`);
   439→      }
   440→    }
   441→
   442→    // --- Length-based scoring ---
   443→    const length = prompt.length;
   444→
   445→    if (length <= COMPLEXITY_SIGNALS.simple.lengthThreshold) {
   446→      simpleScore += 1;
   447→      indicators.push(`short prompt (${length} chars)`);
   448→    } else if (length <= COMPLEXITY_SIGNALS.moderate.lengthThreshold) {
   449→      moderateScore += 1;
   450→      indicators.push(`medium prompt (${length} chars)`);
   451→    } else if (length <= COMPLEXITY_SIGNALS.complex.lengthThreshold) {
   452→      moderateScore += 1;
   453→      complexScore += 1;
   454→      indicators.push(`long prompt (${length} chars)`);
   455→    } else {
   456→      complexScore += 2;
   457→      indicators.push(`very long prompt (${length} chars)`);
   458→    }
   459→
   460→    // --- Step-count scoring ---
   461→    const stepCount = this.countSteps(prompt);
   462→    if (stepCount <= COMPLEXITY_SIGNALS.simple.maxSteps) {
   463→      simpleScore += 1;
   464→    } else if (stepCount <= COMPLEXITY_SIGNALS.moderate.maxSteps) {
   465→      moderateScore += 1;
   466→      indicators.push(`multi-step request (${stepCount} steps detected)`);
   467→    } else {
   468→      complexScore += 2;
   469→      indicators.push(`many-step request (${stepCount} steps detected)`);
   470→    }
   471→
   472→    // --- Final decision ---
   473→    const maxScore = Math.max(simpleScore, moderateScore, complexScore);
   474→
   475→    // Ties go to the higher complexity (safer default)
   476→    if (complexScore === maxScore) return 'complex';
   477→    if (moderateScore === maxScore) return 'moderate';
   478→    return 'simple';
   479→  }
   480→
   481→  /**
   482→   * Count the approximate number of distinct steps or sub-tasks in a prompt.
   483→   */
   484→  private countSteps(prompt: string): number {
   485→    let steps = 1; // at least one step
   486→    for (const pattern of STEP_INDICATORS) {
   487→      const matches = prompt.match(pattern);
   488→      if (matches) {
   489→        steps += matches.length;
   490→      }
   491→    }
   492→    return steps;
   493→  }
   494→
   495→  /**
   496→   * Extract a short snippet from the prompt that matched a pattern,
   497→   * for use as a human-readable indicator.
   498→   */
   499→  private truncateMatch(prompt: string, pattern: RegExp): string {
   500→    const match = prompt.match(pattern);
   501→    if (!match) return '(no match)';
   502→    const text = match[0];
   503→    return text.length > 40 ? text.slice(0, 37) + '...' : text;
   504→  }

   506→  // -----------------------------------------------------------------------
   507→  // Private: model selection
   508→  // -----------------------------------------------------------------------
   509→
   510→  private selectModel(complexity: Complexity, category: TaskCategory, effort: EffortLevel): string {
   511→    // 1. User override takes absolute precedence
   512→    if (this.forcedModel) {
   513→      return this.forcedModel;
   514→    }
   515→
   516→    // 2. Effort-level model mapping takes precedence when the effort is
   517→    //    explicitly above "medium". This lets users escalate thinking
   518→    //    without changing the prompt.
   519→    if (effort === 'ultrathink' || effort === 'high') {
   520→      const effortModel = this.config.effortModels[effort];
   521→      if (effortModel && this.availableModels[effortModel]) {
   522→        return effortModel;
   523→      }
   524→    }
   525→
   526→    // 3. Category-specific override
   527→    const categoryModel = this.config.categoryOverrides[category];
   528→    if (categoryModel && this.availableModels[categoryModel]) {
   529→      return categoryModel;
   530→    }
   531→
   532→    // 4. Effort-level mapping for low/medium
   533→    if (effort === 'low' || effort === 'medium') {
   534→      const effortModel = this.config.effortModels[effort];
   535→      if (effortModel && this.availableModels[effortModel]) {
   536→        return effortModel;
   537→      }
   538→    }
   539→
   540→    // 5. Complexity-based default
   541→    switch (complexity) {
   542→      case 'simple':
   543→        return this.config.simpleModel;
   544→      case 'moderate':
   545→        return this.config.moderateModel;
   546→      case 'complex':
   547→        return this.config.complexModel;
   548→    }
   549→  }
   550→
   551→  /**
   552→   * Return alternative models that could handle the same task but were not
   553→   * selected as the primary choice.
   554→   */
   555→  private getAlternatives(complexity: Complexity, category: TaskCategory): string[] {
   556→    const selected = this.selectModel(complexity, category, this.currentEffort);
   557→    const candidates = new Set<string>();
   558→
   559→    // Add complexity-adjacent models
   560→    candidates.add(this.config.simpleModel);
   561→    candidates.add(this.config.moderateModel);
   562→    candidates.add(this.config.complexModel);
   563→
   564→    // Add effort models
   565→    for (const model of Object.values(this.config.effortModels)) {
   566→      candidates.add(model);
   567→    }
   568→
   569→    // Add category override if present
   570→    const categoryModel = this.config.categoryOverrides[category];
   571→    if (categoryModel) {
   572→      candidates.add(categoryModel);
   573→    }
   574→
   575→    // Remove the selected model and any models not in the available set
   576→    candidates.delete(selected);
   577→    for (const candidate of [...candidates]) {
   578→      if (!this.availableModels[candidate]) {
   579→        candidates.delete(candidate);
   580→      }
   581→    }
   582→
   583→    return [...candidates];
   584→  }

   586→  // -----------------------------------------------------------------------
   587→  // Private: reasoning string
   588→  // -----------------------------------------------------------------------
   589→
   590→  private buildReasoning(
   591→    analysis: PromptAnalysis,
   592→    effort: EffortLevel,
   593→    model: string,
   594→  ): string {
   595→    const parts: string[] = [];
   596→
   597→    if (this.forcedModel) {
   598→      parts.push(`User override forced model to "${this.forcedModel}".`);
   599→      return parts.join(' ');
   600→    }
   601→
   602→    parts.push(`Prompt classified as ${analysis.complexity} ${analysis.category} task.`);
   603→
   604→    switch (analysis.complexity) {
   605→      case 'simple':
   606→        parts.push('Routed to fast model for quick response.');
   607→        break;
   608→      case 'moderate':
   609→        parts.push('Routed to balanced model for capable response.');
   610→        break;
   611→      case 'complex':
   612→        parts.push('Routed to powerful model for thorough analysis.');
   613→        break;
   614→    }
   615→
   616→    if (effort === 'ultrathink') {
   617→      parts.push('Ultrathink mode: using highest-capability model for deep reasoning.');
   618→    } else if (effort === 'high') {
   619→      parts.push('High effort: using advanced model for detailed output.');
   620→    } else if (effort === 'low') {
   621→      parts.push('Low effort: using fast model for concise response.');
   622→    }
   623→
   624→    parts.push(`Selected "${model}".`);
   625→
   626→    return parts.join(' ');
   627→  }

   629→  // -----------------------------------------------------------------------
   630→  // Private: config validation
   631→  // -----------------------------------------------------------------------
   632→
   633→  private validateConfig(): void {
   634→    const requiredModels: string[] = [
   635→      this.config.defaultModel,
   636→      this.config.simpleModel,
   637→      this.config.moderateModel,
   638→      this.config.complexModel,
   639→    ];
   640→
   641→    for (const modelId of requiredModels) {
   642→      if (!this.availableModels[modelId]) {
   643→        throw new Error(
   644→          `RouterConfig references model "${modelId}" which is not in the available models list. ` +
   645→          `Available: ${Object.keys(this.availableModels).join(', ')}`,
   646→        );
   647→      }
   648→    }
   649→
   650→    for (const [level, modelId] of Object.entries(this.config.effortModels)) {
   651→      if (!this.availableModels[modelId]) {
   652→        throw new Error(
   653→          `RouterConfig effortModels.${level} references model "${modelId}" which is not in the available models list. ` +
   654→          `Available: ${Object.keys(this.availableModels).join(', ')}`,
   655→        );
   656→      }
   657→    }
   658→
   659→    for (const [category, modelId] of Object.entries(this.config.categoryOverrides)) {
   660→      if (modelId && !this.availableModels[modelId]) {
   661→        throw new Error(
   662→          `RouterConfig categoryOverrides.${category} references model "${modelId}" which is not in the available models list. ` +
   663→          `Available: ${Object.keys(this.availableModels).join(', ')}`,
   664→        );
   665→      }
   666→    }
   667→
   668→    if (this.config.maxTokenBudget <= 0) {
   669→      throw new Error('RouterConfig.maxTokenBudget must be a positive integer.');
   670→    }
   671→  }
   672→}
