   314→   314→// ---- Interactive Mode ----
   315→   315→async function startInteractive(options: any) {
   316→   316→  const config = initConfig(options.apiKey);
   317→   317→  if (options.model) config.defaultModel = options.model;
   318→   318→  if (options.theme) config.ui.theme = options.theme as 'dark' | 'light' | 'dracula' | 'nord';
   319→   319→  if (options.noStreaming) config.ui.streaming = false;
   320→   320→  if (options.permissionMode) config.permissionMode = options.permissionMode;
   321→   321→  if (options.diffPreview === true) config.diffPreview = true;
   322→   322→  if (options.diffPreview === false) config.diffPreview = false;
   323→   323→  if (options.cache === true) config.promptCache.enabled = true;
   324→   324→  if (options.cache === false) config.promptCache.enabled = false;
   325→   325→  if (options.spendingLimit) config.spendingLimit = options.spendingLimit;
   326→   326→  if (options.autoApprove) {
   327→   327→    config.tools.autoApprove = [...config.tools.autoApprove, ...config.tools.requireApproval];
   328→   328→    config.tools.requireApproval = [];
   329→   329→    config.permissionMode = 'yolo';
   330→   330→  }
   331→   331→
   332→   332→  // Check API key
   333→   333→  if (!config.apiKey && !options.ollama) {
   334→   334→    console.log(chalk.red('\nOpenRouter API key not configured!'));
   335→   335→    console.log(chalk.yellow('\nSet it with:'));
   336→   336→    console.log(`  ${chalk.cyan('neuro config --set-key YOUR_API_KEY')}`);
   337→   337→    console.log(`  ${chalk.cyan('export OPENROUTER_API_KEY=YOUR_API_KEY')}`);
   338→   338→    console.log(`  ${chalk.cyan('neuro -k YOUR_API_KEY')}`);
   339→   339→    console.log(chalk.gray('\nGet your key at: https://openrouter.ai/keys\n'));
   340→   340→    process.exit(1);
   341→   341→  }
   342→   342→
   343→   343→  // Initialize engine
   344→   344→  const engine = new NeuroEngine(config);
   345→   345→
   346→   346→  // Set effort level if specified
   347→   347→  if (options.effort) {
   348→   348→    engine.modelRouter.setEffort(options.effort);
   349→   349→  }
   350→   350→
   351→   351→  // Set output style if specified
   352→   352→  if (options.style) {
   353→   353→    engine.styleManager.setStyle(options.style);
   354→   354→  }
   355→   355→
   356→   356→  // Enable thinking if specified
   357→   357→  if (options.thinking) {
   358→   358→    engine.extendedThinking.setMode('full');
   359→   359→    engine.extendedThinking.toggleDisplay();
   360→   360→  }
   361→   361→
   362→   362→  // Enable sandbox if specified
   363→   363→  if (options.sandbox) {
   364→   364→    engine.sandbox.enable();
   365→   365→  }
   366→   366→
   367→   367→  // Enable prompt cache if specified
   368→   368→  if (options.cache) {
   369→   369→    config.promptCache.enabled = true;
   370→   370→  }
   371→   371→
   372→   372→  // Resume session if requested
   373→   373→  if (options.resume) {
   374→   374→    const session = engine.sessionManager.load(options.resume);
   375→   375→    if (session) {
   376→   376→      engine.ui.success(`Resumed session: ${session.id.slice(0, 20)}...`);
   377→   377→      if (options.fork) {
   378→   378→        const forked = engine.sessionManager.create(process.cwd(), config.defaultModel);
   379→   379→        forked.messages = [...session.messages];
   380→   380→        engine.sessionManager.save();
   381→   381→        engine.ui.success(`Forked to new session: ${forked.id.slice(0, 20)}...`);
   382→   382→      }
   383→   383→    } else {
   384→   384→      engine.ui.error(`Session not found: ${options.resume}`);
   385→   385→    }
   386→   386→  } else if (options.continue) {
   387→   387→    const sessions = engine.sessionManager.list();
   388→   388→    if (sessions.length > 0) {
   389→   389→      const session = engine.sessionManager.load(sessions[0].id);
   390→   390→      if (session) {
   391→   391→        engine.ui.success(`Continued session: ${session.id.slice(0, 20)}...`);
   392→   392→      }
   393→   393→    } else {
   394→   394→      engine.ui.warning('No sessions found to continue');
   395→   395→    }
   396→   396→  }
   397→   397→
   398→   398→  // Initialize auto-updater and check for updates in background
   399→   399→  const updater = new AutoUpdater({ currentVersion: VERSION });
   400→   400→  const updateCheck = updater.checkOnStartup(); // Fire and forget — don't block startup
   401→   401→
   402→   402→  // Print banner
   403→   403→  engine.ui.banner();
   404→   404→  const theme = engine.ui.theme;
   405→   405→  const permMode = engine.approval.getMode();
