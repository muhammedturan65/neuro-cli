   278→  /** Enable auto mode – skip all approval prompts, run safety checks in background */

   292→    }
   293→    this.emitStatus('disabled');

   295→  }

   307→  // -----------------------------------------------------------------------
   308→  // Configuration
   309→  // -----------------------------------------------------------------------

   311→  getConfig(): Readonly<AutoModeConfig> {
   312→    return Object.freeze({ ...this.config });

   314→
   315→  updateConfig(updates: Partial<AutoModeConfig>): void {
   316→    // Merge blocked lists so ALWAYS_BLOCKED are never removed
   317→    if (updates.blockedCommands) {
   318→      const merged = new Set([...ALWAYS_BLOCKED_COMMANDS, ...updates.blockedCommands]);
   319→      updates.blockedCommands = Array.from(merged);
   320→    }
   321→    if (updates.blockedPatterns) {
   322→      const merged = new Set([...ALWAYS_BLOCKED_PATTERNS, ...updates.blockedPatterns]);
   323→      updates.blockedPatterns = Array.from(merged);
   324→    }
   325→    Object.assign(this.config, updates);
   326→    this.persistState();
   327→  }
   328→
   329→  setSafetyLevel(level: SafetyLevel): void {
   330→    this.config.safetyLevel = level;
   331→    // Re-apply the pattern / command block lists for the new level
   332→    this.config.blockedCommands = this.buildBlockedCommands(level);
   333→    this.config.blockedPatterns = this.buildBlockedPatterns(level);
   334→    this.persistState();
   335→  }

   337→  // -----------------------------------------------------------------------
   338→  // Core autonomous execution
   339→  // -----------------------------------------------------------------------
   340→
   341→  /**
   342→   * Run a task fully autonomously.
   343→   *
   344→   * The engine will be called repeatedly until:
   345→   * - maxIterations is reached
   346→   * - maxCost is exceeded
   347→   * - maxTimeMs is exceeded
   348→   * - The task appears complete (engine signals no further action needed)
   349→   * - An error occurs and pauseOnError is true
   350→   * - The user aborts
   351→   */
   352→  async executeAuto(
   353→    prompt: string,
   354→    engine: AutoModeEngine,
   355→    maxIterations?: number,
   356→  ): Promise<AutoModeResult> {
   357→    if (this.running) {
   358→      return { ok: false, iterations: 0, finalText: '', error: 'Auto mode is already running a task' };
   359→    }
   360→
   361→    const limit = maxIterations ?? this.config.maxIterations;
   362→    this.running = true;
   363→    this.abortController = new AbortController();
   364→    this.startTime = Date.now();
   365→    this.stats = this.freshStats();
   366→
   367→    this.emitStatus('running');
   368→
   369→    const result: AutoModeResult = { ok: true, iterations: 0, finalText: '' };
   370→
   371→    try {
   372→      let currentPrompt = prompt;
   373→
   374→      for (let i = 0; i < limit; i++) {
   375→        // ---- Abort check ----
   376→        if (this.abortController.signal.aborted) {
   377→          result.ok = false;
   378→          result.error = 'Aborted by user';
   379→          break;
   380→        }
   381→
   382→        // ---- Time limit ----
   383→        if (this.config.maxTimeMs > 0 && Date.now() - this.startTime > this.config.maxTimeMs) {
   384→          result.error = `Time limit of ${this.config.maxTimeMs}ms exceeded`;
   385→          break;
   386→        }
   387→
   388→        // ---- Cost limit ----
   389→        if (this.config.maxCost > 0 && this.stats.totalCost >= this.config.maxCost) {
   390→          result.error = `Cost limit of $${this.config.maxCost} exceeded`;
   391→          break;
   392→        }
   393→
   394→        // ---- Safety pre-check ----
   395→        const safetyResult = this.preFlightSafetyCheck(currentPrompt);
   396→        if (!safetyResult.allowed) {
   397→          this.stats.blockedOperations++;
   398→          result.error = `Safety check blocked execution: ${safetyResult.reason}`;
   399→          if (this.config.pauseOnError) break;
   400→          // If not pausing on error, try to continue with a modified prompt
   401→          currentPrompt = `The previous action was blocked for safety reasons (${safetyResult.reason}). ` +
   402→            `Please try a different, safer approach to accomplish the goal. Original task: ${prompt}`;
   403→          continue;
   404→        }
   405→
   406→        // ---- Execute ----
   407→        try {
   408→          const response = await engine.runPrompt(currentPrompt);
   409→          this.stats.iterations = i + 1;
   410→          this.stats.filesChanged += response.filesChanged;
   411→          this.stats.commandsRun += response.commandsRun;
   412→          this.stats.totalCost += response.cost;
   413→          this.stats.timeElapsedMs = Date.now() - this.startTime;
   414→
   415→          if (response.error) {
   416→            this.stats.errors++;
   417→            if (this.config.pauseOnError) {
   418→              result.error = response.error;
   419→              break;
   420→            }
   421→          }
   422→
   423→          result.iterations = i + 1;
   424→          result.finalText = response.text;
   425→
   426→          // ---- Auto-commit ----
   427→          if (this.config.autoCommit && response.filesChanged > 0) {
   428→            this.autoGitCommit(i + 1);
   429→          }
   430→
   431→          // ---- Auto-test ----
   432→          if (this.config.autoTest && response.filesChanged > 0) {
   433→            const testResult = this.autoRunTests();
   434→            if (!testResult.passed && this.config.pauseOnError) {
   435→              result.error = `Tests failed after iteration ${i + 1}`;
   436→              break;
   437→            }
   438→          }
   439→
   440→          // ---- Checkpoint ----
   441→          if (i > 0 && i % 5 === 0) {
   442→            this.createCheckpoint(i, undefined, undefined);
   443→          }
   444→
   445→          // ---- Detect completion ----
   446→          if (this.detectCompletion(response.text)) {
   447→            result.completed = true;
   448→            break;
   449→          }
   450→
   451→          // Feed the response back as context for the next iteration
   452→          currentPrompt =
   453→            `Continuing autonomous task. Previous iteration result:\n` +
   454→            `${response.text.slice(0, 2000)}\n\n` +
   455→            `If the task is complete, respond with "TASK_COMPLETE". ` +
   456→            `Otherwise continue working towards: ${prompt}`;
   457→
   458→        } catch (err) {
   459→          this.stats.errors++;
   460→          if (this.config.pauseOnError) {
   461→            result.ok = false;
   462→            result.error = `Execution error: ${err instanceof Error ? err.message : String(err)}`;
   463→            break;
   464→          }
   465→          // Otherwise swallow and continue
   466→        }
   467→      }
   468→    } finally {
   469→      this.running = false;
   470→      this.abortController = null;
   471→      this.emitStatus(this.config.enabled ? 'enabled' : 'disabled');
   472→      this.persistState();
   473→    }
   474→
   475→    return result;
   476→  }

   481→      this.abortController.abort();

   489→  /** Set a high-level goal and let the agent work towards it autonomously */
   490→  setGoal(name: string, description: string, successCriteria: string[] = []): GoalDefinition {
   491→    const goal: GoalDefinition = {
   492→      id: randomUUID(),
   493→      name,
   494→      description,
   495→      successCriteria,
   496→      subGoals: [],
   497→      createdAt: new Date().toISOString(),
   498→      status: 'pending',
   499→      progress: 0,
   500→    };
   501→    this.goals.set(goal.id, goal);
   502→    this.persistGoals();
   503→    return goal;
   504→  }

   506→  /** Get a goal by ID */
   507→  getGoal(goalId: string): GoalDefinition | undefined {
   508→    return this.goals.get(goalId);
   509→  }

   511→  /** List all goals */
   512→  listGoals(): GoalDefinition[] {
   513→    return Array.from(this.goals.values());
   514→  }
   515→
   516→  /** Update goal progress */
   517→  updateGoalProgress(goalId: string, progress: number, status?: GoalDefinition['status']): void {
   518→    const goal = this.goals.get(goalId);
   519→    if (!goal) return;
   520→    goal.progress = Math.min(1, Math.max(0, progress));
   521→    if (status) goal.status = status;
   522→    if (goal.progress >= 1 && goal.status !== 'failed' && goal.status !== 'cancelled') {
   523→      goal.status = 'completed';
   524→      goal.completedAt = new Date().toISOString();
   525→    }
   526→    this.persistGoals();
   527→  }
   528→
   529→  /** Add a sub-goal */
   530→  addSubGoal(parentId: string, name: string, description: string): GoalDefinition | null {
   531→    const parent = this.goals.get(parentId);
   532→    if (!parent) return null;
   533→    const sub: GoalDefinition = {
   534→      id: randomUUID(),
   535→      name,
   536→      description,
   537→      successCriteria: [],
   538→      subGoals: [],
   539→      createdAt: new Date().toISOString(),
   540→      status: 'pending',
   541→      progress: 0,
   542→    };
   543→    parent.subGoals.push(sub);
   544→    this.goals.set(sub.id, sub); // flat index too
   545→    this.persistGoals();
   546→    return sub;
   547→  }
   548→
   549→  /** Delete a goal */
   550→  deleteGoal(goalId: string): boolean {
   551→    if (!this.goals.has(goalId)) return false;
   552→    // Also remove from any parent's subGoals
   553→    for (const g of Array.from(this.goals.values())) {
   554→      g.subGoals = g.subGoals.filter((s: GoalDefinition) => s.id !== goalId);
   555→    }
   556→    this.goals.delete(goalId);
   557→    this.persistGoals();
   558→    return true;
   559→  }
   560→
   561→  /**
   562→   * Execute a goal autonomously – will iterate until the goal is reached
   563→   * or limits are hit
   564→   */
   565→  async executeGoal(goalId: string, engine: AutoModeEngine): Promise<AutoModeResult> {
   566→    const goal = this.goals.get(goalId);
   567→    if (!goal) return { ok: false, iterations: 0, finalText: '', error: `Goal "${goalId}" not found` };
   568→
   569→    goal.status = 'in_progress';
   570→    this.persistGoals();
   571→
   572→    const prompt = this.buildGoalPrompt(goal);
   573→    const result = await this.executeAuto(prompt, engine, this.config.maxIterations);
   574→
   575→    if (result.completed) {
   576→      goal.status = 'completed';
   577→      goal.progress = 1;
   578→      goal.completedAt = new Date().toISOString();
   579→    } else if (result.error) {
   580→      goal.status = 'failed';
   581→    }
   582→    this.persistGoals();
   583→
   584→    return result;
   585→  }

   636→  /** Replay a saved routine */
   637→  async executeRoutine(routineId: string, engine: AutoModeEngine): Promise<AutoModeResult> {
   638→    const routine = this.routines.get(routineId);
   639→    if (!routine) return { ok: false, iterations: 0, finalText: '', error: `Routine "${routineId}" not found` };
   640→    if (routine.steps.length === 0) return { ok: false, iterations: 0, finalText: '', error: 'Routine has no steps' };
   641→
   642→    routine.runCount++;
   643→    routine.lastRunAt = new Date().toISOString();
   644→    this.persistRoutines();
   645→
   646→    const overallResult: AutoModeResult = { ok: true, iterations: 0, finalText: '' };
   647→    const stepResults: string[] = [];
   648→
   649→    for (let i = 0; i < routine.steps.length; i++) {
   650→      const step = routine.steps[i];
   651→
   652→      // Build context from previous steps
   653→      const contextPrompt = stepResults.length > 0
   654→        ? `Context from previous steps:\n${stepResults.map((r, idx) => `Step ${idx + 1}: ${r.slice(0, 500)}`).join('\n')}\n\n`
   655→        : '';
   656→
   657→      const fullPrompt = `${contextPrompt}Step ${i + 1}/${routine.steps.length}: ${step.prompt}`;
   658→
   659→      const stepMaxIter = step.maxIterations ?? Math.max(5, Math.floor(this.config.maxIterations / routine.steps.length));
   660→      const result = await this.executeAuto(fullPrompt, engine, stepMaxIter);
   661→
   662→      overallResult.iterations += result.iterations;
   663→
   664→      if (!result.ok) {
   665→        overallResult.ok = false;
   666→        overallResult.error = `Routine failed at step ${i + 1}: ${result.error}`;
   667→        break;
   668→      }
   669→
   670→      stepResults.push(result.finalText);
   671→      overallResult.finalText = result.finalText;
   672→
   673→      if (step.pauseAfter && i < routine.steps.length - 1) {
   674→        // In auto mode we just log; in interactive mode a real prompt would appear
   675→        // eslint-disable-next-line no-console
   676→        console.log(`[auto-mode] Routine "${routine.name}" paused after step ${i + 1}. Continuing automatically in auto mode.`);
   677→      }
   678→    }
   679→
   680→    if (overallResult.ok) {
   681→      overallResult.completed = true;
   682→    }
   683→
   684→    return overallResult;
   685→  }

   691→  /**
   692→   * Pre-flight safety check – examines the prompt for dangerous commands
   693→   * or patterns that should be blocked even in auto mode.
   694→   */
   695→  preFlightSafetyCheck(prompt: string): { allowed: boolean; reason?: string } {
   696→    const lower = prompt.toLowerCase();
   697→
   698→    // Check always-blocked commands
   699→    for (const cmd of this.config.blockedCommands) {
   700→      if (lower.includes(cmd.toLowerCase())) {
   701→        return { allowed: false, reason: `Blocked command detected: "${cmd}"` };
   702→      }
   703→    }
   704→
   705→    // Safety-level-specific checks for conservative / moderate
   706→    if (this.config.safetyLevel === 'conservative') {
   707→      for (const cmd of CONSERVATIVE_BLOCKED_COMMANDS) {
   708→        if (lower.includes(cmd.toLowerCase())) {
   709→          return { allowed: false, reason: `Conservative mode blocked command: "${cmd}"` };
   710→        }
   711→      }
   712→    } else if (this.config.safetyLevel === 'moderate') {
   713→      for (const cmd of MODERATE_BLOCKED_COMMANDS) {
   714→        if (lower.includes(cmd.toLowerCase())) {
   715→          return { allowed: false, reason: `Moderate mode blocked command: "${cmd}"` };
   716→        }
   717→      }
   718→    }
   719→    // 'aggressive' – only always-blocked commands are blocked
   720→
   721→    // Check blocked file patterns
   722→    for (const pattern of this.config.blockedPatterns) {
   723→      // Simple substring check for path-like strings in prompt
   724→      if (lower.includes(pattern.replace(/\*\*/g, '').replace(/\*/g, '').toLowerCase())) {
   725→        return { allowed: false, reason: `Blocked file pattern detected: "${pattern}"` };
   726→      }
   727→    }
   728→
   729→    return { allowed: true };
   730→  }

   732→  /**
   733→   * Runtime safety check for a specific command about to be executed.
   734→   * This is meant to be called by the engine before running any shell command.
   735→   */
   736→  isCommandAllowed(command: string): { allowed: boolean; reason?: string } {
   737→    const trimmed = command.trim();
   738→
   739→    // Always-blocked
   740→    for (const blocked of ALWAYS_BLOCKED_COMMANDS) {
   741→      if (trimmed.includes(blocked) || trimmed.startsWith(blocked.split(' ')[0])) {
   742→        return { allowed: false, reason: `Command blocked (always): "${blocked}"` };
   743→      }
   744→    }
   745→
   746→    // Config-level blocked
   747→    for (const blocked of this.config.blockedCommands) {
   748→      if (trimmed.includes(blocked)) {
   749→        return { allowed: false, reason: `Command blocked (config): "${blocked}"` };
   750→      }
   751→    }
   752→
   753→    // Safety-level checks
   754→    const levelLists: Record<SafetyLevel, string[]> = {
   755→      conservative: CONSERVATIVE_BLOCKED_COMMANDS,
   756→      moderate: MODERATE_BLOCKED_COMMANDS,
   757→      aggressive: [],
   758→    };
   759→    for (const blocked of levelLists[this.config.safetyLevel]) {
   760→      if (trimmed.includes(blocked)) {
   761→        return { allowed: false, reason: `Command blocked (${this.config.safetyLevel}): "${blocked}"` };
   762→      }
   763→    }
   764→
   765→    return { allowed: true };
   766→  }
   767→
   768→  /**
   769→   * Runtime safety check for a file path about to be modified.
   770→   * This is meant to be called by the engine before any file write.
   771→   */
   772→  isFileModificationAllowed(filePath: string): { allowed: boolean; reason?: string } {
   773→    const normalised = filePath.replace(/\\/g, '/');
   774→
   775→    for (const pattern of this.config.blockedPatterns) {
   776→      if (this.matchesGlobPattern(normalised, pattern)) {
   777→        return { allowed: false, reason: `File pattern blocked: "${pattern}"` };
   778→      }
   779→    }
   780→
   781→    // Safety-level-specific patterns
   782→    const patternLists: Record<SafetyLevel, string[]> = {
   783→      conservative: CONSERVATIVE_BLOCKED_PATTERNS,
   784→      moderate: [],
   785→      aggressive: [],
   786→    };
   787→    for (const pattern of patternLists[this.config.safetyLevel]) {
   788→      if (this.matchesGlobPattern(normalised, pattern)) {
   789→        return { allowed: false, reason: `File pattern blocked (${this.config.safetyLevel}): "${pattern}"` };
   790→      }
   791→    }
   792→
   793→    return { allowed: true };
   794→  }

   825→  /** Create a checkpoint of the current state */
   826→  createCheckpoint(iteration: number, goalId?: string, routineId?: string): AutoModeCheckpoint {
   827→    const cp: AutoModeCheckpoint = {
   828→      id: randomUUID(),
   829→      timestamp: new Date().toISOString(),
   830→      iteration,
   831→      goalId,
   832→      routineId,
   833→      snapshot: JSON.stringify({
   834→        stats: this.stats,
   835→        config: { ...this.config, enabled: false }, // don't persist enabled state in snapshot
   836→      }),
   837→    };
   838→    this.checkpoints.push(cp);
   839→
   840→    // Persist to disk
   841→    try {
   842→      if (!existsSync(CHECKPOINTS_DIR)) mkdirSync(CHECKPOINTS_DIR, { recursive: true });
   843→      const cpFile = join(CHECKPOINTS_DIR, `checkpoint-${cp.id}.json`);
   844→      writeFileSync(cpFile, JSON.stringify(cp, null, 2), 'utf-8');
   845→    } catch {
   846→      // Best-effort persistence
   847→    }
   848→
   849→    // Keep at most 50 checkpoints in memory
   850→    if (this.checkpoints.length > 50) {
   851→      const removed = this.checkpoints.shift();
   852→      if (removed) {
   853→        try {
   854→          const cpFile = join(CHECKPOINTS_DIR, `checkpoint-${removed.id}.json`);
   855→          if (existsSync(cpFile)) unlinkSync(cpFile);
   856→        } catch { /* ignore */ }
   857→      }
   858→    }
   859→
   860→    return cp;
   861→  }

   885→  private buildBlockedCommands(level: SafetyLevel): string[] {
   886→    const base = new Set(ALWAYS_BLOCKED_COMMANDS);
   887→    if (level === 'conservative') {
   888→      CONSERVATIVE_BLOCKED_COMMANDS.forEach(c => base.add(c));
   889→    } else if (level === 'moderate') {
   890→      MODERATE_BLOCKED_COMMANDS.forEach(c => base.add(c));
   891→    }
   892→    // Merge user-defined on top
   893→    this.config.blockedCommands.forEach(c => base.add(c));
   894→    return Array.from(base);
   895→  }

   897→  private buildBlockedPatterns(level: SafetyLevel): string[] {

   901→    }

   906→  private detectCompletion(text: string): boolean {

   910→      'task is complete',
   911→      'goal achieved',

   948→  private autoRunTests(): { passed: boolean; output: string } {
   949→    // Try common test commands
   950→    const testCommands = ['npm test', 'bun test', 'yarn test', 'pnpm test'];
   951→    for (const cmd of testCommands) {
   952→      try {
   953→        const output = execSync(cmd, { stdio: 'pipe', timeout: 60_000 }).toString('utf-8');
   954→        return { passed: true, output };
   955→      } catch (err: unknown) {
   956→        // execSync throws on non-zero exit
   957→        if (err instanceof Error && 'stdout' in err) {
   958→          const output = String((err as unknown as { stdout: Buffer }).stdout);
   959→          return { passed: false, output };
   960→        }
   961→        // Command not found – try the next one
   962→        continue;
   963→      }
   964→    }
   965→    return { passed: true, output: 'No test runner found' };
   966→  }

   968→  private matchesGlobPattern(filePath: string, pattern: string): boolean {
   969→    // Convert simple glob to regex
   970→    const regexStr = pattern
   971→      .replace(/[.+^${}()|[\]\\]/g, '\\$&')   // escape regex special chars (except * and ?)
   972→      .replace(/\*\*/g, '{{GLOBSTAR}}')          // preserve **
   973→      .replace(/\*/g, '[^/]*')                   // * matches anything except /
   974→      .replace(/\?/g, '[^/]')                    // ? matches single non-/
   975→      .replace(/\{\{GLOBSTAR\}\}/g, '.*');       // ** matches anything
   976→    try {
   977→      const regex = new RegExp(regexStr + '$', 'i');
   978→      return regex.test(filePath);
   979→    } catch {
   980→      return false;
   981→    }
   982→  }

   988→  private persistState(): void {

   991→      writeFileSync(AUTO_STATE_FILE, JSON.stringify({

   998→  private loadState(): void {
   999→    try {
  1000→      if (existsSync(AUTO_STATE_FILE)) {
  1001→        const data = JSON.parse(readFileSync(AUTO_STATE_FILE, 'utf-8'));
  1002→        if (data.config) Object.assign(this.config, data.config);
  1003→        if (data.stats) Object.assign(this.stats, data.stats);
  1004→      }
  1005→    } catch { /* ignore corrupt state */ }
  1006→
  1007→    // Load goals
  1008→    try {
  1009→      if (existsSync(GOALS_FILE)) {
  1010→        const data = JSON.parse(readFileSync(GOALS_FILE, 'utf-8')) as GoalDefinition[];
  1011→        for (const g of data) this.goals.set(g.id, g);
  1012→      }
  1013→    } catch { /* ignore */ }
  1014→
  1015→    // Load routines
  1016→    try {
  1017→      if (existsSync(ROUTINES_FILE)) {
  1018→        const data = JSON.parse(readFileSync(ROUTINES_FILE, 'utf-8')) as RoutineDefinition[];
  1019→        for (const r of data) this.routines.set(r.id, r);
  1020→      }
  1021→    } catch { /* ignore */ }
  1022→
  1023→    // Load checkpoints from disk
  1024→    try {
  1025→      if (existsSync(CHECKPOINTS_DIR)) {
  1026→        const files = readdirSync(CHECKPOINTS_DIR).filter(f => f.startsWith('checkpoint-'));
  1027→        for (const f of files.slice(-50)) {
  1028→          try {
  1029→            const cp = JSON.parse(readFileSync(join(CHECKPOINTS_DIR, f), 'utf-8')) as AutoModeCheckpoint;
  1030→            this.checkpoints.push(cp);
  1031→          } catch { /* skip corrupt */ }
  1032→        }
  1033→        // Sort by timestamp
  1034→        this.checkpoints.sort((a, b) => new Date(a.timestamp).getTime() - new Date(b.timestamp).getTime());
  1035→      }
  1036→    } catch { /* ignore */ }
  1037→  }
  1038→
  1039→  private persistGoals(): void {
  1040→    try {
  1041→      if (!existsSync(NEURO_DIR)) mkdirSync(NEURO_DIR, { recursive: true });
  1042→      writeFileSync(GOALS_FILE, JSON.stringify(Array.from(this.goals.values()), null, 2), 'utf-8');
  1043→    } catch { /* best effort */ }
  1044→  }

  1050→    } catch { /* best effort */ }
  1051→  }
