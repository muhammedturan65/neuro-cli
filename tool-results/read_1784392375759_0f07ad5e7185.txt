   299→// ---- Interactive Mode ----
   300→async function startInteractive(options: any) {
   301→  const config = initConfig(options.apiKey);
   302→  if (options.model) config.defaultModel = options.model;
   303→  if (options.theme) config.ui.theme = options.theme as 'dark' | 'light' | 'dracula' | 'nord';
   304→  if (options.noStreaming) config.ui.streaming = false;
   305→  if (options.permissionMode) config.permissionMode = options.permissionMode;
   306→  if (options.diffPreview === true) config.diffPreview = true;
   307→  if (options.diffPreview === false) config.diffPreview = false;
   308→  if (options.cache === true) config.promptCache.enabled = true;
   309→  if (options.cache === false) config.promptCache.enabled = false;
   310→  if (options.spendingLimit) config.spendingLimit = options.spendingLimit;
   311→  if (options.autoApprove) {
   312→    config.tools.autoApprove = [...config.tools.autoApprove, ...config.tools.requireApproval];
   313→    config.tools.requireApproval = [];
   314→    config.permissionMode = 'yolo';
   315→  }
   316→
   317→  // Check API key
   318→  if (!config.apiKey && !options.ollama) {
   319→    console.log(chalk.red('\nOpenRouter API key not configured!'));
   320→    console.log(chalk.yellow('\nSet it with:'));
   321→    console.log(`  ${chalk.cyan('neuro config --set-key YOUR_API_KEY')}`);
   322→    console.log(`  ${chalk.cyan('export OPENROUTER_API_KEY=YOUR_API_KEY')}`);
   323→    console.log(`  ${chalk.cyan('neuro -k YOUR_API_KEY')}`);
   324→    console.log(chalk.gray('\nGet your key at: https://openrouter.ai/keys\n'));
   325→    process.exit(1);
   326→  }
   327→
   328→  // Initialize engine
   329→  const engine = new NeuroEngine(config);
   330→
   331→  // Set effort level if specified
   332→  if (options.effort) {
   333→    engine.modelRouter.setEffort(options.effort);
   334→  }
   335→
   336→  // Set output style if specified
   337→  if (options.style) {
   338→    engine.styleManager.setStyle(options.style);
   339→  }
   340→
   341→  // Enable thinking if specified
   342→  if (options.thinking) {
   343→    engine.extendedThinking.setMode('full');
   344→    engine.extendedThinking.toggleDisplay();
   345→  }
   346→
   347→  // Enable sandbox if specified
   348→  if (options.sandbox) {
   349→    engine.sandbox.enable();
   350→  }
   351→
   352→  // Enable prompt cache if specified
   353→  if (options.cache) {
   354→    config.promptCache.enabled = true;
   355→  }
   356→
   357→  // Resume session if requested
   358→  if (options.resume) {
   359→    const session = engine.sessionManager.load(options.resume);
   360→    if (session) {
   361→      engine.ui.success(`Resumed session: ${session.id.slice(0, 20)}...`);
   362→      if (options.fork) {
   363→        const forked = engine.sessionManager.create(process.cwd(), config.defaultModel);
   364→        forked.messages = [...session.messages];
   365→        engine.sessionManager.save();
   366→        engine.ui.success(`Forked to new session: ${forked.id.slice(0, 20)}...`);
   367→      }
   368→    } else {
   369→      engine.ui.error(`Session not found: ${options.resume}`);
   370→    }
   371→  } else if (options.continue) {
   372→    const sessions = engine.sessionManager.list();
   373→    if (sessions.length > 0) {
   374→      const session = engine.sessionManager.load(sessions[0].id);
   375→      if (session) {
   376→        engine.ui.success(`Continued session: ${session.id.slice(0, 20)}...`);
   377→      }
   378→    } else {
   379→      engine.ui.warning('No sessions found to continue');
   380→    }
   381→  }
   382→
   383→  // Initialize auto-updater and check for updates in background
   384→  const updater = new AutoUpdater({ currentVersion: VERSION });
   385→  const updateCheck = updater.checkOnStartup(); // Fire and forget — don't block startup
   386→
   387→  // Print banner
   388→  engine.ui.banner();
   389→  const theme = engine.ui.theme;
   390→  const permMode = engine.approval.getMode();
   391→  const permIcon = engine.approval.getModeIcon();
   392→  const activeStyle = engine.styleManager.getStyle();
   393→  console.log(theme.muted(`  Model: ${MODELS[config.defaultModel]?.name || config.defaultModel}`));
   394→  console.log(theme.muted(`  Permission: ${permIcon} ${permMode}`));
   395→  console.log(theme.muted(`  Style: ${activeStyle.name}`));
   396→  console.log(theme.muted(`  Thinking: ${engine.extendedThinking.getMode()}`));
   397→  console.log(theme.muted(`  Cache: ${config.promptCache.enabled ? 'on' : 'off'}`));
   398→  console.log(theme.muted(`  Working Dir: ${process.cwd()}`));
   399→  console.log(theme.muted(`  Type /help for commands, Tab for completion, Ctrl+C to exit\n`));
   400→
   401→  // Show update notification if available (after banner)
   402→  updateCheck.then((result) => {
   403→    if (result && result.hasUpdate) {
   404→      updater.showUpdateNotification(result);
   405→    }
   406→  }).catch(() => { /* silently ignore */ });
   407→
   408→  // Create readline with tab completion
   409→  const completionEngine = new CompletionEngine(process.cwd());
   410→  completionEngine.setAgentNames(Array.from(engine.agents.keys()));
   411→
   412→  const rl = createInterface({
   413→    input: process.stdin,
   414→    output: process.stdout,
   415→    prompt: chalk.cyan('❯ '),
   416→    historySize: 100,
   417→    completer: completionEngine.complete,
   418→  });
   419→
   420→  rl.prompt();
   421→
   422→  let currentMode: 'auto' | 'agent' | 'direct' = 'auto';
   423→  let currentAgent: string | undefined;
   424→
   425→  rl.on('line', async (line) => {
   426→    const input = line.trim();
   427→    if (!input) { rl.prompt(); return; }
   428→
   429→    // Add to history
   430→    completionEngine.addHistory(input);
   431→
   432→    // Handle slash commands
   433→    if (input.startsWith('/')) {
   434→      const [cmd, ...args] = input.slice(1).split(' ');
   435→
   436→      switch (cmd) {
   437→        case 'help':
   438→          printHelp(engine);
   439→          break;
   440→
   441→        case 'model':
   442→          if (args[0]) {
   443→            engine.switchModel(args[0]);
   444→            engine.fallback.addFallbackModel(config.defaultModel);
   445→          } else {
   446→            engine.ui.modelList(config.defaultModel);
   447→          }
   448→          break;
   449→
   450→        case 'agent':
   451→          if (args[0]) {
   452→            const agentNames = Array.from(engine.agents.keys());
   453→            const found = agentNames.find(n => n.toLowerCase() === args[0].toLowerCase());
   454→            if (found) { currentAgent = found; currentMode = 'direct'; engine.ui.success(`Switched to agent: ${found}`); }
   455→            else { engine.ui.error(`Agent not found. Available: ${agentNames.join(', ')}`); }
   456→          } else {
   457→            const agents = Array.from(engine.agents.entries()).map(([name]) => ({ name, description: '', model: config.defaultModel }));
   458→            engine.ui.agentList(agents);
   459→          }
   460→          break;
   461→
   462→        case 'auto':
   463→          currentMode = 'auto'; currentAgent = undefined;
   464→          engine.ui.success('Mode: Auto (smart orchestration)');
   465→          break;
   466→
   467→        case 'orchestrate':
   468→        case 'plan':
   469→          currentMode = 'agent'; currentAgent = undefined;
   470→          engine.ui.success('Mode: Multi-agent orchestration');
   471→          break;
   472→
   473→        case 'direct':
   474→          currentMode = 'direct';
   475→          engine.ui.success('Mode: Direct (single agent)');
   476→          break;
   477→
   478→        case 'plan-mode':
   479→          engine.approval.setMode('plan');
   480→          engine.ui.success(`${engine.approval.getModeIcon()} Permission: plan (read-only)`);
   481→          break;
   482→
   483→        case 'stats':
   484→          const stats = engine.getSessionStats();
   485→          engine.ui.sessionStats(stats.inputTokens, stats.outputTokens, stats.cost);
   486→          break;
   487→
   488→        case 'clear':
   489→          console.clear(); engine.ui.banner();
   490→          break;
   491→
   492→        case 'theme':
   493→          if (args[0]) {
   494→            config.ui.theme = args[0] as 'dark' | 'light' | 'dracula' | 'nord';

   918→              engine.ui.info('Could not create PR (gh CLI not available or not a GitHub repo)');
   919→            }
   920→          } catch (e) { engine.ui.error(`Git operation failed: ${e instanceof Error ? e.message : String(e)}`); }
   921→          break;
   922→
   923→        case 'code-review':
   924→          engine.ui.info('Starting multi-agent code review...');
   925→          try {
   926→            const reviewResult = await engine.processMessage(
   927→              'Perform a thorough code review of all recent changes in this repository. Check for: bugs, security issues, performance problems, code style, test coverage. Provide findings with severity levels (CRITICAL, WARNING, INFO).',
   928→              'agent',
   929→            );
   930→          } catch (e) { engine.ui.error(`Code review failed: ${e instanceof Error ? e.message : String(e)}`); }
   931→          break;
   932→
   933→        case 'feedback':
   934→          console.log(chalk.bold('\nFeedback:\n'));
   935→          console.log('  Report issues: https://github.com/neuro-cli/neuro/issues');
   936→          console.log('  Discussions:   https://github.com/neuro-cli/neuro/discussions');
   937→          console.log(chalk.gray('\n  Your feedback helps make NeuroCLI better!'));
   938→          console.log();
   939→          break;
   940→
   941→        case 'cost':
   942→          engine.spendingMonitor.printReport();
   943→          if (engine.promptCache) {
   944→            console.log(chalk.bold('\nCache Savings:'));
   945→            engine.promptCache.printStats();
   946→          }
   947→          break;
   948→
   949→        case 'update':
   950→        case 'upgrade':
   951→          const updateSub = args[0];
   952→          if (updateSub === 'now') {
   953→            engine.ui.info('Updating NeuroCLI...');
   954→            const updateResult = await updater.performUpdate();
   955→            if (updateResult.success) {
   956→              engine.ui.success(updateResult.message);
   957→              console.log(chalk.yellow('  Please restart NeuroCLI to use the new version.'));
   958→            } else {
   959→              engine.ui.error(updateResult.message);
   960→            }
   961→          } else if (updateSub === 'check') {
   962→            engine.ui.info('Checking for updates...');
   963→            const checkResult = await updater.checkForUpdate(true);
   964→            if (checkResult.hasUpdate) {
   965→              updater.showUpdateDetails(checkResult);
   966→              updater.showUpdateNotification(checkResult);
   967→            } else {
   968→              updater.showUpToDate();
   969→            }
   970→          } else if (updateSub === 'dismiss') {
   971→            const checkResult = await updater.checkForUpdate(true);
   972→            if (checkResult.hasUpdate) {
   973→              updater.dismissVersion(checkResult.latestVersion);
   974→              engine.ui.success(`Dismissed update notification for v${checkResult.latestVersion}`);
   975→            } else {
   976→              engine.ui.info('No update to dismiss');
   977→            }
   978→          } else if (updateSub === 'auto') {
   979→            const enableAuto = args[1] !== 'off';
   980→            updater.setAutoUpdate(enableAuto);
   981→            engine.ui.success(`Auto-update: ${enableAuto ? 'enabled' : 'disabled'}`);
   982→          } else if (updateSub === 'interval') {
   983→            const hours = args[1] ? parseFloat(args[1]) : 24;
   984→            if (isNaN(hours) || hours < 1) {
   985→              engine.ui.error('Interval must be at least 1 hour');
   986→            } else {
   987→              updater.setCheckInterval(hours);
   988→              engine.ui.success(`Update check interval set to ${hours} hours`);
   989→            }
   990→          } else if (updateSub === 'reset') {
   991→            updater.resetDismissed();
   992→            updater.forceNextCheck();
   993→            engine.ui.success('Update preferences reset');
   994→          } else {
   995→            // Default: interactive update flow
   996→            await updater.interactiveUpdate();
   997→          }
   998→          break;
   999→
  1000→        case 'exit':
  1001→        case 'quit':
  1002→        case 'q':
  1003→          engine.mcpClient.disconnectAll().catch(() => {});
  1004→          engine.approval.close();
  1005→          engine.ui.info('Goodbye!');
  1006→          process.exit(0);
  1007→          break;
  1008→
  1009→        default:
  1010→          engine.ui.error(`Unknown command: /${cmd}. Type /help for available commands.`);
  1011→      }
  1012→
  1013→      rl.prompt();
  1014→      return;
  1015→    }
  1016→
  1017→    // Process message with the engine
  1018→    try {
  1019→      await engine.processMessage(input, currentMode, currentAgent);
  1020→    } catch (error) {
  1021→      engine.ui.error(error instanceof Error ? error.message : String(error));
  1022→    }
  1023→
  1024→    rl.prompt();
  1025→  });
  1026→
  1027→  rl.on('close', () => {
  1028→    engine.mcpClient.disconnectAll().catch(() => {});
  1029→    engine.approval.close();
  1030→    engine.ui.info('Goodbye!');
  1031→    process.exit(0);
  1032→  });
  1033→}
