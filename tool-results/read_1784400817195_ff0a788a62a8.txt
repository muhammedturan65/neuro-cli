   760→  /**
   761→   * Process a user message
   762→   */
   763→  async processMessage(
   764→    message: string,
   765→    mode: 'auto' | 'agent' | 'direct' = 'auto',
   766→    targetAgent?: string,
   767→  ): Promise<{ content: string; usage: TokenUsage; execution?: AgentExecution }> {
   768→    // Check spending limit via spending monitor
   769→    const limitCheck = this.spendingMonitor.checkLimit();
   770→    if (!limitCheck.allowed) {
   771→      this.ui.error(`Spending limit reached (${limitCheck.limitReached}). Use /spending to check or /config to adjust.`);
   772→      return { content: 'Spending limit reached.', usage: { inputTokens: 0, outputTokens: 0, cost: 0 } };
   773→    }
   774→
   775→    // Auto-activate skills based on prompt
   776→    const activatedSkills = this.skillSystem.autoActivate(message);
   777→    if (activatedSkills.length > 0) {
   778→      for (const skill of activatedSkills) {
   779→        this.ui.info(`Skill activated: ${skill.skill.name} (${skill.activatedBy})`);
   780→      }
   781→    }
   782→
   783→    // Model routing (if auto mode)
   784→    let routeDecision: RoutingDecision | null = null;
   785→    if (mode === 'auto') {
   786→      routeDecision = this.modelRouter.route(message);
   787→      if (routeDecision.model !== this.config.defaultModel) {
   788→        this.ui.info(`Model router: ${routeDecision.complexity} task -> ${MODELS[routeDecision.model]?.name || routeDecision.model}`);
   789→      }
   790→    }
   791→
   792→    // Check prompt cache
   793→    if (this.promptCache && this.config.promptCache.enabled) {
   794→      const session = this.sessionManager.getCurrent();
   795→      if (session) {
   796→        const cached = this.promptCache.get(
   797→          routeDecision?.model || this.config.defaultModel,
   798→          session.messages,
   799→        );
   800→        if (cached) {
   801→          this.ui.info('Cache hit - using cached response');
   802→          return {
   803→            content: cached.response,
   804→            usage: { inputTokens: cached.inputTokens, outputTokens: cached.outputTokens, cost: 0 },
   805→          };
   806→        }
   807→      }
   808→    }
   809→
   810→    // Start or get session
   811→    let session = this.sessionManager.getCurrent();
   812→    if (!session) {
   813→      session = this.sessionManager.create(process.cwd(), this.config.defaultModel);
   814→    }
   815→
   816→    // Add user message to session
   817→    this.sessionManager.addMessage({
   818→      role: 'user',
   819→      content: message,
   820→      timestamp: Date.now(),
   821→    });
   822→
   823→    // Build system prompt additions from skills and styles
   824→    const skillAdditions = this.skillSystem.getSystemPromptAdditions();
   825→    const styleAddition = this.styleManager.getSystemPromptAddition();
   826→    const thinkingAddition = this.extendedThinking.getSystemPromptAddition();
   827→
   828→    // Build UI callbacks
   829→    const callbacks: AgentCallbacks = {
   830→      onThinking: (thinking) => this.ui.thinking(thinking),
   831→      onToken: (token) => this.ui.streamingToken(token),
   832→      onToolCall: (name, args) => {
   833→        // Check .neuroignore for file paths
   834→        if (args.path && typeof args.path === 'string' && this.neuroIgnore.isIgnored(args.path)) {
   835→          this.ui.warning(`Ignored path: ${args.path} (matches .neuroignore rule)`);
   836→          return;
   837→        }
   838→        // Sandbox check before tool execution
   839→        if (this.sandbox.isEnabled() && !this.checkSandboxForTool(name, args)) {
   840→          return;
   841→        }
   842→        this.ui.toolCall(name, args);
   843→
   844→        // Record in undo/redo for file modification tools
   845→        if (['write_file', 'edit_file', 'apply_diff', 'delete_file'].includes(name)) {
   846→          // The undo/redo push will happen in the tool execution result handler
   847→        }
   848→      },
   849→      onToolResult: (name, result, isError) => this.ui.toolResult(name, result, isError),
   850→      onApprovalNeeded: async (name, args, risk) => {
   851→        return this.handleApproval(name, args, risk as 'low' | 'medium' | 'high');
   852→      },
   853→      onIteration: (i, max) => {
   854→        this.ui.info(`Iteration ${i}/${max}`);
   855→      },
   856→    };
   857→
   858→    let result;
   859→    const activeModel = routeDecision?.model || this.config.defaultModel;
   860→
   861→    if (mode === 'direct' && targetAgent) {
   862→      const agent = this.agents.get(targetAgent);
   863→      if (!agent) {
   864→        this.ui.error(`Agent not found: ${targetAgent}`);
   865→        return { content: '', usage: { inputTokens: 0, outputTokens: 0, cost: 0 } };
   866→      }
   867→      // Override agent model if defaultModel differs from agent's model
   868→      const originalModel = agent.configModel;
   869→      if (this.config.defaultModel !== originalModel) {
   870→        agent.configModel = this.config.defaultModel;
   871→      }
   872→      this.ui.startStreaming();
   873→      try {
   874→        result = await agent.run(message, callbacks);
   875→      } finally {
   876→        this.ui.endStreaming();
   877→      }
   878→      // Restore original model
   879→      agent.configModel = originalModel;
   880→    } else if (mode === 'agent') {
   881→      const orchestrateResult = await this.orchestrator.orchestrate(message, callbacks);
   882→      result = {
   883→        content: orchestrateResult.content,
   884→        toolCallsMade: 0,
   885→        iterations: orchestrateResult.execution.iterations,
   886→        usage: orchestrateResult.totalUsage,
   887→        execution: orchestrateResult.execution,
   888→      };
   889→    } else {
   890→      const complexity = routeDecision?.complexity || this.assessComplexity(message);
   891→      const category = routeDecision?.category || this.modelRouter.getCategory(message);
   892→
   893→      if (complexity === 'simple') {
   894→        // Route to the best agent based on task category
   895→        const targetAgentName = this.selectAgentForCategory(category);
   896→        const agent = this.agents.get(targetAgentName);
   897→        if (agent) {
   898→          this.ui.info(`Using ${targetAgentName} agent (${category} task)`);
   899→          this.ui.startStreaming();
   900→          try {
   901→            result = await agent.run(message, callbacks);
   902→          } finally {
   903→            this.ui.endStreaming();
   904→          }
   905→        } else {
   906→          // Fallback to Coder
   907→          const coderAgent = this.agents.get('Coder');
   908→          if (coderAgent) {
   909→            this.ui.startStreaming();
   910→            try {
   911→              result = await coderAgent.run(message, callbacks);
   912→            } finally {
   913→              this.ui.endStreaming();
   914→            }
   915→          } else {
   916→            this.ui.warning('No agents initialized, using orchestration mode');
   917→            const orchestrateResult = await this.orchestrator.orchestrate(message, callbacks);
   918→            result = {
   919→              content: orchestrateResult.content,
   920→              toolCallsMade: 0,
   921→              iterations: orchestrateResult.execution.iterations,
   922→              usage: orchestrateResult.totalUsage,
   923→              execution: orchestrateResult.execution,
   924→            };
   925→          }
   926→        }
   927→      } else {
   928→        this.ui.thinking('Analyzing task complexity... Using multi-agent orchestration');
   929→        const orchestrateResult = await this.orchestrator.orchestrate(message, callbacks);
   930→        result = {
   931→          content: orchestrateResult.content,
   932→          toolCallsMade: 0,
   933→          iterations: orchestrateResult.execution.iterations,
   934→          usage: orchestrateResult.totalUsage,
   935→          execution: orchestrateResult.execution,
   936→        };
   937→      }
   938→    }
   939→
   940→    // Parse extended thinking blocks from response
   941→    const thinkingResult = this.extendedThinking.parseResponse(result.content);
   942→    if (thinkingResult.hadThinking && this.extendedThinking.isDisplayEnabled()) {
   943→      // Thinking blocks were already displayed during streaming
   944→    }
   945→    // Use cleaned response (without thinking blocks) if thinking is hidden
   946→    if (thinkingResult.hadThinking && !this.extendedThinking.isDisplayEnabled()) {
   947→      result.content = thinkingResult.cleanedResponse;
   948→    }
   949→
   950→    // Record spending
   951→    const spendResult = this.spendingMonitor.record({
   952→      model: activeModel,
   953→      inputTokens: result.usage.inputTokens,
   954→      outputTokens: result.usage.outputTokens,
   955→      cost: result.usage.cost,
   956→      sessionId: this.sessionManager.getCurrent()?.id || 'unknown',
   957→    });
   958→    if (spendResult.warning) {
   959→      this.ui.warning(spendResult.warning);
   960→    }
   961→    if (!spendResult.allowed) {
   962→      this.ui.error('Spending limit reached. Operation completed but further requests may be blocked.');
   963→    }
   964→
   965→    // Cache the response if caching is enabled
   966→    if (this.promptCache && this.config.promptCache.enabled && session) {
   967→      this.promptCache.set(activeModel, session.messages, result.content, result.usage);
   968→    }
   969→
   970→    // Print usage
   971→    this.ui.tokenUsage(result.usage, activeModel);
   972→
   973→    // Update session
   974→    this.sessionManager.addMessage({
   975→      role: 'assistant',
   976→      content: result.content,
   977→      timestamp: Date.now(),
   978→    });
   979→    this.sessionManager.updateUsage(
   980→      result.usage.inputTokens,
   981→      result.usage.outputTokens,
   982→      result.usage.cost,
   983→    );
   984→    this.sessionManager.save();
   985→
   986→    return {
   987→      content: result.content,
   988→      usage: result.usage,
   989→      execution: result.execution,
   990→    };
   991→  }

   993→  /**
   994→   * Check sandbox permissions for a tool call
   995→   */
   996→  private checkSandboxForTool(toolName: string, args: Record<string, unknown>): boolean {
   997→    // File operations
   998→    if (['write_file', 'edit_file', 'apply_diff'].includes(toolName)) {
   999→      const path = args.path as string;
  1000→      if (path && !this.sandbox.canWrite(path)) {
  1001→        this.ui.warning(`Sandbox: Write access denied for ${path}`);
  1002→        return false;
  1003→      }
  1004→      // Check neuroignore
  1005→      if (path && this.neuroIgnore.isIgnored(path)) {
  1006→        this.ui.warning(`Ignored: ${path} is in .neuroignore`);
  1007→        return false;
  1008→      }
  1009→      // Backup file before modification
  1010→      if (path) this.sandbox.backupFile(path);
  1011→    }
  1012→
  1013→    if (['delete_file'].includes(toolName)) {
  1014→      const path = args.path as string;
  1015→      if (path && !this.sandbox.canDelete(path)) {
  1016→        this.ui.warning(`Sandbox: Delete access denied for ${path}`);
  1017→        return false;
  1018→      }
  1019→      if (path && this.neuroIgnore.isIgnored(path)) {
  1020→        this.ui.warning(`Ignored: ${path} is in .neuroignore`);
  1021→        return false;
  1022→      }
  1023→    }
  1024→
  1025→    if (['read_file', 'search_files', 'list_directory'].includes(toolName)) {
  1026→      const path = (args.path || args.directory) as string;
  1027→      if (path && !this.sandbox.canRead(path)) {
  1028→        this.ui.warning(`Sandbox: Read access denied for ${path}`);
  1029→        return false;
  1030→      }
  1031→      if (path && this.neuroIgnore.isIgnored(path)) {
  1032→        this.ui.warning(`Ignored: ${path} is in .neuroignore`);
  1033→        return false;
  1034→      }
  1035→    }
  1036→
  1037→    // Command execution
  1038→    if (['run_command', 'bash'].includes(toolName)) {
  1039→      const command = args.command as string;
  1040→      if (command && !this.sandbox.canRunCommand(command)) {
  1041→        this.ui.warning('Sandbox: Command execution denied');
  1042→        return false;
  1043→      }
  1044→    }
  1045→
  1046→    // Network access
  1047→    if (['web_search', 'web_fetch'].includes(toolName)) {
  1048→      if (!this.sandbox.canAccessNetwork()) {
  1049→        this.ui.warning('Sandbox: Network access denied');
  1050→        return false;
  1051→      }
  1052→    }
  1053→
  1054→    return true;
  1055→  }
