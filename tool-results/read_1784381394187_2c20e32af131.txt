     1→// ============================================================
     2→// NeuroCLI - Background Session Manager
     3→// Manages long-running sessions that execute independently in
     4→// the background — inspired by Claude Code background sessions
     5→// and Agent View. Supports persistence, output capture, desktop
     6→// notifications, pause/resume, and attach/detach workflows.
     7→// Uses only Node.js built-in modules.
     8→// ============================================================
     9→
    10→import { EventEmitter } from 'events';
    11→import {
    12→  join,
    13→} from 'path';
    14→import {
    15→  existsSync,
    16→  mkdirSync,
    17→  writeFileSync,
    18→  readFileSync,
    19→  unlinkSync,
    20→  readdirSync,
    21→  appendFileSync,
    22→} from 'fs';
    23→import { homedir } from 'os';
    24→import { spawn } from 'child_process';

    26→// ---- Exported Interfaces ----
    27→
    28→export interface BackgroundSessionConfig {
    29→  id: string;
    30→  name: string;
    31→  prompt: string;
    32→  model?: string;
    33→  workingDirectory: string;
    34→  autoApprove: boolean;
    35→  maxIterations: number;
    36→  maxCost: number;
    37→  notifyOnComplete: boolean;
    38→  saveOutput: boolean;
    39→  tags?: string[];
    40→}
    41→
    42→export interface BackgroundSessionState {
    43→  id: string;
    44→  name: string;
    45→  status: 'created' | 'running' | 'paused' | 'completed' | 'failed' | 'stopped';
    46→  createdAt: number;
    47→  startedAt?: number;
    48→  completedAt?: number;
    49→  iterations: number;
    50→  tokensUsed: number;
    51→  cost: number;
    52→  lastOutput?: string;
    53→  outputFilePath?: string;
    54→  tags: string[];
    55→}
    56→
    57→export interface BackgroundSessionOutput {
    58→  sessionId: string;
    59→  lines: string[];
    60→  totalLines: number;
    61→  fetchedAt: number;
    62→}
    63→
    64→export interface BackgroundSessionHistoryEntry {
    65→  timestamp: number;
    66→  type: 'info' | 'iteration' | 'output' | 'error' | 'pause' | 'resume' | 'stop' | 'complete';
    67→  message: string;
    68→  data?: Record<string, unknown>;
    69→}
    70→
    71→// ---- Internal Types ----
    72→
    73→interface SessionRecord {
    74→  config: BackgroundSessionConfig;
    75→  state: BackgroundSessionState;
    76→  history: BackgroundSessionHistoryEntry[];
    77→  abortController: AbortController;
    78→  outputBuffer: string[];
    79→  attached: boolean;
    80→  /** Resolves when the session completes. */
    81→  completionPromise: Promise<void>;
    82→  completionResolve: (() => void) | null;
    83→  completionReject: ((error: Error) => void) | null;
    84→  /** Pause handle — resolves when unpaused. */
    85→  pausePromise: Promise<void> | null;
    86→  pauseResolve: (() => void) | null;
    87→}

    89→// ---- Constants ----
    90→
    91→const SESSION_DIR = join(homedir(), '.neuro', 'background-sessions');
    92→const OUTPUT_DIR = join(homedir(), '.neuro', 'background-output');
    93→const HISTORY_DIR = join(homedir(), '.neuro', 'background-history');
    94→const DEFAULT_MAX_ITERATIONS = 100;
    95→const DEFAULT_MAX_COST = 5.0; // USD
    96→const OUTPUT_BUFFER_MAX = 1000;
    97→const CLEANUP_AGE_MS = 7 * 24 * 60 * 60 * 1000; // 7 days

   100→// BackgroundSessionManager
   101→// ============================================================
   102→
   103→export class BackgroundSessionManager extends EventEmitter {
   104→  private sessions: Map<string, SessionRecord> = new Map();
   105→  private currentlyAttached: string | null = null;
   106→
   107→  constructor() {
   108→    super();
   109→    this.ensureDir(SESSION_DIR);
   110→    this.ensureDir(OUTPUT_DIR);
   111→    this.ensureDir(HISTORY_DIR);
   112→
   113→    // Restore persisted sessions on startup
   114→    this.restoreSessions();
   115→  }
   116→
   117→  // ---- Public API ----
   118→
   119→  /**
   120→   * Create a new background session. The session starts in "created" status.
   121→   * Call `startSession(sessionId)` to begin execution.
   122→   */
   123→  createSession(config: BackgroundSessionConfig): string {
   124→    if (this.sessions.has(config.id)) {
   125→      throw new Error(`Session with id "${config.id}" already exists`);
   126→    }
   127→
   128→    const outputFilePath = config.saveOutput
   129→      ? join(OUTPUT_DIR, `${config.id}.log`)
   130→      : undefined;
   131→
   132→    const state: BackgroundSessionState = {
   133→      id: config.id,
   134→      name: config.name,
   135→      status: 'created',
   136→      createdAt: Date.now(),
   137→      iterations: 0,
   138→      tokensUsed: 0,
   139→      cost: 0,
   140→      outputFilePath,
   141→      tags: config.tags ?? [],
   142→    };
   143→
   144→    let completionResolve: (() => void) | null = null;
   145→    let completionReject: ((error: Error) => void) | null = null;
   146→    const completionPromise = new Promise<void>((resolve, reject) => {
   147→      completionResolve = resolve;
   148→      completionReject = reject;
   149→    });
   150→
   151→    const record: SessionRecord = {
   152→      config,
   153→      state,
   154→      history: [],
   155→      abortController: new AbortController(),
   156→      outputBuffer: [],
   157→      attached: false,
   158→      completionPromise,
   159→      completionResolve,
   160→      completionReject,
   161→      pausePromise: null,
   162→      pauseResolve: null,
   163→    };
   164→
   165→    this.sessions.set(config.id, record);
   166→    this.addHistoryEntry(config.id, 'info', `Session "${config.name}" created`);
   167→
   168→    if (outputFilePath) {
   169→      this.initializeOutputFile(outputFilePath, config);
   170→    }
   171→
   172→    this.persistSession(config.id);
   173→    this.emit('session:created', config.id, state);
   174→
   175→    return config.id;
   176→  }
   177→
   178→  /**
   179→   * Start running a session in the background.
   180→   */
   181→  startSession(sessionId: string): boolean {
   182→    const record = this.sessions.get(sessionId);
   183→    if (!record) return false;
   184→
   185→    if (record.state.status !== 'created' && record.state.status !== 'paused') {
   186→      return false; // Can only start from 'created' or 'paused'
   187→    }
   188→
   189→    if (record.state.status === 'paused' && record.pauseResolve) {
   190→      // Unpause by resolving the pause promise
   191→      record.pauseResolve();
   192→      record.pausePromise = null;
   193→      record.pauseResolve = null;
   194→      record.state.status = 'running';
   195→      this.addHistoryEntry(sessionId, 'resume', `Session "${record.config.name}" resumed`);
   196→      this.persistSession(sessionId);
   197→      this.emit('session:resumed', sessionId, record.state);
   198→      return true;
   199→    }
   200→
   201→    // Fresh start
   202→    record.abortController = new AbortController();
   203→    record.state.status = 'running';
   204→    record.state.startedAt = Date.now();
   205→
   206→    this.addHistoryEntry(sessionId, 'info', `Session "${record.config.name}" started`);
   207→    this.persistSession(sessionId);
   208→    this.emit('session:started', sessionId, record.state);
   209→
   210→    this.runSessionLoop(record)
   211→      .then(() => {
   212→        record.state.status = 'completed';
   213→        record.state.completedAt = Date.now();
   214→        record.completionResolve?.();
   215→        this.addHistoryEntry(sessionId, 'complete', `Session completed after ${record.state.iterations} iterations`);
   216→        this.persistSession(sessionId);
   217→
   218→        if (record.config.notifyOnComplete) {
   219→          this.sendDesktopNotification(
   220→            'NeuroCLI Session Complete',
   221→            `"${record.config.name}" finished successfully (${record.state.iterations} iterations, $${record.state.cost.toFixed(4)}).`
   222→          );
   223→        }
   224→
   225→        this.emit('session:completed', sessionId, record.state);
   226→      })
   227→      .catch((err: unknown) => {
   228→        if (record.state.status === 'stopped' || record.state.status === 'paused') {
   229→          return; // Handled by stop/pause
   230→        }
   231→        const errorMsg = err instanceof Error ? err.message : String(err);
   232→        record.state.status = 'failed';
   233→        record.state.completedAt = Date.now();
   234→        record.state.lastOutput = errorMsg;
   235→        record.completionReject?.(err instanceof Error ? err : new Error(errorMsg));
   236→        this.addHistoryEntry(sessionId, 'error', `Session failed: ${errorMsg}`);
   237→        this.persistSession(sessionId);
   238→
   239→        if (record.config.notifyOnComplete) {
   240→          this.sendDesktopNotification(
   241→            'NeuroCLI Session Failed',
   242→            `"${record.config.name}" encountered an error: ${errorMsg}`
   243→          );
   244→        }
   245→
   246→        this.emit('session:failed', sessionId, errorMsg, record.state);
   247→      });
   248→
   249→    return true;
   250→  }
   251→
   252→  /**
   253→   * Stop a running session. Cannot be resumed.
   254→   */
   255→  stopSession(sessionId: string): boolean {
   256→    const record = this.sessions.get(sessionId);
   257→    if (!record) return false;
   258→
   259→    if (record.state.status !== 'running') return false;
   260→
   261→    record.abortController.abort();
   262→    record.state.status = 'stopped';
   263→    record.state.completedAt = Date.now();
   264→
   265→    // If paused, also resolve the pause so the loop can exit
   266→    if (record.pauseResolve) {
   267→      record.pauseResolve();
   268→      record.pausePromise = null;
   269→      record.pauseResolve = null;
   270→    }
   271→
   272→    this.addHistoryEntry(sessionId, 'stop', `Session "${record.config.name}" stopped by user`);
   273→    this.persistSession(sessionId);
   274→    this.emit('session:stopped', sessionId, record.state);
   275→
   276→    return true;
   277→  }
   278→
   279→  /**
   280→   * Pause a running session. It can be resumed later.
   281→   */
   282→  pauseSession(sessionId: string): boolean {
   283→    const record = this.sessions.get(sessionId);
   284→    if (!record) return false;
   285→
   286→    if (record.state.status !== 'running') return false;
   287→
   288→    // Set up a pause barrier that the loop will wait on
   289→    let pauseResolve: (() => void) | null = null;
   290→    const pausePromise = new Promise<void>((resolve) => {
   291→      pauseResolve = resolve;
   292→    });
   293→
   294→    record.pausePromise = pausePromise;
   295→    record.pauseResolve = pauseResolve;
   296→    record.state.status = 'paused';
   297→
   298→    this.addHistoryEntry(sessionId, 'pause', `Session "${record.config.name}" paused by user`);
   299→    this.persistSession(sessionId);
   300→    this.emit('session:paused', sessionId, record.state);
   301→
   302→    return true;
   303→  }
   304→
   305→  /**
   306→   * Resume a paused session.
   307→   */
   308→  resumeSession(sessionId: string): boolean {
   309→    const record = this.sessions.get(sessionId);
   310→    if (!record) return false;
   311→    if (record.state.status !== 'paused') return false;
   312→
   313→    return this.startSession(sessionId); // startSession handles unpause
   314→  }

   316→  /**
   317→   * Get detailed status of a session.
   318→   */
   319→  getSessionStatus(sessionId: string): BackgroundSessionState | null {
   320→    const record = this.sessions.get(sessionId);
   321→    if (!record) return null;
   322→    return { ...record.state };
   323→  }

   325→  /**
   326→   * List all background sessions.
   327→   */
   328→  listSessions(): BackgroundSessionState[] {
   329→    const result: BackgroundSessionState[] = [];
   330→    const records = Array.from(this.sessions.values());
   331→    for (const record of records) {
   332→      result.push({ ...record.state });
   333→    }
   334→    return result.sort((a, b) => b.createdAt - a.createdAt);
   335→  }

   337→  /**
   338→   * Get recent output from a session.
   339→   * @param lines Number of recent lines to return (default 50).
   340→   */
   341→  getSessionOutput(sessionId: string, lines: number = 50): BackgroundSessionOutput | null {
   342→    const record = this.sessions.get(sessionId);
   343→    if (!record) return null;
   344→
   345→    // If we have an output file, read from it
   346→    if (record.state.outputFilePath && existsSync(record.state.outputFilePath)) {
   347→      try {
   348→        const content = readFileSync(record.state.outputFilePath, 'utf-8');
   349→        const allLines = content.split('\n').filter(Boolean);
   350→        const fetchedLines = allLines.slice(-lines);
   351→        return {
   352→          sessionId,
   353→          lines: fetchedLines,
   354→          totalLines: allLines.length,
   355→          fetchedAt: Date.now(),
   356→        };
   357→      } catch {
   358→        // Fall through to buffer
   359→      }
   360→    }
   361→
   362→    // Fall back to in-memory buffer
   363→    const fetchedLines = record.outputBuffer.slice(-lines);
   364→    return {
   365→      sessionId,
   366→      lines: fetchedLines,
   367→      totalLines: record.outputBuffer.length,
   368→      fetchedAt: Date.now(),
   369→    };
   370→  }

   372→  /**
   373→   * Attach to a background session for interactive monitoring.
   374→   * Only one session can be attached at a time.
   375→   */
   376→  attachSession(sessionId: string): boolean {
   377→    const record = this.sessions.get(sessionId);
   378→    if (!record) return false;
   379→
   380→    // Detach from any currently attached session
   381→    if (this.currentlyAttached && this.currentlyAttached !== sessionId) {
   382→      this.detachSession();
   383→    }
   384→
   385→    record.attached = true;
   386→    this.currentlyAttached = sessionId;
   387→    this.addHistoryEntry(sessionId, 'info', `User attached to session`);
   388→    this.emit('session:attached', sessionId);
   389→
   390→    return true;
   391→  }
   392→
   393→  /**
   394→   * Detach from the currently attached session.
   395→   */
   396→  detachSession(): boolean {
   397→    if (!this.currentlyAttached) return false;
   398→
   399→    const record = this.sessions.get(this.currentlyAttached);
   400→    if (record) {
   401→      record.attached = false;
   402→      this.addHistoryEntry(this.currentlyAttached, 'info', `User detached from session`);
   403→    }
   404→
   405→    this.emit('session:detached', this.currentlyAttached);
   406→    this.currentlyAttached = null;
   407→    return true;
   408→  }

   410→  /**
   411→   * Wait for a session to complete.
   412→   */
   413→  async waitForSession(sessionId: string): Promise<void> {
   414→    const record = this.sessions.get(sessionId);
   415→    if (!record) {
   416→      throw new Error(`Session "${sessionId}" not found`);
   417→    }
   418→
   419→    if (record.state.status === 'completed') return;
   420→    if (record.state.status === 'failed') {
   421→      throw new Error(record.state.lastOutput ?? `Session "${sessionId}" failed`);
   422→    }
   423→    if (record.state.status === 'stopped') {
   424→      throw new Error(`Session "${sessionId}" was stopped`);
   425→    }
   426→
   427→    return record.completionPromise;
   428→  }

   430→  /**
   431→   * Get the full execution history of a session.
   432→   */
   433→  getSessionHistory(sessionId: string): BackgroundSessionHistoryEntry[] | null {
   434→    const record = this.sessions.get(sessionId);
   435→    if (!record) return null;
   436→    return [...record.history];
   437→  }

   439→  /**
   440→   * Clean up old completed/failed/stopped sessions older than
   441→   * the specified age (default 7 days).
   442→   */
   443→  cleanup(maxAgeMs: number = CLEANUP_AGE_MS): number {
   444→    const cutoff = Date.now() - maxAgeMs;
   445→    let removed = 0;
   446→
   447→    const toRemove: string[] = [];
   448→    const entries = Array.from(this.sessions.entries());
   449→    for (const [id, record] of entries) {
   450→      const isTerminal =
   451→        record.state.status === 'completed' ||
   452→        record.state.status === 'failed' ||
   453→        record.state.status === 'stopped';
   454→
   455→      if (!isTerminal) continue;
   456→
   457→      const endTime = record.state.completedAt ?? record.state.createdAt;
   458→      if (endTime < cutoff) {
   459→        toRemove.push(id);
   460→      }
   461→    }
   462→
   463→    for (const id of toRemove) {
   464→      this.removeSession(id);
   465→      removed++;
   466→    }
   467→
   468→    return removed;
   469→  }

   472→   * Get the currently attached session ID (if any).
   473→   */
   474→  getAttachedSessionId(): string | null {
   475→    return this.currentlyAttached;

   478→  /**
   479→   * Find sessions by tag.
   480→   */
   481→  findByTag(tag: string): BackgroundSessionState[] {
   482→    const results: BackgroundSessionState[] = [];
   483→    const records = Array.from(this.sessions.values());
   484→    for (const record of records) {
   485→      if (record.state.tags.includes(tag)) {
   486→        results.push({ ...record.state });
   487→      }
   488→    }

   492→  // ---- Private: Session Execution Loop ----
   493→
   494→  /**
   495→   * The main execution loop for a background session.
   496→   *
   497→   * In production, this would call an LLM, execute tools, handle
   498→   * approvals, etc. Here we simulate with iteration tracking,
   499→   * cost accounting, pause/resume, abort, and output capture.
   500→   */
   501→  private async runSessionLoop(record: SessionRecord): Promise<void> {
   502→    const maxIterations = record.config.maxIterations || DEFAULT_MAX_ITERATIONS;
   503→    const maxCost = record.config.maxCost || DEFAULT_MAX_COST;
   504→    const abortSignal = record.abortController.signal;
   505→
   506→    for (let i = 0; i < maxIterations; i++) {
   507→      // ---- Abort check ----
   508→      if (abortSignal.aborted) {
   509→        throw new Error('Session aborted');
   510→      }
   511→
   512→      // ---- Pause check ----
   513→      if (record.pausePromise) {
   514→        this.addHistoryEntry(record.config.id, 'pause', `Paused at iteration ${i}`);
   515→        await record.pausePromise;
   516→        // After resuming, check abort again
   517→        if (abortSignal.aborted) {
   518→          throw new Error('Session aborted after resume');
   519→        }
   520→      }
   521→
   522→      // ---- Simulate iteration ----
   523→      const iterationTokens = 50 + Math.floor(Math.random() * 200);
   524→      const iterationCost = (iterationTokens / 1_000_000) * 3.0; // simulated $3/1M tokens
   525→
   526→      record.state.iterations = i + 1;
   527→      record.state.tokensUsed += iterationTokens;
   528→      record.state.cost += iterationCost;
   529→
   530→      const outputLine = `[iter ${i + 1}] Processing... (tokens: ${record.state.tokensUsed}, cost: $${record.state.cost.toFixed(4)})`;
   531→      record.state.lastOutput = outputLine;
   532→
   533→      // Capture output
   534→      this.captureOutput(record, outputLine);
   535→
   536→      // Add history entry every 10 iterations to avoid bloat
   537→      if ((i + 1) % 10 === 0 || i === 0) {
   538→        this.addHistoryEntry(
   539→          record.config.id,
   540→          'iteration',
   541→          `Iteration ${i + 1}/${maxIterations}`,
   542→          { tokensUsed: record.state.tokensUsed, cost: record.state.cost }
   543→        );
   544→      }
   545→
   546→      // ---- Cost limit check ----
   547→      if (record.state.cost >= maxCost) {
   548→        const msg = `Cost limit ($${maxCost}) reached at iteration ${i + 1}`;
   549→        this.captureOutput(record, msg);
   550→        this.addHistoryEntry(record.config.id, 'info', msg);
   551→        break;
   552→      }
   553→
   554→      // Persist state periodically (every 5 iterations)
   555→      if ((i + 1) % 5 === 0) {
   556→        this.persistSession(record.config.id);
   557→      }
   558→
   559→      // Yield control for cooperative multitasking
   560→      await this.yieldControl();
   561→    }
   562→
   563→    // Final persist
   564→    this.persistSession(record.config.id);
   565→  }
   566→
   567→  /**
   568→   * Yield control to the event loop.
   569→   */
   570→  private yieldControl(): Promise<void> {
   571→    return new Promise((resolve) => setImmediate(resolve));
   572→  }

   574→  // ---- Private: Output Management ----
   575→
   576→  /**
   577→   * Capture an output line to both the in-memory buffer and
   578→   * the output file (if configured).
   579→   */
   580→  private captureOutput(record: SessionRecord, line: string): void {
   581→    // In-memory buffer (bounded)
   582→    record.outputBuffer.push(line);
   583→    if (record.outputBuffer.length > OUTPUT_BUFFER_MAX) {
   584→      record.outputBuffer.shift();
   585→    }
   586→
   587→    // File output
   588→    if (record.state.outputFilePath) {
   589→      try {
   590→        appendFileSync(record.state.outputFilePath, line + '\n', 'utf-8');
   591→      } catch {
   592→        // Non-critical — best effort file I/O
   593→      }
   594→    }
   595→  }

   597→  /**
   598→   * Initialize an output file with a header.
   599→   */
   600→  private initializeOutputFile(filePath: string, config: BackgroundSessionConfig): void {
   601→    try {
   602→      const header = [
   603→        `=== NeuroCLI Background Session Output ===`,
   604→        `Session: ${config.name} (${config.id})`,
   605→        `Model: ${config.model ?? 'default'}`,
   606→        `Working Directory: ${config.workingDirectory}`,
   607→        `Max Iterations: ${config.maxIterations}`,
   608→        `Max Cost: $${config.maxCost}`,
   609→        `Started: ${new Date().toISOString()}`,
   610→        `==========================================\n`,
   611→      ].join('\n');
   612→
   613→      writeFileSync(filePath, header, 'utf-8');
   614→    } catch {
   615→      // Non-critical
   616→    }
   617→  }

   621→  /**
   622→   * Add a history entry to a session.
   623→   */
   624→  private addHistoryEntry(
   625→    sessionId: string,
   626→    type: BackgroundSessionHistoryEntry['type'],
   627→    message: string,
   628→    data?: Record<string, unknown>
   629→  ): void {
   630→    const record = this.sessions.get(sessionId);
   631→    if (!record) return;
   632→
   633→    record.history.push({
   634→      timestamp: Date.now(),
   635→      type,
   636→      message,
   637→      data,
   638→    });
   639→
   640→    // Persist history to disk (append)
   641→    this.persistHistory(sessionId, {
   642→      timestamp: Date.now(),
   643→      type,
   644→      message,
   645→      data,
   646→    });
   647→  }

   649→  // ---- Private: Persistence ----
   650→
   651→  /**
   652→   * Persist session state to disk so it survives app restarts.
   653→   */
   654→  private persistSession(sessionId: string): void {
   655→    const record = this.sessions.get(sessionId);
   656→    if (!record) return;
   657→
   658→    const filePath = join(SESSION_DIR, `${sessionId}.json`);
   659→    try {
   660→      const data = {
   661→        config: record.config,
   662→        state: record.state,
   663→        // Don't persist runtime objects like AbortController or promises
   664→        persistedAt: Date.now(),
   665→      };
   666→      writeFileSync(filePath, JSON.stringify(data, null, 2), 'utf-8');
   667→    } catch {
   668→      // Non-critical — persistence is best-effort
   669→    }
   670→  }
   671→
   672→  /**
   673→   * Persist a single history entry to the session's history file.
   674→   */
   675→  private persistHistory(sessionId: string, entry: BackgroundSessionHistoryEntry): void {
   676→    const filePath = join(HISTORY_DIR, `${sessionId}.jsonl`);
   677→    try {
   678→      appendFileSync(filePath, JSON.stringify(entry) + '\n', 'utf-8');
   679→    } catch {
   680→      // Non-critical
   681→    }
   682→  }
   683→
   684→  /**
   685→   * Restore sessions from disk on startup.
   686→   * Only restores terminal-state sessions; running sessions
   687→   * from a previous process are marked as 'stopped'.
   688→   */
   689→  private restoreSessions(): void {
   690→    if (!existsSync(SESSION_DIR)) return;
   691→
   692→    try {
   693→      for (const file of readdirSync(SESSION_DIR)) {
   694→        if (!file.endsWith('.json')) continue;
   695→
   696→        const filePath = join(SESSION_DIR, file);
   697→        try {
   698→          const content = readFileSync(filePath, 'utf-8');
   699→          const data = JSON.parse(content) as {
   700→            config: BackgroundSessionConfig;
   701→            state: BackgroundSessionState;
   702→          };
   703→
   704→          // If the session was running when the app exited, mark it stopped
   705→          if (data.state.status === 'running' || data.state.status === 'paused') {
   706→            data.state.status = 'stopped';
   707→            data.state.completedAt = data.state.completedAt ?? Date.now();
   708→          }
   709→
   710→          // Re-create a minimal record (no running loop)
   711→          let completionResolve: (() => void) | null = null;
   712→          let completionReject: ((error: Error) => void) | null = null;
   713→          const completionPromise = new Promise<void>((resolve, reject) => {
   714→            completionResolve = resolve;
   715→            completionReject = reject;
   716→          });
   717→
   718→          // Resolve immediately for terminal-state sessions
   719→          if (
   720→            data.state.status === 'completed' ||
   721→            data.state.status === 'failed' ||
   722→            data.state.status === 'stopped'
   723→          ) {
   724→            // Pre-resolved
   725→            completionPromise.catch(() => {}); // swallow
   726→          }
   727→
   728→          // Restore history from disk if available
   729→          const history = this.loadHistoryFromDisk(data.config.id);
   730→
   731→          const record: SessionRecord = {
   732→            config: data.config,
   733→            state: data.state,
   734→            history,
   735→            abortController: new AbortController(),
   736→            outputBuffer: [],
   737→            attached: false,
   738→            completionPromise,
   739→            completionResolve,
   740→            completionReject,
   741→            pausePromise: null,
   742→            pauseResolve: null,
   743→          };
   744→
   745→          this.sessions.set(data.config.id, record);
   746→        } catch {
   747→          // Corrupted file — skip
   748→        }
   749→      }
   750→    } catch {
   751→      // Session directory read failure — skip
   752→    }
   753→  }
   754→
   755→  /**
   756→   * Load history entries from the JSONL file on disk.
   757→   */
   758→  private loadHistoryFromDisk(sessionId: string): BackgroundSessionHistoryEntry[] {
   759→    const filePath = join(HISTORY_DIR, `${sessionId}.jsonl`);
   760→    if (!existsSync(filePath)) return [];
   761→
   762→    const entries: BackgroundSessionHistoryEntry[] = [];
   763→    try {
   764→      const content = readFileSync(filePath, 'utf-8');
   765→      for (const line of content.split('\n')) {
   766→        if (!line.trim()) continue;
   767→        try {
   768→          entries.push(JSON.parse(line));
   769→        } catch {
   770→          // Skip malformed lines
   771→        }
   772→      }
   773→    } catch {
   774→      // Read failure — return what we have
   775→    }
   776→
   777→    return entries;
   778→  }
   779→
   780→  /**
   781→   * Remove a session and its associated files.
   782→   */
   783→  private removeSession(sessionId: string): void {
   784→    const record = this.sessions.get(sessionId);
   785→    if (!record) return;
   786→
   787→    // Remove session file
   788→    const sessionFile = join(SESSION_DIR, `${sessionId}.json`);
   789→    try { if (existsSync(sessionFile)) unlinkSync(sessionFile); } catch { /* ignore */ }
   790→
   791→    // Remove history file
   792→    const historyFile = join(HISTORY_DIR, `${sessionId}.jsonl`);
   793→    try { if (existsSync(historyFile)) unlinkSync(historyFile); } catch { /* ignore */ }
   794→
   795→    // Remove output file
   796→    if (record.state.outputFilePath) {
   797→      try { if (existsSync(record.state.outputFilePath)) unlinkSync(record.state.outputFilePath); } catch { /* ignore */ }
   798→    }
   799→
   800→    // Detach if currently attached
   801→    if (this.currentlyAttached === sessionId) {
   802→      this.currentlyAttached = null;
   803→    }
   804→
   805→    this.sessions.delete(sessionId);
   806→    this.emit('session:removed', sessionId);
   807→  }

   811→  /**
   812→   * Send a desktop notification using the operating system's
   813→   * native notification mechanism.
   814→   */
   815→  private sendDesktopNotification(title: string, body: string): void {
   816→    try {
   817→      const platform = process.platform;
   818→
   819→      if (platform === 'darwin') {
   820→        // macOS: use osascript to display a notification
   821→        const escapedTitle = title.replace(/"/g, '\\"');
   822→        const escapedBody = body.replace(/"/g, '\\"');
   823→        spawn('osascript', [
   824→          '-e',
   825→          `display notification "${escapedBody}" with title "${escapedTitle}"`,
   826→        ]).unref();
   827→      } else if (platform === 'linux') {
   828→        // Linux: try notify-send
   829→        spawn('notify-send', [title, body]).unref();
   830→      } else if (platform === 'win32') {
   831→        // Windows: use PowerShell toast
   832→        const escapedTitle = title.replace(/'/g, "''");
   833→        const escapedBody = body.replace(/'/g, "''");
   834→        const psCmd = `
   835→          Add-Type -AssemblyName System.Windows.Forms
   836→          $notify = New-Object System.Windows.Forms.NotifyIcon
   837→          $notify.Icon = [System.Drawing.SystemIcons]::Information
   838→          $notify.Visible = $true
   839→          $notify.ShowBalloonTip(5000, '${escapedTitle}', '${escapedBody}', [System.Windows.Forms.ToolTipIcon]::Info)
   840→        `.trim();
   841→        spawn('powershell', ['-NoProfile', '-Command', psCmd], {
   842→          windowsHide: true,
   843→        }).unref();
   844→      }
   845→      // If platform is unsupported, silently skip notification
   846→    } catch {
   847→      // Notification failure must never crash the manager
   848→    }
   849→  }
