   314→// ---- Interactive Mode ----
   315→async function startInteractive(options: any) {
   316→  const config = initConfig(options.apiKey);
   317→  if (options.model) config.defaultModel = options.model;
   318→  if (options.theme) config.ui.theme = options.theme as 'dark' | 'light' | 'dracula' | 'nord';
   319→  if (options.noStreaming) config.ui.streaming = false;
   320→  if (options.permissionMode) config.permissionMode = options.permissionMode;
   321→  if (options.diffPreview === true) config.diffPreview = true;
   322→  if (options.diffPreview === false) config.diffPreview = false;
   323→  if (options.cache === true) config.promptCache.enabled = true;
   324→  if (options.cache === false) config.promptCache.enabled = false;
   325→  if (options.spendingLimit) config.spendingLimit = options.spendingLimit;
   326→  if (options.autoApprove) {
   327→    config.tools.autoApprove = [...config.tools.autoApprove, ...config.tools.requireApproval];
   328→    config.tools.requireApproval = [];
   329→    config.permissionMode = 'yolo';
   330→  }
   331→
   332→  // Check API key
   333→  if (!config.apiKey && !options.ollama) {
   334→    console.log(chalk.red('\nOpenRouter API key not configured!'));
   335→    console.log(chalk.yellow('\nSet it with:'));
   336→    console.log(`  ${chalk.cyan('neuro config --set-key YOUR_API_KEY')}`);
   337→    console.log(`  ${chalk.cyan('export OPENROUTER_API_KEY=YOUR_API_KEY')}`);
   338→    console.log(`  ${chalk.cyan('neuro -k YOUR_API_KEY')}`);
   339→    console.log(chalk.gray('\nGet your key at: https://openrouter.ai/keys\n'));
   340→    process.exit(1);
   341→  }
   342→
   343→  // Initialize engine
   344→  const engine = new NeuroEngine(config);
   345→
   346→  // Set effort level if specified
   347→  if (options.effort) {
   348→    engine.modelRouter.setEffort(options.effort);
   349→  }
   350→
   351→  // Set output style if specified
   352→  if (options.style) {
   353→    engine.styleManager.setStyle(options.style);
   354→  }
   355→
   356→  // Enable thinking if specified
   357→  if (options.thinking) {
   358→    engine.extendedThinking.setMode('full');
   359→    engine.extendedThinking.toggleDisplay();
   360→  }
   361→
   362→  // Enable sandbox if specified
   363→  if (options.sandbox) {
   364→    engine.sandbox.enable();
   365→  }
   366→
   367→  // Enable prompt cache if specified
   368→  if (options.cache) {
   369→    config.promptCache.enabled = true;
   370→  }
   371→
   372→  // Resume session if requested
   373→  if (options.resume) {
   374→    const session = engine.sessionManager.load(options.resume);
   375→    if (session) {
   376→      engine.ui.success(`Resumed session: ${session.id.slice(0, 20)}...`);
   377→      if (options.fork) {
   378→        const forked = engine.sessionManager.create(process.cwd(), config.defaultModel);
   379→        forked.messages = [...session.messages];
   380→        engine.sessionManager.save();
   381→        engine.ui.success(`Forked to new session: ${forked.id.slice(0, 20)}...`);
   382→      }
   383→    } else {
   384→      engine.ui.error(`Session not found: ${options.resume}`);
   385→    }
   386→  } else if (options.continue) {
   387→    const sessions = engine.sessionManager.list();
   388→    if (sessions.length > 0) {
   389→      const session = engine.sessionManager.load(sessions[0].id);
   390→      if (session) {
   391→        engine.ui.success(`Continued session: ${session.id.slice(0, 20)}...`);
   392→      }
   393→    } else {
   394→      engine.ui.warning('No sessions found to continue');
   395→    }
   396→  }
   397→
   398→  // Initialize auto-updater and check for updates in background
   399→  const updater = new AutoUpdater({ currentVersion: VERSION });
   400→  const updateCheck = updater.checkOnStartup(); // Fire and forget — don't block startup
   401→
   402→  // Print banner
   403→  engine.ui.banner();
   404→  const theme = engine.ui.theme;
   405→  const permMode = engine.approval.getMode();
   406→  const permIcon = engine.approval.getModeIcon();
   407→  const activeStyle = engine.styleManager.getStyle();
   408→  console.log(theme.muted(`  Model: ${MODELS[config.defaultModel]?.name || config.defaultModel}`));
   409→  console.log(theme.muted(`  Permission: ${permIcon} ${permMode}`));
   410→  console.log(theme.muted(`  Style: ${activeStyle.name}`));
   411→  console.log(theme.muted(`  Thinking: ${engine.extendedThinking.getMode()}`));
   412→  console.log(theme.muted(`  Cache: ${config.promptCache.enabled ? 'on' : 'off'}`));
   413→  console.log(theme.muted(`  Working Dir: ${process.cwd()}`));
   414→  console.log(theme.muted(`  Type /help for commands, Tab for completion, Ctrl+C to exit\n`));
   415→
   416→  // Show update notification if available (after banner)
   417→  updateCheck.then((result) => {
   418→    if (result && result.hasUpdate) {
   419→      updater.showUpdateNotification(result);
   420→    }
   421→  }).catch(() => { /* silently ignore */ });
   422→
   423→  // Create readline with tab completion
   424→  const completionEngine = new CompletionEngine(process.cwd());
   425→  completionEngine.setAgentNames(Array.from(engine.agents.keys()));
   426→
   427→  const rl = createInterface({
   428→    input: process.stdin,
   429→    output: process.stdout,
   430→    prompt: chalk.cyan('❯ '),
   431→    historySize: 100,
   432→    completer: completionEngine.complete,
   433→  });
   434→
   435→  rl.prompt();
   436→
   437→  // Wire the main readline to the approval system to prevent dual-readline bug
   438→  // (double character input when approval prompt and main rl both read stdin)
   439→  engine.approval.setMainReadline(rl);
   440→
   441→  let currentMode: 'auto' | 'agent' | 'direct' = 'auto';
   442→  let currentAgent: string | undefined;
   443→  let isProcessing = false;
   444→
   445→  rl.on('line', async (line) => {
   446→    try {
   447→    const input = line.trim();
   448→    if (!input) { rl.prompt(); return; }
   449→
   450→    // Prevent input while processing (approval prompt may be active)
   451→    if (isProcessing) {
   452→      return;
   453→    }
   454→
   455→    // Add to history
   456→    completionEngine.addHistory(input);
   457→
   458→    // Handle slash commands
   459→    if (input.startsWith('/')) {
   460→      const [cmd, ...args] = input.slice(1).split(' ');
   461→
   462→      switch (cmd) {
   463→        case 'help':
   464→          printHelp(engine);
   465→          break;
   466→
   467→        case 'model':
   468→          if (args[0]) {
   469→            engine.switchModel(args[0]);
   470→            engine.fallback.addFallbackModel(config.defaultModel);
   471→          } else {
   472→            engine.ui.modelList(config.defaultModel);
   473→          }
   474→          break;
   475→
   476→        case 'agent':
   477→          if (args[0]) {
   478→            const agentNames = Array.from(engine.agents.keys());
   479→            const found = agentNames.find(n => n.toLowerCase() === args[0].toLowerCase());
   480→            if (found) { currentAgent = found; currentMode = 'direct'; engine.ui.success(`Switched to agent: ${found}`); }
   481→            else { engine.ui.error(`Agent not found. Available: ${agentNames.join(', ')}`); }
   482→          } else {
   483→            const agents = Array.from(engine.agents.entries()).map(([name]) => ({ name, description: '', model: config.defaultModel }));
   484→            engine.ui.agentList(agents);
   485→          }
   486→          break;
   487→
   488→        case 'auto':
   489→          currentMode = 'auto'; currentAgent = undefined;

   924→          } catch (e) { engine.ui.error(`Failed to import: ${e instanceof Error ? e.message : String(e)}`); }
   925→          break;
   926→
   927→        case 'commit-push-pr':
   928→          engine.ui.info('Running commit + push + PR workflow...');
   929→          try {
   930→            const { execSync } = await import('child_process');
   931→            // Stage all
   932→            execSync('git add -A', { cwd: process.cwd(), encoding: 'utf-8' });
   933→            // Commit
   934→            const commitMsg = args.join(' ') || 'Update from NeuroCLI';
   935→            execSync(`git commit -m "${commitMsg}" --no-gpg-sign`, { cwd: process.cwd(), encoding: 'utf-8' });
   936→            // Push
   937→            execSync('git push', { cwd: process.cwd(), encoding: 'utf-8' });
   938→            engine.ui.success('Changes committed and pushed');
   939→            // Try to create PR with gh CLI
   940→            try {
   941→              const prResult = execSync(`gh pr create --title "${commitMsg}" --body "Auto-generated by NeuroCLI"`, { cwd: process.cwd(), encoding: 'utf-8' });
   942→              engine.ui.success(`PR created: ${prResult.trim()}`);
   943→            } catch {
   944→              engine.ui.info('Could not create PR (gh CLI not available or not a GitHub repo)');
   945→            }
   946→          } catch (e) { engine.ui.error(`Git operation failed: ${e instanceof Error ? e.message : String(e)}`); }
   947→          break;
   948→
   949→        case 'code-review':
   950→          engine.ui.info('Starting multi-agent code review...');
   951→          try {
   952→            const reviewResult = await engine.processMessage(
   953→              'Perform a thorough code review of all recent changes in this repository. Check for: bugs, security issues, performance problems, code style, test coverage. Provide findings with severity levels (CRITICAL, WARNING, INFO).',
   954→              'agent',
   955→            );
   956→          } catch (e) { engine.ui.error(`Code review failed: ${e instanceof Error ? e.message : String(e)}`); }
   957→          break;
   958→
   959→        case 'feedback':
   960→          console.log(chalk.bold('\nFeedback:\n'));
   961→          console.log('  Report issues: https://github.com/neuro-cli/neuro/issues');
   962→          console.log('  Discussions:   https://github.com/neuro-cli/neuro/discussions');
   963→          console.log(chalk.gray('\n  Your feedback helps make NeuroCLI better!'));
   964→          console.log();
   965→          break;
   966→
   967→        case 'cost':
   968→          engine.spendingMonitor.printReport();
   969→          if (engine.promptCache) {
   970→            console.log(chalk.bold('\nCache Savings:'));
   971→            engine.promptCache.printStats();
   972→          }
   973→          break;
   974→
   975→        case 'update':
   976→        case 'upgrade':
   977→          const updateSub = args[0];
   978→          if (updateSub === 'now') {
   979→            engine.ui.info('Updating NeuroCLI...');
   980→            const updateResult = await updater.performUpdate();
   981→            if (updateResult.success) {
   982→              engine.ui.success(updateResult.message);
   983→              console.log(chalk.yellow('  Please restart NeuroCLI to use the new version.'));
   984→            } else {
   985→              engine.ui.error(updateResult.message);
   986→            }
   987→          } else if (updateSub === 'check') {
   988→            engine.ui.info('Checking for updates...');
   989→            const checkResult = await updater.checkForUpdate(true);
   990→            if (checkResult.hasUpdate) {
   991→              updater.showUpdateDetails(checkResult);
   992→              updater.showUpdateNotification(checkResult);
   993→            } else {
   994→              updater.showUpToDate();
   995→            }
   996→          } else if (updateSub === 'dismiss') {
   997→            const checkResult = await updater.checkForUpdate(true);
   998→            if (checkResult.hasUpdate) {
   999→              updater.dismissVersion(checkResult.latestVersion);
  1000→              engine.ui.success(`Dismissed update notification for v${checkResult.latestVersion}`);
  1001→            } else {
  1002→              engine.ui.info('No update to dismiss');
  1003→            }
  1004→          } else if (updateSub === 'auto') {
  1005→            const enableAuto = args[1] !== 'off';
  1006→            updater.setAutoUpdate(enableAuto);
  1007→            engine.ui.success(`Auto-update: ${enableAuto ? 'enabled' : 'disabled'}`);
  1008→          } else if (updateSub === 'interval') {
  1009→            const hours = args[1] ? parseFloat(args[1]) : 24;
  1010→            if (isNaN(hours) || hours < 1) {
  1011→              engine.ui.error('Interval must be at least 1 hour');
  1012→            } else {
  1013→              updater.setCheckInterval(hours);
  1014→              engine.ui.success(`Update check interval set to ${hours} hours`);
  1015→            }
  1016→          } else if (updateSub === 'reset') {
  1017→            updater.resetDismissed();
  1018→            updater.forceNextCheck();
  1019→            engine.ui.success('Update preferences reset');
  1020→          } else {
  1021→            // Default: interactive update flow
  1022→            await updater.interactiveUpdate();
  1023→          }
  1024→          break;
  1025→
  1026→        case 'exit':
  1027→        case 'quit':
  1028→        case 'q':
  1029→          engine.mcpClient.disconnectAll().catch(() => {});
  1030→          engine.approval.close();
  1031→          engine.ui.info('Goodbye!');
  1032→          process.exit(0);
  1033→          break;
  1034→
  1035→        default:
  1036→          engine.ui.error(`Unknown command: /${cmd}. Type /help for available commands.`);
  1037→      }
  1038→
  1039→      rl.prompt();
  1040→      return;
  1041→    }
  1042→
  1043→    // Process message with the engine
  1044→    try {
  1045→      isProcessing = true;
  1046→      await engine.processMessage(input, currentMode, currentAgent);
  1047→    } catch (error) {
  1048→      engine.ui.error(error instanceof Error ? error.message : String(error));
  1049→    } finally {
  1050→      isProcessing = false;
  1051→    }
  1052→
  1053→    rl.prompt();
  1054→    } catch (error) {
  1055→      // Catch-all for any unhandled error in the line handler
  1056→      console.error(chalk.red(`\n⚠️ Error: ${error instanceof Error ? error.message : String(error)}`));
  1057→      isProcessing = false;
  1058→      rl.prompt();
  1059→    }
  1060→  });
  1061→
  1062→  rl.on('close', async () => {
  1063→    try { await engine.mcpClient.disconnectAll(); } catch {}
  1064→    try { engine.approval.close(); } catch {}
  1065→    try { engine.sessionManager.save(); } catch {}
  1066→    engine.ui.info('Goodbye!');
  1067→    process.exit(0);
  1068→  });
  1069→}
