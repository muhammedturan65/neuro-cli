   257→   257→   257→// ---- Interactive Mode ----
   258→   258→   258→async function startInteractive(options: any) {
   259→   259→   259→  const config = initConfig(options.apiKey);
   260→   260→   260→  if (options.model) config.defaultModel = options.model;
   261→   261→   261→  if (options.theme) config.ui.theme = options.theme as 'dark' | 'light' | 'dracula' | 'nord';
   262→   262→   262→  if (options.noStreaming) config.ui.streaming = false;
   263→   263→   263→  if (options.permissionMode) config.permissionMode = options.permissionMode;
   264→   264→   264→  if (options.diffPreview === true) config.diffPreview = true;
   265→   265→   265→  if (options.diffPreview === false) config.diffPreview = false;
   266→   266→   266→  if (options.cache === true) config.promptCache.enabled = true;
   267→   267→   267→  if (options.cache === false) config.promptCache.enabled = false;
   268→   268→   268→  if (options.spendingLimit) config.spendingLimit = options.spendingLimit;
   269→   269→   269→  if (options.autoApprove) {
   270→   270→   270→    config.tools.autoApprove = [...config.tools.autoApprove, ...config.tools.requireApproval];
   271→   271→   271→    config.tools.requireApproval = [];
   272→   272→   272→    config.permissionMode = 'yolo';
   273→   273→   273→  }
   274→   274→   274→
   275→   275→   275→  // Check API key
   276→   276→   276→  if (!config.apiKey && !options.ollama) {
   277→   277→   277→    console.log(chalk.red('\nOpenRouter API key not configured!'));
   278→   278→   278→    console.log(chalk.yellow('\nSet it with:'));
   279→   279→   279→    console.log(`  ${chalk.cyan('neuro config --set-key YOUR_API_KEY')}`);
   280→   280→   280→    console.log(`  ${chalk.cyan('export OPENROUTER_API_KEY=YOUR_API_KEY')}`);
   281→   281→   281→    console.log(`  ${chalk.cyan('neuro -k YOUR_API_KEY')}`);
   282→   282→   282→    console.log(chalk.gray('\nGet your key at: https://openrouter.ai/keys\n'));
   283→   283→   283→    process.exit(1);
   284→   284→   284→  }
   285→   285→   285→
   286→   286→   286→  // Initialize engine
   287→   287→   287→  const engine = new NeuroEngine(config);
   288→   288→   288→
   289→   289→   289→  // Set effort level if specified
   290→   290→   290→  if (options.effort) {
   291→   291→   291→    engine.modelRouter.setEffort(options.effort);
   292→   292→   292→  }
   293→   293→   293→
   294→   294→   294→  // Set output style if specified
   295→   295→   295→  if (options.style) {
   296→   296→   296→    engine.styleManager.setStyle(options.style);
   297→   297→   297→  }
   298→   298→   298→
   299→   299→   299→  // Enable thinking if specified
   300→   300→   300→  if (options.thinking) {
   301→   301→   301→    engine.extendedThinking.setMode('full');
   302→   302→   302→    engine.extendedThinking.toggleDisplay();
   303→   303→   303→  }
   304→   304→   304→
   305→   305→   305→  // Enable sandbox if specified
   306→   306→   306→  if (options.sandbox) {
   307→   307→   307→    engine.sandbox.enable();
   308→   308→   308→  }
   309→   309→   309→
   310→   310→   310→  // Enable prompt cache if specified
   311→   311→   311→  if (options.cache) {
   312→   312→   312→    config.promptCache.enabled = true;
   313→   313→   313→  }
   314→   314→   314→
   315→   315→   315→  // Resume session if requested
   316→   316→   316→  if (options.resume) {
   317→   317→   317→    const session = engine.sessionManager.load(options.resume);
   318→   318→   318→    if (session) {
   319→   319→   319→      engine.ui.success(`Resumed session: ${session.id.slice(0, 20)}...`);
   320→   320→   320→      if (options.fork) {
   321→   321→   321→        const forked = engine.sessionManager.create(process.cwd(), config.defaultModel);
   322→   322→   322→        forked.messages = [...session.messages];
   323→   323→   323→        engine.sessionManager.save();
   324→   324→   324→        engine.ui.success(`Forked to new session: ${forked.id.slice(0, 20)}...`);
   325→   325→   325→      }
   326→   326→   326→    } else {
   327→   327→   327→      engine.ui.error(`Session not found: ${options.resume}`);
   328→   328→   328→    }
   329→   329→   329→  } else if (options.continue) {
   330→   330→   330→    const sessions = engine.sessionManager.list();
   331→   331→   331→    if (sessions.length > 0) {
   332→   332→   332→      const session = engine.sessionManager.load(sessions[0].id);
   333→   333→   333→      if (session) {
   334→   334→   334→        engine.ui.success(`Continued session: ${session.id.slice(0, 20)}...`);
   335→   335→   335→      }
   336→   336→   336→    } else {
   337→   337→   337→      engine.ui.warning('No sessions found to continue');
   338→   338→   338→    }
   339→   339→   339→  }
   340→   340→   340→
   341→   341→   341→  // Print banner
   342→   342→   342→  engine.ui.banner();
   343→   343→   343→  const theme = engine.ui.theme;
   344→   344→   344→  const permMode = engine.approval.getMode();
   345→   345→   345→  const permIcon = engine.approval.getModeIcon();
