   232→  // Per-role cost tracking
   233→  private roleCosts: Record<string, number> = {
   234→    orchestrator: 0,
   235→    worker: 0,
   236→    evaluator: 0,
   237→    reviewer: 0,
   238→  };
   239→
   240→  // Per-model cost tracking
   241→  private modelCosts: Record<string, number> = {};
   242→
   243→  // Per-task cost entries
   244→  private taskCostEntries: CostTaskEntry[] = [];
   245→
   246→  // Session totals
   247→  private sessionTotalCost: number = 0;
   248→  private sessionTotalTokens: number = 0;
   249→  private sessionStartTime: number;
   250→
   251→  // Dynamic switching: worker failure tracking
   252→  private workerFailures: WorkerFailureRecord[] = [];
   253→  private workerUpgradeUntil: number = 0;
   254→
   255→  // Current effective model per role (may differ from config during fallback)
   256→  private effectiveModels: Record<string, string>;
   257→
   258→  // Budget exceeded flag
   259→  private budgetExceeded: boolean = false;
   260→
   261→  constructor(config: OrchestratorConfig, apiClient: OpenRouterClient) {
   262→    this.config = config;
   263→    this.client = apiClient;
   264→    this.sessionStartTime = Date.now();
   265→
   266→    // Initialize effective models from config
   267→    this.effectiveModels = {
   268→      orchestrator: config.roles.orchestrator.model,
   269→      worker: config.roles.worker.model,
   270→      evaluator: config.roles.evaluator.model,
   271→      reviewer: config.roles.reviewer.model,
   272→    };
   273→
   274→    // Set up fallback chain with all fallback models from all roles
   275→    const allFallbacks = new Set<string>();
   276→    for (const role of Object.values(config.roles)) {
   277→      for (const fb of role.fallbackModels) {
   278→        allFallbacks.add(fb);
   279→      }
   280→    }
   281→    this.fallbackChain = new FallbackChain(apiClient, {
   282→      models: [...allFallbacks],
   283→      maxRetries: 2,
   284→      retryDelayMs: 3000,
   285→      fallbackOnErrors: ['rate_limit', 'overloaded', 'context_length_exceeded', 'timeout', 'server_error'],
   286→    });
   287→
   288→    // Set up spending monitor with budget from config
   289→    this.spendingMonitor = new SpendingMonitor({
   290→      sessionLimit: config.costBudget.maxPerSession,
   291→      dailyLimit: 0,
   292→      warnAtPercent: [config.costBudget.warnThreshold],
   293→      autoStopAtLimit: true,
   294→      trackByModel: true,
   295→    });
   296→  }

   302→  /**
   303→   * Classify a task by complexity and suggest the appropriate model role.
   304→   * Uses a lightweight LLM call when possible; falls back to local heuristics.
   305→   */
   306→  async classifyTask(prompt: string, context?: string): Promise<TaskClassification> {
   307→    const fullPrompt = context ? `${context}\n\n---\n\n${prompt}` : prompt;
   308→
   309→    // Try LLM-based classification using the cheapest available model
   310→    try {
   311→      const classificationModel = this.cheapestAvailableModel();
   312→      const messages: Message[] = [
   313→        { role: 'system', content: CLASSIFICATION_SYSTEM_PROMPT },
   314→        { role: 'user', content: fullPrompt },
   315→      ];
   316→
   317→      const response = await this.client.quickChat(
   318→        classificationModel,
   319→        messages,
   320→        undefined,
   321→        0.1,
   322→      );
   323→
   324→      const parsed = this.parseClassificationResponse(response.content);
   325→      if (parsed) {
   326→        // Record the cost of classification
   327→        this.recordCost('orchestrator', classificationModel, response.usage, 'task-classification');
   328→        return parsed;
   329→      }
   330→    } catch {
   331→      // Fall through to heuristic classification
   332→    }
   333→
   334→    // Local heuristic fallback
   335→    return this.classifyTaskHeuristic(prompt);
   336→  }
   337→
   338→  /**
   339→   * Route a classification to the appropriate model ID.
   340→   */
   341→  routeToModel(classification: TaskClassification): string {
   342→    // Check for dynamic upgrade: if worker has been failing, upgrade
   343→    if (classification.suggestedRole === 'worker' && this.isWorkerUpgraded()) {
   344→      return this.getEffectiveModel('orchestrator');
   345→    }
   346→
   347→    return this.getEffectiveModel(classification.suggestedRole);
   348→  }
   349→
   350→  // ===================================================================
   351→  // Execution Patterns
   352→  // ===================================================================
   353→
   354→  /**
   355→   * Full orchestration pipeline:
   356→   *   orchestrator plans -> worker executes -> evaluator checks -> reviewer reviews
   357→   *
   358→   * Quality gates can re-route to orchestrator if evaluator confidence is low.
   359→   */
   360→  async orchestrateAndExecute(prompt: string, context?: string): Promise<OrchestrationResult> {
   361→    const startTime = Date.now();
   362→    const phases: OrchestrationPhase[] = [];
   363→    let escalated = false;
   364→    let escalationReason: string | undefined;
   365→
   366→    // Check budget before starting
   367→    if (!this.isWithinBudget()) {
   368→      return {
   369→        plan: this.emptyResponse('orchestrator'),
   370→        execution: this.emptyResponse('worker'),
   371→        totalCost: 0,
   372→        totalTokens: 0,
   373→        phases: [],
   374→        escalated: false,
   375→        success: false,
   376→        timestamp: Date.now(),
   377→      };
   378→    }
   379→
   380→    // --- Phase 1: Orchestrator plans ---
   381→    const planStart = Date.now();
   382→    let plan: ModelResponse;
   383→    try {
   384→      plan = await this.callOrchestrator(
   385→        `Plan the approach for the following task. Break it into clear steps.\n\n${context ? `Context:\n${context}\n\n` : ''}Task: ${prompt}`,
   386→      );
   387→    } catch (err) {
   388→      const errMsg = err instanceof Error ? err.message : String(err);
   389→      return {
   390→        plan: this.emptyResponse('orchestrator'),
   391→        execution: this.emptyResponse('worker'),
   392→        totalCost: 0,
   393→        totalTokens: 0,
   394→        phases: [{ role: 'orchestrator', model: this.getEffectiveModel('orchestrator'), inputTokens: 0, outputTokens: 0, cost: 0, durationMs: Date.now() - planStart, success: false }],
   395→        escalated: false,
   396→        escalationReason: `Orchestrator failed: ${errMsg}`,
   397→        success: false,
   398→        timestamp: Date.now(),
   399→      };
   400→    }
   401→
   402→    phases.push({
   403→      role: 'orchestrator',
   404→      model: plan.model,
   405→      inputTokens: plan.usage.inputTokens,
   406→      outputTokens: plan.usage.outputTokens,
   407→      cost: plan.usage.cost,
   408→      durationMs: Date.now() - planStart,
   409→      success: true,
   410→    });
   411→
   412→    // Check budget after orchestrator
   413→    if (!this.isWithinBudget()) {
   414→      return {
   415→        plan,
   416→        execution: this.emptyResponse('worker'),
   417→        evaluation: undefined,
   418→        review: undefined,
   419→        totalCost: this.sumPhaseCosts(phases),
   420→        totalTokens: this.sumPhaseTokens(phases),
   421→        phases,
   422→        escalated: false,
   423→        escalationReason: 'Budget exceeded after orchestrator phase',
   424→        success: false,
   425→        timestamp: Date.now(),
   426→      };
   427→    }
   428→
   429→    // --- Phase 2: Worker executes ---
   430→    const execStart = Date.now();
   431→    const workerPrompt = `Based on this plan:\n\n${plan.content}\n\nExecute the task:\n${prompt}${context ? `\n\nContext:\n${context}` : ''}`;
   432→    let execution: ModelResponse;
   433→    try {
   434→      execution = await this.callWorker(workerPrompt);
   435→    } catch (err) {
   436→      const errMsg = err instanceof Error ? err.message : String(err);
   437→      this.recordWorkerFailure(this.getEffectiveModel('worker'), errMsg);
   438→      phases.push({
   439→        role: 'worker',
   440→        model: this.getEffectiveModel('worker'),
   441→        inputTokens: 0,
   442→        outputTokens: 0,
   443→        cost: 0,
   444→        durationMs: Date.now() - execStart,
   445→        success: false,
   446→      });
   447→      return {
   448→        plan,
   449→        execution: this.emptyResponse('worker'),
   450→        totalCost: this.sumPhaseCosts(phases),
   451→        totalTokens: this.sumPhaseTokens(phases),
   452→        phases,
   453→        escalated: false,
   454→        escalationReason: `Worker failed: ${errMsg}`,
   455→        success: false,
   456→        timestamp: Date.now(),
   457→      };
   458→    }
   459→
   460→    phases.push({
   461→      role: 'worker',
   462→      model: execution.model,
   463→      inputTokens: execution.usage.inputTokens,
   464→      outputTokens: execution.usage.outputTokens,
   465→      cost: execution.usage.cost,
   466→      durationMs: Date.now() - execStart,
   467→      success: true,
   468→    });
   469→
   470→    // Check budget after worker
   471→    if (!this.isWithinBudget()) {
   472→      return {
   473→        plan,
   474→        execution,
   475→        evaluation: undefined,
   476→        review: undefined,
   477→        totalCost: this.sumPhaseCosts(phases),
   478→        totalTokens: this.sumPhaseTokens(phases),
   479→        phases,
   480→        escalated: false,
   481→        escalationReason: 'Budget exceeded after worker phase',
   482→        success: true,
   483→        timestamp: Date.now(),
   484→      };
   485→    }
   486→
   487→    // --- Phase 3: Evaluator checks (if quality gates enabled) ---
   488→    let evaluation: ModelResponse | undefined;
   489→    if (this.config.qualityGates.enabled) {
   490→      const evalStart = Date.now();
   491→      const evalPrompt = `Evaluate the quality of this work output.\n\nOriginal task: ${prompt}\n\nPlan:\n${plan.content}\n\nExecution output:\n${execution.content}\n\nRate your confidence that the execution correctly and completely fulfills the task on a scale of 0.0 to 1.0. Respond with JSON: {"confidence": <number>, "issues": [<string>], "passed": <boolean>}`;
   492→
   493→      try {
   494→        evaluation = await this.callEvaluator(evalPrompt);
   495→
   496→        // Parse evaluator confidence
   497→        const evalResult = this.parseEvaluationResponse(evaluation.content);
   498→        if (evalResult) {
   499→          evaluation.confidence = evalResult.confidence;
   500→
   501→          // If confidence is below threshold, escalate to orchestrator
   502→          if (evalResult.confidence < this.config.qualityGates.minConfidence) {
   503→            escalated = true;
   504→            escalationReason = `Evaluator confidence ${evalResult.confidence.toFixed(2)} below threshold ${this.config.qualityGates.minConfidence}. Issues: ${evalResult.issues.join('; ')}`;
   505→
   506→            // Re-execute with orchestrator model
   507→            const retryStart = Date.now();
   508→            const retryPrompt = `The worker's output was evaluated as insufficient (confidence: ${evalResult.confidence}).\n\nIssues: ${evalResult.issues.join('; ')}\n\nOriginal task: ${prompt}\n\nPrevious plan:\n${plan.content}\n\nPrevious output:\n${execution.content}\n\nPlease produce a better execution that addresses these issues.`;
   509→
   510→            try {
   511→              const retryExecution = await this.callOrchestrator(retryPrompt);
   512→              phases.push({
   513→                role: 'orchestrator',
   514→                model: retryExecution.model,
   515→                inputTokens: retryExecution.usage.inputTokens,
   516→                outputTokens: retryExecution.usage.outputTokens,
   517→                cost: retryExecution.usage.cost,
   518→                durationMs: Date.now() - retryStart,
   519→                success: true,
   520→              });
   521→              execution = retryExecution;
   522→            } catch {
   523→              phases.push({
   524→                role: 'orchestrator',
   525→                model: this.getEffectiveModel('orchestrator'),
   526→                inputTokens: 0,
   527→                outputTokens: 0,
   528→                cost: 0,
   529→                durationMs: Date.now() - retryStart,
   530→                success: false,
   531→              });
   532→            }
   533→          }
   534→        }
   535→
   536→        phases.push({
   537→          role: 'evaluator',
   538→          model: evaluation.model,
   539→          inputTokens: evaluation.usage.inputTokens,
   540→          outputTokens: evaluation.usage.outputTokens,
   541→          cost: evaluation.usage.cost,
   542→          durationMs: Date.now() - evalStart,
   543→          success: true,
   544→        });
   545→      } catch {
   546→        // Evaluator failure is non-fatal; continue without evaluation
   547→        phases.push({
   548→          role: 'evaluator',
   549→          model: this.getEffectiveModel('evaluator'),
   550→          inputTokens: 0,
   551→          outputTokens: 0,
   552→          cost: 0,
   553→          durationMs: Date.now() - evalStart,
   554→          success: false,
   555→        });
   556→      }
   557→    }
   558→
   559→    // Check budget after evaluator
   560→    if (!this.isWithinBudget()) {
   561→      return {
   562→        plan,
   563→        execution,
   564→        evaluation,
   565→        review: undefined,
   566→        totalCost: this.sumPhaseCosts(phases),
   567→        totalTokens: this.sumPhaseTokens(phases),
   568→        phases,
   569→        escalated,
   570→        escalationReason: escalated ? escalationReason : 'Budget exceeded after evaluator phase',
   571→        success: true,
   572→        timestamp: Date.now(),
   573→      };
   574→    }
   575→
   576→    // --- Phase 4: Reviewer reviews (if quality gates enabled) ---
   577→    let review: ModelResponse | undefined;
   578→    if (this.config.qualityGates.enabled) {
   579→      const reviewStart = Date.now();
   580→      const reviewPrompt = `Review the following code/output for correctness, style, and best practices.\n\nOriginal task: ${prompt}\n\nExecution output:\n${execution.content}\n\nProvide a brief review with: {"approved": <boolean>, "comments": [<string>], "severity": "low"|"medium"|"high"}`;
   581→
   582→      try {
   583→        review = await this.callReviewer(reviewPrompt);
   584→        phases.push({
   585→          role: 'reviewer',
   586→          model: review.model,
   587→          inputTokens: review.usage.inputTokens,
   588→          outputTokens: review.usage.outputTokens,
   589→          cost: review.usage.cost,
   590→          durationMs: Date.now() - reviewStart,
   591→          success: true,
   592→        });
   593→      } catch {
   594→        // Reviewer failure is non-fatal
   595→        phases.push({
   596→          role: 'reviewer',
   597→          model: this.getEffectiveModel('reviewer'),
   598→          inputTokens: 0,
   599→          outputTokens: 0,
   600→          cost: 0,
   601→          durationMs: Date.now() - reviewStart,
   602→          success: false,
   603→        });
   604→      }

   792→  /**
   793→   * Call a model for a specific role, with fallback chain support.
   794→   */
   795→  private async callRole(roleName: string, prompt: string): Promise<ModelResponse> {
   796→    const role = this.getRoleByName(roleName);
   797→    if (!role) {
   798→      throw new Error(`Unknown role: ${roleName}`);
   799→    }
   800→
   801→    const primaryModel = this.effectiveModels[roleName] ?? role.model;
   802→    const messages: Message[] = [
   803→      { role: 'system', content: this.getSystemPromptForRole(roleName) },
   804→      { role: 'user', content: prompt },
   805→    ];
   806→
   807→    const startTime = Date.now();
   808→
   809→    try {
   810→      // Try primary model first
   811→      const response = await this.client.chat({
   812→        model: primaryModel,
   813→        messages,
   814→        maxTokens: role.maxTokensPerRequest,
   815→        stream: false,
   816→        temperature: this.getTemperatureForRole(roleName),
   817→      });
   818→
   819→      const modelResponse: ModelResponse = {
   820→        content: response.content,
   821→        model: primaryModel,
   822→        role: roleName,
   823→        usage: response.usage,
   824→        toolCalls: response.toolCalls ?? [],
   825→        timestamp: Date.now(),
   826→      };
   827→
   828→      this.recordCost(roleName, primaryModel, response.usage, prompt.slice(0, 80));
   829→      return modelResponse;
   830→
   831→    } catch (primaryError) {
   832→      // Primary model failed; try fallback chain
   833→      const errMsg = primaryError instanceof Error ? primaryError.message : String(primaryError);
   834→
   835→      for (const fallbackModel of role.fallbackModels) {
   836→        try {
   837→          const response = await this.client.chat({
   838→            model: fallbackModel,
   839→            messages,
   840→            maxTokens: role.maxTokensPerRequest,
   841→            stream: false,
   842→            temperature: this.getTemperatureForRole(roleName),
   843→          });
   844→
   845→          // Update effective model for this role
   846→          this.effectiveModels[roleName] = fallbackModel;
   847→
   848→          const modelResponse: ModelResponse = {
   849→            content: response.content,
   850→            model: fallbackModel,
   851→            role: roleName,
   852→            usage: response.usage,
   853→            toolCalls: response.toolCalls ?? [],
   854→            timestamp: Date.now(),
   855→          };
   856→
   857→          this.recordCost(roleName, fallbackModel, response.usage, prompt.slice(0, 80));
   858→          return modelResponse;
   859→
   860→        } catch {
   861→          // Continue to next fallback
   862→          continue;
   863→        }
   864→      }
   865→
   866→      // All models failed including fallbacks
   867→      throw new Error(`All models failed for role "${roleName}" (primary: ${primaryModel}). Primary error: ${errMsg}`);
   868→    }
   869→  }

   876→   * Record a cost entry for a role/model combination.

   878→  private recordCost(role: string, model: string, usage: TokenUsage, taskDescription: string): void {
   879→    const cost = usage.cost;
   880→
   881→    this.roleCosts[role] = (this.roleCosts[role] ?? 0) + cost;
   882→    this.modelCosts[model] = (this.modelCosts[model] ?? 0) + cost;

   885→
   886→    this.taskCostEntries.push({
   887→      taskDescription: taskDescription.slice(0, 120),
   888→      cost,
   889→      role,
   890→      model,
   891→      timestamp: Date.now(),
   892→    });
   893→
   894→    // Also record in the spending monitor for integration with the broader system
   895→    this.spendingMonitor.record({
   896→      model,
   897→      inputTokens: usage.inputTokens,
   898→      outputTokens: usage.outputTokens,
   899→      cost,
   900→      sessionId: 'multi-model-session',
   901→    });
   902→
   903→    // Check warn threshold
   904→    if (this.config.costBudget.maxPerSession > 0) {
   905→      const percentUsed = (this.sessionTotalCost / this.config.costBudget.maxPerSession) * 100;
   906→      if (percentUsed >= this.config.costBudget.warnThreshold) {
   907→        console.warn(
   908→          `\x1b[33m\x1b[1m[BUDGET WARNING]\x1b[0m Session spending at ${percentUsed.toFixed(1)}% of budget ($${this.sessionTotalCost.toFixed(4)} / $${this.config.costBudget.maxPerSession.toFixed(2)})`,
   909→        );
   910→      }
   911→    }
   912→  }

   947→  /**
   948→   * Local heuristic classification fallback.
   949→   * Used when the LLM classification call fails or is unavailable.
   950→   */
   951→  private classifyTaskHeuristic(prompt: string): TaskClassification {
   952→    const length = prompt.length;
   953→
   954→    // Check critical patterns first
   955→    for (const pattern of CRITICAL_PATTERNS) {
   956→      if (pattern.test(prompt)) {
   957→        return {
   958→          complexity: 'critical',
   959→          suggestedRole: 'orchestrator',
   960→          suggestedModel: this.getEffectiveModel('orchestrator'),
   961→          estimatedTokens: Math.max(length * 3, 2000),
   962→          reasoning: 'Critical patterns detected (production/security/migration)',
   963→        };
   964→      }
   965→    }
   966→
   967→    // Check complex patterns
   968→    for (const pattern of COMPLEX_PATTERNS) {
   969→      if (pattern.test(prompt)) {
   970→        return {
   971→          complexity: 'complex',
   972→          suggestedRole: 'orchestrator',
   973→          suggestedModel: this.getEffectiveModel('orchestrator'),
   974→          estimatedTokens: Math.max(length * 3, 2000),
   975→          reasoning: 'Complex multi-phase task indicators detected',
   976→        };
   977→      }
   978→    }
   979→
   980→    // Check simple patterns
   981→    for (const pattern of SIMPLE_PATTERNS) {
   982→      if (pattern.test(prompt)) {
   983→        return {
   984→          complexity: 'simple',
   985→          suggestedRole: 'worker',
   986→          suggestedModel: this.getEffectiveModel('worker'),
   987→          estimatedTokens: Math.max(Math.ceil(length / 4), 500),
   988→          reasoning: 'Simple task indicators detected',
   989→        };
   990→      }
   991→    }
   992→
   993→    // Check trivial patterns
   994→    for (const pattern of TRIVIAL_PATTERNS) {
   995→      if (pattern.test(prompt)) {
   996→        return {
   997→          complexity: 'trivial',
   998→          suggestedRole: 'worker',
   999→          suggestedModel: this.getEffectiveModel('worker'),
  1000→          estimatedTokens: Math.max(Math.ceil(length / 4), 100),
  1001→          reasoning: 'Trivial interaction detected',
  1002→        };
  1003→      }
  1004→    }
  1005→
  1006→    // Default to moderate based on length
  1007→    if (length > 500) {
  1008→      return {
  1009→        complexity: 'moderate',
  1010→        suggestedRole: 'orchestrator',
  1011→        suggestedModel: this.getEffectiveModel('orchestrator'),
  1012→        estimatedTokens: Math.ceil(length / 3) + 1000,
  1013→        reasoning: 'Long prompt suggests moderate complexity',
  1014→      };
  1015→    }
  1016→
  1017→    return {
  1018→      complexity: 'moderate',
  1019→      suggestedRole: 'worker',
  1020→      suggestedModel: this.getEffectiveModel('worker'),
  1021→      estimatedTokens: Math.ceil(length / 3) + 500,
  1022→      reasoning: 'Default moderate classification',
  1023→    };
  1024→  }

  1085→  private isWorkerUpgraded(): boolean {
  1086→    if (!this.config.dynamicSwitching) return false;
  1087→    if (this.workerUpgradeUntil <= 0) return false;
  1088→    if (Date.now() > this.workerUpgradeUntil) {
  1089→      this.workerUpgradeUntil = 0;
  1090→      // Reset effective model back to worker's configured model
  1091→      this.effectiveModels.worker = this.config.roles.worker.model;
  1092→      return false;
  1093→    }
  1094→    return true;

  1105→  private cheapestAvailableModel(): string {

  1108→    if (freeToolModels.length > 0) {

  1114→    // Fall back to any free model

  1117→      return freeModels[0].id;

  1120→    // Last resort: use worker model
