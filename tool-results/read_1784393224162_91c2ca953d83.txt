   299→   299→// ---- Interactive Mode ----
   300→   300→async function startInteractive(options: any) {
   301→   301→  const config = initConfig(options.apiKey);
   302→   302→  if (options.model) config.defaultModel = options.model;
   303→   303→  if (options.theme) config.ui.theme = options.theme as 'dark' | 'light' | 'dracula' | 'nord';
   304→   304→  if (options.noStreaming) config.ui.streaming = false;
   305→   305→  if (options.permissionMode) config.permissionMode = options.permissionMode;
   306→   306→  if (options.diffPreview === true) config.diffPreview = true;
   307→   307→  if (options.diffPreview === false) config.diffPreview = false;
   308→   308→  if (options.cache === true) config.promptCache.enabled = true;
   309→   309→  if (options.cache === false) config.promptCache.enabled = false;
   310→   310→  if (options.spendingLimit) config.spendingLimit = options.spendingLimit;
   311→   311→  if (options.autoApprove) {
   312→   312→    config.tools.autoApprove = [...config.tools.autoApprove, ...config.tools.requireApproval];
   313→   313→    config.tools.requireApproval = [];
   314→   314→    config.permissionMode = 'yolo';
   315→   315→  }
   316→   316→
   317→   317→  // Check API key
   318→   318→  if (!config.apiKey && !options.ollama) {
   319→   319→    console.log(chalk.red('\nOpenRouter API key not configured!'));
   320→   320→    console.log(chalk.yellow('\nSet it with:'));
   321→   321→    console.log(`  ${chalk.cyan('neuro config --set-key YOUR_API_KEY')}`);
   322→   322→    console.log(`  ${chalk.cyan('export OPENROUTER_API_KEY=YOUR_API_KEY')}`);
   323→   323→    console.log(`  ${chalk.cyan('neuro -k YOUR_API_KEY')}`);
   324→   324→    console.log(chalk.gray('\nGet your key at: https://openrouter.ai/keys\n'));
   325→   325→    process.exit(1);
   326→   326→  }
   327→   327→
   328→   328→  // Initialize engine
   329→   329→  const engine = new NeuroEngine(config);
   330→   330→
   331→   331→  // Set effort level if specified
   332→   332→  if (options.effort) {
   333→   333→    engine.modelRouter.setEffort(options.effort);
   334→   334→  }
   335→   335→
   336→   336→  // Set output style if specified
   337→   337→  if (options.style) {
   338→   338→    engine.styleManager.setStyle(options.style);
   339→   339→  }
   340→   340→
   341→   341→  // Enable thinking if specified
   342→   342→  if (options.thinking) {
   343→   343→    engine.extendedThinking.setMode('full');
   344→   344→    engine.extendedThinking.toggleDisplay();
   345→   345→  }
   346→   346→
   347→   347→  // Enable sandbox if specified
   348→   348→  if (options.sandbox) {
   349→   349→    engine.sandbox.enable();
   350→   350→  }
   351→   351→
   352→   352→  // Enable prompt cache if specified
   353→   353→  if (options.cache) {
   354→   354→    config.promptCache.enabled = true;
   355→   355→  }
   356→   356→
   357→   357→  // Resume session if requested
   358→   358→  if (options.resume) {
   359→   359→    const session = engine.sessionManager.load(options.resume);
   360→   360→    if (session) {
   361→   361→      engine.ui.success(`Resumed session: ${session.id.slice(0, 20)}...`);
   362→   362→      if (options.fork) {
   363→   363→        const forked = engine.sessionManager.create(process.cwd(), config.defaultModel);
   364→   364→        forked.messages = [...session.messages];
   365→   365→        engine.sessionManager.save();
   366→   366→        engine.ui.success(`Forked to new session: ${forked.id.slice(0, 20)}...`);
   367→   367→      }
   368→   368→    } else {
   369→   369→      engine.ui.error(`Session not found: ${options.resume}`);
   370→   370→    }
   371→   371→  } else if (options.continue) {
   372→   372→    const sessions = engine.sessionManager.list();
   373→   373→    if (sessions.length > 0) {
   374→   374→      const session = engine.sessionManager.load(sessions[0].id);
   375→   375→      if (session) {
   376→   376→        engine.ui.success(`Continued session: ${session.id.slice(0, 20)}...`);
   377→   377→      }
   378→   378→    } else {
   379→   379→      engine.ui.warning('No sessions found to continue');
   380→   380→    }
   381→   381→  }
   382→   382→
   383→   383→  // Initialize auto-updater and check for updates in background
   384→   384→  const updater = new AutoUpdater({ currentVersion: VERSION });
   385→   385→  const updateCheck = updater.checkOnStartup(); // Fire and forget — don't block startup
   386→   386→
   387→   387→  // Print banner
   388→   388→  engine.ui.banner();
   389→   389→  const theme = engine.ui.theme;
   390→   390→  const permMode = engine.approval.getMode();
   391→   391→  const permIcon = engine.approval.getModeIcon();
   392→   392→  const activeStyle = engine.styleManager.getStyle();
   393→   393→  console.log(theme.muted(`  Model: ${MODELS[config.defaultModel]?.name || config.defaultModel}`));
   394→   394→  console.log(theme.muted(`  Permission: ${permIcon} ${permMode}`));
   395→   395→  console.log(theme.muted(`  Style: ${activeStyle.name}`));
   396→   396→  console.log(theme.muted(`  Thinking: ${engine.extendedThinking.getMode()}`));
   397→   397→  console.log(theme.muted(`  Cache: ${config.promptCache.enabled ? 'on' : 'off'}`));
   398→   398→  console.log(theme.muted(`  Working Dir: ${process.cwd()}`));
   399→   399→  console.log(theme.muted(`  Type /help for commands, Tab for completion, Ctrl+C to exit\n`));
   400→   400→
   401→   401→  // Show update notification if available (after banner)
   402→   402→  updateCheck.then((result) => {
   403→   403→    if (result && result.hasUpdate) {
   404→   404→      updater.showUpdateNotification(result);
   405→   405→    }

  1045→  1045→  });
