   341тЖТ   341тЖТ  /**
   342тЖТ   342тЖТ   * Run a task fully autonomously.
   343тЖТ   343тЖТ   *
   344тЖТ   344тЖТ   * The engine will be called repeatedly until:
   345тЖТ   345тЖТ   * - maxIterations is reached
   346тЖТ   346тЖТ   * - maxCost is exceeded
   347тЖТ   347тЖТ   * - maxTimeMs is exceeded
   348тЖТ   348тЖТ   * - The task appears complete (engine signals no further action needed)
   349тЖТ   349тЖТ   * - An error occurs and pauseOnError is true
   350тЖТ   350тЖТ   * - The user aborts
   351тЖТ   351тЖТ   */
   352тЖТ   352тЖТ  async executeAuto(
   353тЖТ   353тЖТ    prompt: string,
   354тЖТ   354тЖТ    engine: AutoModeEngine,
   355тЖТ   355тЖТ    maxIterations?: number,
   356тЖТ   356тЖТ  ): Promise<AutoModeResult> {
   357тЖТ   357тЖТ    if (this.running) {
   358тЖТ   358тЖТ      return { ok: false, iterations: 0, finalText: '', error: 'Auto mode is already running a task' };
   359тЖТ   359тЖТ    }
   360тЖТ   360тЖТ
   361тЖТ   361тЖТ    const limit = maxIterations ?? this.config.maxIterations;
   362тЖТ   362тЖТ    this.running = true;
   363тЖТ   363тЖТ    this.abortController = new AbortController();
   364тЖТ   364тЖТ    this.startTime = Date.now();
   365тЖТ   365тЖТ    this.stats = this.freshStats();
   366тЖТ   366тЖТ
   367тЖТ   367тЖТ    this.emitStatus('running');
   368тЖТ   368тЖТ
   369тЖТ   369тЖТ    const result: AutoModeResult = { ok: true, iterations: 0, finalText: '' };
   370тЖТ   370тЖТ
   371тЖТ   371тЖТ    try {
   372тЖТ   372тЖТ      let currentPrompt = prompt;
   373тЖТ   373тЖТ
   374тЖТ   374тЖТ      for (let i = 0; i < limit; i++) {
   375тЖТ   375тЖТ        // ---- Abort check ----
   376тЖТ   376тЖТ        if (this.abortController.signal.aborted) {
   377тЖТ   377тЖТ          result.ok = false;
   378тЖТ   378тЖТ          result.error = 'Aborted by user';
   379тЖТ   379тЖТ          break;
   380тЖТ   380тЖТ        }
   381тЖТ   381тЖТ
   382тЖТ   382тЖТ        // ---- Time limit ----
   383тЖТ   383тЖТ        if (this.config.maxTimeMs > 0 && Date.now() - this.startTime > this.config.maxTimeMs) {
   384тЖТ   384тЖТ          result.error = `Time limit of ${this.config.maxTimeMs}ms exceeded`;
   385тЖТ   385тЖТ          break;
   386тЖТ   386тЖТ        }
   387тЖТ   387тЖТ
   388тЖТ   388тЖТ        // ---- Cost limit ----
   389тЖТ   389тЖТ        if (this.config.maxCost > 0 && this.stats.totalCost >= this.config.maxCost) {
   390тЖТ   390тЖТ          result.error = `Cost limit of $${this.config.maxCost} exceeded`;
   391тЖТ   391тЖТ          break;
   392тЖТ   392тЖТ        }
   393тЖТ   393тЖТ
   394тЖТ   394тЖТ        // ---- Safety pre-check ----
   395тЖТ   395тЖТ        const safetyResult = this.preFlightSafetyCheck(currentPrompt);
   396тЖТ   396тЖТ        if (!safetyResult.allowed) {
   397тЖТ   397тЖТ          this.stats.blockedOperations++;
   398тЖТ   398тЖТ          result.error = `Safety check blocked execution: ${safetyResult.reason}`;
   399тЖТ   399тЖТ          if (this.config.pauseOnError) break;
   400тЖТ   400тЖТ          // If not pausing on error, try to continue with a modified prompt
   401тЖТ   401тЖТ          currentPrompt = `The previous action was blocked for safety reasons (${safetyResult.reason}). ` +
   402тЖТ   402тЖТ            `Please try a different, safer approach to accomplish the goal. Original task: ${prompt}`;
   403тЖТ   403тЖТ          continue;
   404тЖТ   404тЖТ        }
   405тЖТ   405тЖТ
   406тЖТ   406тЖТ        // ---- Execute ----
   407тЖТ   407тЖТ        try {
   408тЖТ   408тЖТ          const response = await engine.runPrompt(currentPrompt);
   409тЖТ   409тЖТ          this.stats.iterations = i + 1;
   410тЖТ   410тЖТ          this.stats.filesChanged += response.filesChanged;
   411тЖТ   411тЖТ          this.stats.commandsRun += response.commandsRun;
   412тЖТ   412тЖТ          this.stats.totalCost += response.cost;
   413тЖТ   413тЖТ          this.stats.timeElapsedMs = Date.now() - this.startTime;
   414тЖТ   414тЖТ
   415тЖТ   415тЖТ          if (response.error) {
   416тЖТ   416тЖТ            this.stats.errors++;
   417тЖТ   417тЖТ            if (this.config.pauseOnError) {
   418тЖТ   418тЖТ              result.error = response.error;
   419тЖТ   419тЖТ              break;
   420тЖТ   420тЖТ            }
   421тЖТ   421тЖТ          }
   422тЖТ   422тЖТ
   423тЖТ   423тЖТ          result.iterations = i + 1;
   424тЖТ   424тЖТ          result.finalText = response.text;
   425тЖТ   425тЖТ
   426тЖТ   426тЖТ          // ---- Auto-commit ----
   427тЖТ   427тЖТ          if (this.config.autoCommit && response.filesChanged > 0) {
   428тЖТ   428тЖТ            this.autoGitCommit(i + 1);
   429тЖТ   429тЖТ          }
   430тЖТ   430тЖТ
   431тЖТ   431тЖТ          // ---- Auto-test ----
   432тЖТ   432тЖТ          if (this.config.autoTest && response.filesChanged > 0) {
   433тЖТ   433тЖТ            const testResult = this.autoRunTests();
   434тЖТ   434тЖТ            if (!testResult.passed && this.config.pauseOnError) {
   435тЖТ   435тЖТ              result.error = `Tests failed after iteration ${i + 1}`;
   436тЖТ   436тЖТ              break;
   437тЖТ   437тЖТ            }
   438тЖТ   438тЖТ          }
   439тЖТ   439тЖТ
   440тЖТ   440тЖТ          // ---- Checkpoint ----
   441тЖТ   441тЖТ          if (i > 0 && i % 5 === 0) {
   442тЖТ   442тЖТ            this.createCheckpoint(i, undefined, undefined);
   443тЖТ   443тЖТ          }
   444тЖТ   444тЖТ
   445тЖТ   445тЖТ          // ---- Detect completion ----
   446тЖТ   446тЖТ          if (this.detectCompletion(response.text)) {
   447тЖТ   447тЖТ            result.completed = true;
   448тЖТ   448тЖТ            break;
   449тЖТ   449тЖТ          }
   450тЖТ   450тЖТ
   451тЖТ   451тЖТ          // Feed the response back as context for the next iteration
   452тЖТ   452тЖТ          currentPrompt =
   453тЖТ   453тЖТ            `Continuing autonomous task. Previous iteration result:\n` +
   454тЖТ   454тЖТ            `${response.text.slice(0, 2000)}\n\n` +
   455тЖТ   455тЖТ            `If the task is complete, respond with "TASK_COMPLETE". ` +
   456тЖТ   456тЖТ            `Otherwise continue working towards: ${prompt}`;
   457тЖТ   457тЖТ
   458тЖТ   458тЖТ        } catch (err) {
   459тЖТ   459тЖТ          this.stats.errors++;
   460тЖТ   460тЖТ          if (this.config.pauseOnError) {
   461тЖТ   461тЖТ            result.ok = false;
   462тЖТ   462тЖТ            result.error = `Execution error: ${err instanceof Error ? err.message : String(err)}`;
   463тЖТ   463тЖТ            break;
   464тЖТ   464тЖТ          }
   465тЖТ   465тЖТ          // Otherwise swallow and continue
   466тЖТ   466тЖТ        }
   467тЖТ   467тЖТ      }
   468тЖТ   468тЖТ    } finally {
   469тЖТ   469тЖТ      this.running = false;
   470тЖТ   470тЖТ      this.abortController = null;
   471тЖТ   471тЖТ      this.emitStatus(this.config.enabled ? 'enabled' : 'disabled');
   472тЖТ   472тЖТ      this.persistState();
   473тЖТ   473тЖТ    }
   474тЖТ   474тЖТ
   475тЖТ   475тЖТ    return result;
   476тЖТ   476тЖТ  }

   636тЖТ   636тЖТ  /** Replay a saved routine */
   637тЖТ   637тЖТ  async executeRoutine(routineId: string, engine: AutoModeEngine): Promise<AutoModeResult> {
   638тЖТ   638тЖТ    const routine = this.routines.get(routineId);
   639тЖТ   639тЖТ    if (!routine) return { ok: false, iterations: 0, finalText: '', error: `Routine "${routineId}" not found` };

   641тЖТ   641тЖТ

   655тЖТ   655тЖТ        : '';
   656тЖТ   656тЖТ
   657тЖТ   657тЖТ      const fullPrompt = `${contextPrompt}Step ${i + 1}/${routine.steps.length}: ${step.prompt}`;
   658тЖТ   658тЖТ
   659тЖТ   659тЖТ      const stepMaxIter = step.maxIterations ?? Math.max(5, Math.floor(this.config.maxIterations / routine.steps.length));
   660тЖТ   660тЖТ      const result = await this.executeAuto(fullPrompt, engine, stepMaxIter);
   661тЖТ   661тЖТ
   662тЖТ   662тЖТ      overallResult.iterations += result.iterations;
   663тЖТ   663тЖТ

   670тЖТ   670тЖТ      stepResults.push(result.finalText);
   671тЖТ   671тЖТ      overallResult.finalText = result.finalText;
   672тЖТ   672тЖТ

   675тЖТ   675тЖТ        // eslint-disable-next-line no-console
   676тЖТ   676тЖТ        console.log(`[auto-mode] Routine "${routine.name}" paused after step ${i + 1}. Continuing automatically in auto mode.`);
   677тЖТ   677тЖТ      }
   678тЖТ   678тЖТ    }
   679тЖТ   679тЖТ
   680тЖТ   680тЖТ    if (overallResult.ok) {
   681тЖТ   681тЖТ      overallResult.completed = true;
   682тЖТ   682тЖТ    }
   683тЖТ   683тЖТ
   684тЖТ   684тЖТ    return overallResult;
   685тЖТ   685тЖТ  }

   691тЖТ   691тЖТ  /**
   692тЖТ   692тЖТ   * Pre-flight safety check тАУ examines the prompt for dangerous commands
   693тЖТ   693тЖТ   * or patterns that should be blocked even in auto mode.
   694тЖТ   694тЖТ   */
   695тЖТ   695тЖТ  preFlightSafetyCheck(prompt: string): { allowed: boolean; reason?: string } {
   696тЖТ   696тЖТ    const lower = prompt.toLowerCase();
   697тЖТ   697тЖТ
   698тЖТ   698тЖТ    // Check always-blocked commands
   699тЖТ   699тЖТ    for (const cmd of this.config.blockedCommands) {
   700тЖТ   700тЖТ      if (lower.includes(cmd.toLowerCase())) {
   701тЖТ   701тЖТ        return { allowed: false, reason: `Blocked command detected: "${cmd}"` };
   702тЖТ   702тЖТ      }
   703тЖТ   703тЖТ    }
   704тЖТ   704тЖТ
   705тЖТ   705тЖТ    // Safety-level-specific checks for conservative / moderate
   706тЖТ   706тЖТ    if (this.config.safetyLevel === 'conservative') {
   707тЖТ   707тЖТ      for (const cmd of CONSERVATIVE_BLOCKED_COMMANDS) {
   708тЖТ   708тЖТ        if (lower.includes(cmd.toLowerCase())) {
   709тЖТ   709тЖТ          return { allowed: false, reason: `Conservative mode blocked command: "${cmd}"` };
   710тЖТ   710тЖТ        }
   711тЖТ   711тЖТ      }
   712тЖТ   712тЖТ    } else if (this.config.safetyLevel === 'moderate') {
   713тЖТ   713тЖТ      for (const cmd of MODERATE_BLOCKED_COMMANDS) {
   714тЖТ   714тЖТ        if (lower.includes(cmd.toLowerCase())) {
   715тЖТ   715тЖТ          return { allowed: false, reason: `Moderate mode blocked command: "${cmd}"` };
   716тЖТ   716тЖТ        }
   717тЖТ   717тЖТ      }
   718тЖТ   718тЖТ    }
   719тЖТ   719тЖТ    // 'aggressive' тАУ only always-blocked commands are blocked
   720тЖТ   720тЖТ
   721тЖТ   721тЖТ    // Check blocked file patterns
   722тЖТ   722тЖТ    for (const pattern of this.config.blockedPatterns) {
   723тЖТ   723тЖТ      // Simple substring check for path-like strings in prompt
   724тЖТ   724тЖТ      if (lower.includes(pattern.replace(/\*\*/g, '').replace(/\*/g, '').toLowerCase())) {
   725тЖТ   725тЖТ        return { allowed: false, reason: `Blocked file pattern detected: "${pattern}"` };
   726тЖТ   726тЖТ      }
   727тЖТ   727тЖТ    }
   728тЖТ   728тЖТ
   729тЖТ   729тЖТ    return { allowed: true };
   730тЖТ   730тЖТ  }
   731тЖТ   731тЖТ
   732тЖТ   732тЖТ  /**
   733тЖТ   733тЖТ   * Runtime safety check for a specific command about to be executed.
   734тЖТ   734тЖТ   * This is meant to be called by the engine before running any shell command.
   735тЖТ   735тЖТ   */
   736тЖТ   736тЖТ  isCommandAllowed(command: string): { allowed: boolean; reason?: string } {
   737тЖТ   737тЖТ    const trimmed = command.trim();
   738тЖТ   738тЖТ
   739тЖТ   739тЖТ    // Always-blocked
   740тЖТ   740тЖТ    for (const blocked of ALWAYS_BLOCKED_COMMANDS) {
   741тЖТ   741тЖТ      if (trimmed.includes(blocked) || trimmed.startsWith(blocked.split(' ')[0])) {
   742тЖТ   742тЖТ        return { allowed: false, reason: `Command blocked (always): "${blocked}"` };
   743тЖТ   743тЖТ      }
   744тЖТ   744тЖТ    }
   745тЖТ   745тЖТ
   746тЖТ   746тЖТ    // Config-level blocked
   747тЖТ   747тЖТ    for (const blocked of this.config.blockedCommands) {
   748тЖТ   748тЖТ      if (trimmed.includes(blocked)) {
   749тЖТ   749тЖТ        return { allowed: false, reason: `Command blocked (config): "${blocked}"` };
   750тЖТ   750тЖТ      }
   751тЖТ   751тЖТ    }
   752тЖТ   752тЖТ
   753тЖТ   753тЖТ    // Safety-level checks
   754тЖТ   754тЖТ    const levelLists: Record<SafetyLevel, string[]> = {
   755тЖТ   755тЖТ      conservative: CONSERVATIVE_BLOCKED_COMMANDS,
   756тЖТ   756тЖТ      moderate: MODERATE_BLOCKED_COMMANDS,
   757тЖТ   757тЖТ      aggressive: [],
   758тЖТ   758тЖТ    };
   759тЖТ   759тЖТ    for (const blocked of levelLists[this.config.safetyLevel]) {
   760тЖТ   760тЖТ      if (trimmed.includes(blocked)) {
   761тЖТ   761тЖТ        return { allowed: false, reason: `Command blocked (${this.config.safetyLevel}): "${blocked}"` };
   762тЖТ   762тЖТ      }
   763тЖТ   763тЖТ    }
   764тЖТ   764тЖТ
   765тЖТ   765тЖТ    return { allowed: true };
   766тЖТ   766тЖТ  }
   767тЖТ   767тЖТ
   768тЖТ   768тЖТ  /**
   769тЖТ   769тЖТ   * Runtime safety check for a file path about to be modified.
   770тЖТ   770тЖТ   * This is meant to be called by the engine before any file write.
   771тЖТ   771тЖТ   */
   772тЖТ   772тЖТ  isFileModificationAllowed(filePath: string): { allowed: boolean; reason?: string } {
   773тЖТ   773тЖТ    const normalised = filePath.replace(/\\/g, '/');
   774тЖТ   774тЖТ
   775тЖТ   775тЖТ    for (const pattern of this.config.blockedPatterns) {
   776тЖТ   776тЖТ      if (this.matchesGlobPattern(normalised, pattern)) {
   777тЖТ   777тЖТ        return { allowed: false, reason: `File pattern blocked: "${pattern}"` };
   778тЖТ   778тЖТ      }
   779тЖТ   779тЖТ    }
   780тЖТ   780тЖТ
   781тЖТ   781тЖТ    // Safety-level-specific patterns
   782тЖТ   782тЖТ    const patternLists: Record<SafetyLevel, string[]> = {
   783тЖТ   783тЖТ      conservative: CONSERVATIVE_BLOCKED_PATTERNS,
   784тЖТ   784тЖТ      moderate: [],
   785тЖТ   785тЖТ      aggressive: [],
   786тЖТ   786тЖТ    };
   787тЖТ   787тЖТ    for (const pattern of patternLists[this.config.safetyLevel]) {
   788тЖТ   788тЖТ      if (this.matchesGlobPattern(normalised, pattern)) {
   789тЖТ   789тЖТ        return { allowed: false, reason: `File pattern blocked (${this.config.safetyLevel}): "${pattern}"` };
   790тЖТ   790тЖТ      }
   791тЖТ   791тЖТ    }
   792тЖТ   792тЖТ
   793тЖТ   793тЖТ    return { allowed: true };
   794тЖТ   794тЖТ  }

   827тЖТ   827тЖТ    const cp: AutoModeCheckpoint = {

   839тЖТ   839тЖТ

   841тЖТ   841тЖТ    try {

   844тЖТ   844тЖТ      writeFileSync(cpFile, JSON.stringify(cp, null, 2), 'utf-8');
   845тЖТ   845тЖТ    } catch {
   846тЖТ   846тЖТ      // Best-effort persistence
   847тЖТ   847тЖТ    }
   848тЖТ   848тЖТ
   849тЖТ   849тЖТ    // Keep at most 50 checkpoints in memory
   850тЖТ   850тЖТ    if (this.checkpoints.length > 50) {
   851тЖТ   851тЖТ      const removed = this.checkpoints.shift();
   852тЖТ   852тЖТ      if (removed) {
   853тЖТ   853тЖТ        try {
   854тЖТ   854тЖТ          const cpFile = join(CHECKPOINTS_DIR, `checkpoint-${removed.id}.json`);
   855тЖТ   855тЖТ          if (existsSync(cpFile)) unlinkSync(cpFile);
   856тЖТ   856тЖТ        } catch { /* ignore */ }
   857тЖТ   857тЖТ      }
   858тЖТ   858тЖТ    }
   859тЖТ   859тЖТ
   860тЖТ   860тЖТ    return cp;
   861тЖТ   861тЖТ  }

   963тЖТ   963тЖТ      }

   968тЖТ   968тЖТ  private matchesGlobPattern(filePath: string, pattern: string): boolean {
   969тЖТ   969тЖТ    // Convert simple glob to regex
   970тЖТ   970тЖТ    const regexStr = pattern
   971тЖТ   971тЖТ      .replace(/[.+^${}()|[\]\\]/g, '\\$&')   // escape regex special chars (except * and ?)
   972тЖТ   972тЖТ      .replace(/\*\*/g, '{{GLOBSTAR}}')          // preserve **
   973тЖТ   973тЖТ      .replace(/\*/g, '[^/]*')                   // * matches anything except /
   974тЖТ   974тЖТ      .replace(/\?/g, '[^/]')                    // ? matches single non-/
   975тЖТ   975тЖТ      .replace(/\{\{GLOBSTAR\}\}/g, '.*');       // ** matches anything
   976тЖТ   976тЖТ    try {
   977тЖТ   977тЖТ      const regex = new RegExp(regexStr + '$', 'i');
   978тЖТ   978тЖТ      return regex.test(filePath);
   979тЖТ   979тЖТ    } catch {
   980тЖТ   980тЖТ      return false;
   981тЖТ   981тЖТ    }
   982тЖТ   982тЖТ  }

   998тЖТ   998тЖТ  private loadState(): void {

  1004тЖТ  1004тЖТ      }
  1005тЖТ  1005тЖТ    } catch { /* ignore corrupt state */ }
  1006тЖТ  1006тЖТ
  1007тЖТ  1007тЖТ    // Load goals
  1008тЖТ  1008тЖТ    try {
  1009тЖТ  1009тЖТ      if (existsSync(GOALS_FILE)) {
  1010тЖТ  1010тЖТ        const data = JSON.parse(readFileSync(GOALS_FILE, 'utf-8')) as GoalDefinition[];

  1014тЖТ  1014тЖТ
  1015тЖТ  1015тЖТ    // Load routines
  1016тЖТ  1016тЖТ    try {
  1017тЖТ  1017тЖТ      if (existsSync(ROUTINES_FILE)) {
  1018тЖТ  1018тЖТ        const data = JSON.parse(readFileSync(ROUTINES_FILE, 'utf-8')) as RoutineDefinition[];
  1019тЖТ  1019тЖТ        for (const r of data) this.routines.set(r.id, r);
  1020тЖТ  1020тЖТ      }
  1021тЖТ  1021тЖТ    } catch { /* ignore */ }
  1022тЖТ  1022тЖТ
  1023тЖТ  1023тЖТ    // Load checkpoints from disk
  1024тЖТ  1024тЖТ    try {
  1025тЖТ  1025тЖТ      if (existsSync(CHECKPOINTS_DIR)) {
  1026тЖТ  1026тЖТ        const files = readdirSync(CHECKPOINTS_DIR).filter(f => f.startsWith('checkpoint-'));
  1027тЖТ  1027тЖТ        for (const f of files.slice(-50)) {
  1028тЖТ  1028тЖТ          try {
  1029тЖТ  1029тЖТ            const cp = JSON.parse(readFileSync(join(CHECKPOINTS_DIR, f), 'utf-8')) as AutoModeCheckpoint;
  1030тЖТ  1030тЖТ            this.checkpoints.push(cp);
  1031тЖТ  1031тЖТ          } catch { /* skip corrupt */ }
  1032тЖТ  1032тЖТ        }
  1033тЖТ  1033тЖТ        // Sort by timestamp
  1034тЖТ  1034тЖТ        this.checkpoints.sort((a, b) => new Date(a.timestamp).getTime() - new Date(b.timestamp).getTime());
  1035тЖТ  1035тЖТ      }
  1036тЖТ  1036тЖТ    } catch { /* ignore */ }
  1037тЖТ  1037тЖТ  }
