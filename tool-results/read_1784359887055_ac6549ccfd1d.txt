    57→      agent: opts.agent,
    58→      maxTurns: opts.maxTurns,
    59→      allowedTools: opts.allowedTools?.split(','),
    60→      outputFormat: opts.format || 'text',

    62→      continueSession: opts.continue,

   182→        client.addServer(name, {

   215→// ---- Interactive Mode ----
   216→async function startInteractive(options: any) {
   217→  const config = initConfig(options.apiKey);
   218→  if (options.model) config.defaultModel = options.model;
   219→  if (options.theme) config.ui.theme = options.theme as 'dark' | 'light' | 'dracula' | 'nord';
   220→  if (options.noStreaming) config.ui.streaming = false;
   221→  if (options.permissionMode) config.permissionMode = options.permissionMode;
   222→  if (options.diffPreview === true) config.diffPreview = true;
   223→  if (options.diffPreview === false) config.diffPreview = false;
   224→  if (options.autoApprove) {
   225→    config.tools.autoApprove = [...config.tools.autoApprove, ...config.tools.requireApproval];
   226→    config.tools.requireApproval = [];
   227→    config.permissionMode = 'yolo';
   228→  }
   229→
   230→  // Check API key
   231→  if (!config.apiKey) {
   232→    console.log(chalk.red('\nOpenRouter API key not configured!'));
   233→    console.log(chalk.yellow('\nSet it with:'));
   234→    console.log(`  ${chalk.cyan('neuro config --set-key YOUR_API_KEY')}`);
   235→    console.log(`  ${chalk.cyan('export OPENROUTER_API_KEY=YOUR_API_KEY')}`);
   236→    console.log(`  ${chalk.cyan('neuro -k YOUR_API_KEY')}`);
   237→    console.log(chalk.gray('\nGet your key at: https://openrouter.ai/keys\n'));
   238→    process.exit(1);
   239→  }
   240→
   241→  // Initialize engine
   242→  const engine = new NeuroEngine(config);
   243→
   244→  // Resume session if requested
   245→  if (options.resume) {
   246→    const session = engine.sessionManager.load(options.resume);
   247→    if (session) {
   248→      engine.ui.success(`Resumed session: ${session.id.slice(0, 20)}...`);
   249→    } else {
   250→      engine.ui.error(`Session not found: ${options.resume}`);
   251→    }
   252→  } else if (options.continue) {
   253→    const sessions = engine.sessionManager.list();
   254→    if (sessions.length > 0) {
   255→      const session = engine.sessionManager.load(sessions[0].id);
   256→      if (session) {
   257→        engine.ui.success(`Continued session: ${session.id.slice(0, 20)}...`);
   258→      }
   259→    } else {
   260→      engine.ui.warning('No sessions found to continue');
   261→    }
   262→  }
   263→
   264→  // Print banner
   265→  engine.ui.banner();
   266→  const theme = engine.ui.theme;
   267→  const permMode = engine.approval.getMode();
   268→  const permIcon = engine.approval.getModeIcon();
   269→  console.log(theme.muted(`  Model: ${MODELS[config.defaultModel]?.name || config.defaultModel}`));
   270→  console.log(theme.muted(`  Permission: ${permIcon} ${permMode}`));
   271→  console.log(theme.muted(`  Working Dir: ${process.cwd()}`));
   272→  console.log(theme.muted(`  Type /help for commands, Tab for completion, Ctrl+C to exit\n`));
   273→
   274→  // Create readline with tab completion
   275→  const completionEngine = new CompletionEngine(process.cwd());
   276→  completionEngine.setAgentNames(Array.from(engine.agents.keys()));
   277→
   278→  const rl = createInterface({
   279→    input: process.stdin,
   280→    output: process.stdout,
   281→    prompt: chalk.cyan('❯ '),
   282→    historySize: 100,
   283→    completer: completionEngine.complete,
   284→  });
   285→
   286→  rl.prompt();
   287→
   288→  let currentMode: 'auto' | 'agent' | 'direct' = 'auto';
   289→  let currentAgent: string | undefined;
   290→
   291→  rl.on('line', async (line) => {
   292→    const input = line.trim();
   293→    if (!input) { rl.prompt(); return; }
   294→
   295→    // Handle slash commands
   296→    if (input.startsWith('/')) {
   297→      const [cmd, ...args] = input.slice(1).split(' ');
   298→
   299→      switch (cmd) {
   300→        case 'help':
   301→          printHelp(engine);
   302→          break;
   303→
   304→        case 'model':
   305→          if (args[0]) {
   306→            engine.switchModel(args[0]);
   307→            // Update fallback chain to put new model first
   308→            engine.fallback.addFallbackModel(config.defaultModel);
   309→          } else {
   310→            engine.ui.modelList(config.defaultModel);
   311→          }
   312→          break;
   313→
   314→        case 'agent':
   315→          if (args[0]) {
   316→            const agentNames = Array.from(engine.agents.keys());
   317→            const found = agentNames.find(n => n.toLowerCase() === args[0].toLowerCase());
   318→            if (found) { currentAgent = found; currentMode = 'direct'; engine.ui.success(`Switched to agent: ${found}`); }
   319→            else { engine.ui.error(`Agent not found. Available: ${agentNames.join(', ')}`); }
   320→          } else {
   321→            const agents = Array.from(engine.agents.entries()).map(([name]) => ({ name, description: '', model: config.defaultModel }));
   322→            engine.ui.agentList(agents);
   323→          }
   324→          break;
   325→
   326→        case 'auto':
   327→          currentMode = 'auto'; currentAgent = undefined;
   328→          engine.ui.success('Mode: Auto (smart orchestration)');
   329→          break;
   330→
   331→        case 'orchestrate':
   332→        case 'plan':
   333→          currentMode = 'agent'; currentAgent = undefined;
   334→          engine.ui.success('Mode: Multi-agent orchestration');
   335→          break;
   336→
   337→        case 'direct':
   338→          currentMode = 'direct';
   339→          engine.ui.success('Mode: Direct (single agent)');
   340→          break;
   341→
   342→        case 'plan-mode':
   343→          engine.approval.setMode('plan');
   344→          engine.ui.success(`${engine.approval.getModeIcon()} Permission: plan (read-only)`);
   345→          break;
   346→
   347→        case 'stats':
   348→          const stats = engine.getSessionStats();
   349→          engine.ui.sessionStats(stats.inputTokens, stats.outputTokens, stats.cost);
   350→          break;
   351→
   352→        case 'clear':
   353→          console.clear(); engine.ui.banner();
   354→          break;
   355→
   356→        case 'theme':
   357→          if (args[0]) {
   358→            config.ui.theme = args[0] as 'dark' | 'light' | 'dracula' | 'nord';
   359→            engine.ui = new TerminalUI(config.ui.theme, config.ui.showTokenCount, config.ui.showCost);
   360→            engine.ui.success(`Theme: ${args[0]}`);
   361→          } else { console.log('Available themes: dracula, dark, nord, light'); }
   362→          break;
   363→
   364→        case 'resume':
   365→          if (args[0]) {
   366→            const sessionId = args[0] === 'latest' ? engine.sessionManager.list()[0]?.id : args[0];
   367→            if (sessionId) {
   368→              const session = engine.sessionManager.load(sessionId);
   369→              if (session) engine.ui.success(`Resumed session: ${sessionId.slice(0, 20)}...`);
   370→              else engine.ui.error(`Session not found: ${sessionId}`);
   371→            } else { engine.ui.error('No sessions found'); }
   372→          } else {
   373→            const sessions = engine.sessionManager.list();
   374→            if (sessions.length === 0) { engine.ui.info('No sessions found'); break; }
   375→            console.log(chalk.bold('\nRecent sessions:\n'));
   376→            for (const s of sessions.slice(0, 10)) {
   377→              const date = new Date(s.createdAt).toLocaleString();
   378→              console.log(`  ${chalk.cyan(s.id.slice(0, 20))}  ${chalk.gray(date)}  ${chalk.gray(`${s.messageCount} msgs`)}`);
   379→            }
   380→            console.log(chalk.gray('\nUse /resume <id> to resume a session'));
   381→          }
   382→          break;
   383→
   384→        case 'compact':
   385→          engine.ui.info('Compacting conversation context...');
   386→          // Trigger context compaction
   387→          engine.contextManager.manage(engine.sessionManager.getCurrent()?.messages || []);
   388→          engine.ui.success('Context compacted');
   389→          break;
   390→
   391→        case 'undo':
   392→          engine.ui.info('Undoing last change...');
   393→          try {
   394→            const { GitCheckpointSystem } = await import('./context/git-checkpoint.js');
   395→            const cps = new GitCheckpointSystem(process.cwd());
   396→            await cps.undo();
   397→            engine.ui.success('Last change undone');
   398→          } catch (e) {
   399→            engine.ui.error('Could not undo: ' + (e instanceof Error ? e.message : String(e)));
   400→          }
   401→          break;
   402→
   403→        case 'redo':
   404→          engine.ui.info('Redo not yet implemented (requires undo stack)');
   405→          break;
   406→
   407→        case 'fork':
   408→          engine.ui.info('Forking current session...');
   409→          const currentSession = engine.sessionManager.getCurrent();
   410→          if (currentSession) {
   411→            const forked = engine.sessionManager.create(process.cwd(), config.defaultModel);
   412→            forked.messages = [...currentSession.messages];
   413→            engine.sessionManager.save();
   414→            engine.ui.success(`Forked to new session: ${forked.id.slice(0, 20)}...`);
   415→          } else {
   416→            engine.ui.error('No active session to fork');
   417→          }
   418→          break;
   419→
   420→        case 'mcp':
   421→          const mcpSub = args[0];
   422→          if (mcpSub === 'list') {
   423→            const servers = engine.mcpClient.listServers();
   424→            if (servers.length === 0) { engine.ui.info('No MCP servers configured'); break; }
   425→            console.log(chalk.bold('\nMCP Servers:\n'));
   426→            for (const s of servers) {
   427→              const status = s.connected ? chalk.green('● connected') : chalk.gray('○ disconnected');
   428→              console.log(`  ${chalk.cyan(s.name.padEnd(20))} ${status}  ${chalk.gray(`${s.toolCount} tools`)}  ${chalk.gray(s.config.transport)}`);
   429→            }
   430→          } else if (mcpSub === 'connect' && args[1]) {
   431→            try {
   432→              const cfg = engine.mcpClient.loadConfig();
   433→              if (cfg.mcpServers[args[1]]) {
   434→                await engine.mcpClient.connect(args[1], cfg.mcpServers[args[1]]);
   435→                engine.ui.success(`Connected to MCP server: ${args[1]}`);
   436→              } else { engine.ui.error(`MCP server "${args[1]}" not found`); }
   437→            } catch (e) { engine.ui.error(`Failed to connect: ${e instanceof Error ? e.message : String(e)}`); }
   438→          } else if (mcpSub === 'disconnect' && args[1]) {
   439→            await engine.mcpClient.disconnect(args[1]);
   440→            engine.ui.success(`Disconnected from ${args[1]}`);
   441→          } else if (mcpSub === 'health') {
   442→            engine.mcpClient.healthReport();
   443→          } else {
   444→            console.log(chalk.bold('\nMCP Commands:\n'));
   445→            console.log('  /mcp list              List MCP servers');
   446→            console.log('  /mcp connect <name>    Connect to a server');
   447→            console.log('  /mcp disconnect <name> Disconnect from a server');
   448→            console.log('  /mcp health            Show MCP health report');
   449→            console.log('  neuro mcp add <name> <cmd>  Add server (CLI)');
   450→            console.log('  neuro mcp list               List servers (CLI)');
   451→          }
   452→          break;
   453→
   454→        case 'permission':
   455→        case 'perm':
   456→          if (args[0]) {
   457→            const validModes = ['manual', 'auto', 'plan', 'yolo'];
   458→            if (validModes.includes(args[0])) {
   459→              engine.approval.setMode(args[0] as any);
   460→              engine.ui.success(`${engine.approval.getModeIcon()} Permission: ${args[0]} (${engine.approval.getModeDescription()})`);
   461→            } else { engine.ui.error(`Invalid mode. Use: ${validModes.join(', ')}`); }
   462→          } else {
   463→            const newMode = engine.approval.cycleMode();
   464→            engine.ui.success(`${engine.approval.getModeIcon()} Permission: ${newMode} (${engine.approval.getModeDescription()})`);
   465→          }
   466→          break;
   467→
   468→        case 'init':
   469→          engine.ui.info('Initializing NEURO.md for this project...');
   470→          try {
   471→            const { NeuroMdSystem } = await import('./context/neuro-md.js');
   472→            const nmd = new NeuroMdSystem();
   473→            await nmd.autoGenerate(process.cwd());
   474→            engine.ui.success('NEURO.md created');
   475→          } catch (e) { engine.ui.error('Could not create NEURO.md: ' + (e instanceof Error ? e.message : String(e))); }
   476→          break;
   477→
   478→        case 'unpause':
   479→          engine.doomLoop.unpause();
   480→          break;
   481→
   482→        case 'sandbox':
   483→          if (args[0] === 'on' || args[0] === 'enable') {
   484→            engine.sandbox.enable();
   485→            engine.ui.success('🔒 Sandbox mode enabled');
   486→          } else if (args[0] === 'off' || args[0] === 'disable') {
   487→            engine.sandbox.disable();
   488→            engine.ui.success('🔓 Sandbox mode disabled');
   489→          } else if (args[0] === 'status') {
   490→            engine.sandbox.printStatus();
   491→          } else if (args[0] === 'undo') {
   492→            const undone = engine.sandbox.undoAll();
   493→            engine.ui.success(`Undone ${undone} file modifications`);
   494→          } else {
   495→            const enabled = engine.sandbox.toggle();
   496→            engine.ui.success(enabled ? '🔒 Sandbox mode enabled' : '🔓 Sandbox mode disabled');
   497→          }
   498→          break;
   499→
   500→        case 'plugins':
   501→        case 'plugin':
   502→          const pluginSub = args[0];
   503→          if (pluginSub === 'list') {
   504→            const plugins = engine.pluginManager.listPlugins();
   505→            if (plugins.length === 0) { engine.ui.info('No plugins loaded'); break; }
   506→            console.log(chalk.bold('\nPlugins:\n'));
   507→            for (const p of plugins) {
   508→              console.log(`  ${chalk.cyan(p.name)} v${p.version} - ${chalk.gray(p.description)} ${chalk.green(`${p.toolCount} tools`)}`);
   509→            }
   510→          } else if (pluginSub === 'load' && args[1]) {
   511→            try {
   512→              await engine.pluginManager.loadByName(args[1]);
   513→              engine.ui.success(`Plugin "${args[1]}" loaded`);
   514→            } catch (e) { engine.ui.error(`Failed to load plugin: ${e instanceof Error ? e.message : String(e)}`); }
   515→          } else {
   516→            console.log(chalk.bold('\nPlugin Commands:\n'));
   517→            console.log('  /plugins list           List loaded plugins');
   518→            console.log('  /plugins load <name>    Load a plugin');
   519→          }
   520→          break;
   521→
   522→        case 'whitelist':
   523→          if (args[0] === 'add' && args[1]) {
   524→            engine.approval.addToWhitelist(args[1]);
   525→            engine.ui.success(`Added "${args[1]}" to whitelist`);
   526→          } else if (args[0] === 'remove' && args[1]) {
   527→            engine.approval.removeFromWhitelist(args[1]);
   528→            engine.ui.success(`Removed "${args[1]}" from whitelist`);
   529→          } else if (args[0] === 'list') {
   530→            const wl = engine.approval.getWhitelist();
   531→            console.log(chalk.bold('\nWhitelisted tools:\n'));
   532→            for (const t of wl) console.log(`  ${chalk.green('✓')} ${t}`);
   533→          } else {
   534→            console.log('Usage: /whitelist add|remove|list <tool>');
   535→          }
   536→          break;
   537→
   538→        case 'blacklist':
   539→          if (args[0] === 'add' && args[1]) {
   540→            engine.approval.addToBlacklist(args[1]);
   541→            engine.ui.success(`Added "${args[1]}" to blacklist`);
   542→          } else if (args[0] === 'remove' && args[1]) {
   543→            engine.approval.removeFromBlacklist(args[1]);
   544→            engine.ui.success(`Removed "${args[1]}" from blacklist`);
   545→          } else if (args[0] === 'list') {
   546→            const bl = engine.approval.getBlacklist();
   547→            console.log(chalk.bold('\nBlacklisted tools:\n'));
   548→            for (const t of bl) console.log(`  ${chalk.red('✗')} ${t}`);
   549→          } else {
   550→            console.log('Usage: /blacklist add|remove|list <tool>');
   551→          }
   552→          break;
   553→
   554→        case 'doctor':
   555→          console.log(chalk.bold('\nNeuroCLI Health Check:\n'));
   556→          console.log(`  API Key: ${config.apiKey ? chalk.green('configured') : chalk.red('MISSING')}`);
   557→          console.log(`  Default Model: ${chalk.cyan(config.defaultModel)} ${MODELS[config.defaultModel] ? chalk.green('valid') : chalk.red('INVALID')}`);
   558→          console.log(`  MCP Servers: ${chalk.cyan(String(engine.mcpClient.listServers().length))}`);
   559→          console.log(`  Permission Mode: ${chalk.cyan(engine.approval.getMode())}`);
   560→          console.log(`  Diff Preview: ${config.diffPreview ? chalk.green('enabled') : chalk.gray('disabled')}`);
   561→          console.log(`  Fallback Chain: ${config.fallbackChain.models.length > 0 ? chalk.green(config.fallbackChain.models.length + ' models') : chalk.yellow('none')}`);
   562→          console.log(`  Doom Loop Protection: ${config.doomLoop.autoBreak ? chalk.green('enabled') : chalk.yellow('disabled')}`);
   563→          console.log(`  Sandbox: ${engine.sandbox.isEnabled() ? chalk.green('enabled') : chalk.gray('disabled')}`);
   564→          console.log(`  Plugins: ${chalk.cyan(String(engine.pluginManager.listPlugins().length))}`);
   565→          console.log(`  Sessions: ${chalk.cyan(String(engine.sessionManager.list().length))}`);
   566→          console.log(`  Spending Limit: ${config.spendingLimit > 0 ? chalk.cyan('$' + config.spendingLimit.toFixed(2)) : chalk.gray('unlimited')}`);
   567→          console.log();
   568→          break;
   569→
   570→        case 'export':
   571→          const exportSession = engine.sessionManager.getCurrent();
   572→          if (exportSession) {
   573→            const json = JSON.stringify(exportSession, null, 2);
   574→            console.log(json);
   575→          } else { engine.ui.warning('No active session to export'); }
   576→          break;
   577→
   578→        case 'exit':
   579→        case 'quit':
   580→        case 'q':
   581→          engine.mcpClient.disconnectAll().catch(() => {});
   582→          engine.approval.close();
   583→          engine.ui.info('Goodbye!');
   584→          process.exit(0);
   585→          break;
   586→
   587→        default:
   588→          engine.ui.error(`Unknown command: /${cmd}. Type /help for available commands.`);
   589→      }
   590→
   591→      rl.prompt();
   592→      return;
   593→    }
   594→
   595→    // Process message with the engine
   596→    try {
   597→      await engine.processMessage(input, currentMode, currentAgent);
   598→    } catch (error) {
   599→      engine.ui.error(error instanceof Error ? error.message : String(error));
   600→    }
   601→
   602→    rl.prompt();
   603→  });
   604→
   605→  rl.on('close', () => {
   606→    engine.mcpClient.disconnectAll().catch(() => {});
   607→    engine.approval.close();
   608→    engine.ui.info('Goodbye!');
   609→    process.exit(0);
   610→  });
   611→}
