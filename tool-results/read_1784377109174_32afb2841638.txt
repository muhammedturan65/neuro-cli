   308→  /**
   309→   * Register plugin tools with the tool registry
   310→   */
   311→  private registerPluginTools(): void {
   312→    const pluginTools = this.pluginManager.getToolDefinitions();
   313→    for (const toolDef of pluginTools) {
   314→      this.registry.register({
   315→        name: toolDef.name,
   316→        risk: toolDef.risk,
   317→        execute: async (args, context) => {
   318→          const result = await this.pluginManager.executeTool(toolDef.name, args, {
   319→            workingDirectory: context.workingDirectory,
   320→            sessionId: context.sessionId,
   321→            agentName: context.agentName,
   322→            onProgress: context.onProgress || (() => {}),
   323→            callTool: async (name, callArgs) => {
   324→              return this.registry.execute(name, callArgs, context);
   325→            },
   326→            memory: {
   327→              get: () => undefined,
   328→              set: () => {},
   329→              delete: () => {},
   330→              list: () => [],
   331→            },
   332→          });
   333→          return result.content;
   334→        },
   335→      });
   336→    }
   337→  }

   339→  /**
   340→   * Register custom tools from .neuro/tools/
   341→   */
   342→  private registerCustomTools(): void {
   343→    const customTools = this.customToolLoader.getAll();
   344→    for (const toolDef of customTools) {
   345→      const executor = this.customToolLoader.createExecutor(toolDef);
   346→      this.registry.register({
   347→        name: `custom_${toolDef.name}`,
   348→        risk: toolDef.risk || 'medium',
   349→        execute: async (args) => {
   350→          try {
   351→            const result = await executor(args);
   352→            return typeof result === 'string' ? result : JSON.stringify(result);
   353→          } catch (error) {
   354→            return `Custom tool error: ${error instanceof Error ? error.message : String(error)}`;
   355→          }
   356→        },
   357→      });
   358→    }
   359→  }

   364→  private loadCustomAgents(): void {

   377→    }
   378→  }

   414→  /**
   415→   * Process a user message
   416→   */
   417→  async processMessage(
   418→    message: string,
   419→    mode: 'auto' | 'agent' | 'direct' = 'auto',
   420→    targetAgent?: string,
   421→  ): Promise<{ content: string; usage: TokenUsage; execution?: AgentExecution }> {
   422→    // Check spending limit via spending monitor
   423→    const limitCheck = this.spendingMonitor.checkLimit();
   424→    if (!limitCheck.allowed) {
   425→      this.ui.error(`Spending limit reached (${limitCheck.limitReached}). Use /spending to check or /config to adjust.`);
   426→      return { content: 'Spending limit reached.', usage: { inputTokens: 0, outputTokens: 0, cost: 0 } };
   427→    }
   428→
   429→    // Auto-activate skills based on prompt
   430→    const activatedSkills = this.skillSystem.autoActivate(message);
   431→    if (activatedSkills.length > 0) {
   432→      for (const skill of activatedSkills) {
   433→        this.ui.info(`Skill activated: ${skill.skill.name} (${skill.activatedBy})`);
   434→      }
   435→    }
   436→
   437→    // Model routing (if auto mode)
   438→    let routeDecision: RoutingDecision | null = null;
   439→    if (mode === 'auto') {
   440→      routeDecision = this.modelRouter.route(message);
   441→      if (routeDecision.model !== this.config.defaultModel) {
   442→        this.ui.info(`Model router: ${routeDecision.complexity} task -> ${MODELS[routeDecision.model]?.name || routeDecision.model}`);
   443→      }
   444→    }
   445→
   446→    // Check prompt cache
   447→    if (this.promptCache && this.config.promptCache.enabled) {
   448→      const session = this.sessionManager.getCurrent();
   449→      if (session) {
   450→        const cached = this.promptCache.get(
   451→          routeDecision?.model || this.config.defaultModel,
   452→          session.messages,
   453→        );
   454→        if (cached) {
   455→          this.ui.info('Cache hit - using cached response');
   456→          return {
   457→            content: cached.response,
   458→            usage: { inputTokens: cached.inputTokens, outputTokens: cached.outputTokens, cost: 0 },
   459→          };
   460→        }
   461→      }
   462→    }
   463→
   464→    // Start or get session
   465→    let session = this.sessionManager.getCurrent();
   466→    if (!session) {
   467→      session = this.sessionManager.create(process.cwd(), this.config.defaultModel);
   468→    }
   469→
   470→    // Add user message to session
   471→    this.sessionManager.addMessage({
   472→      role: 'user',
   473→      content: message,
   474→      timestamp: Date.now(),
   475→    });
   476→
   477→    // Build system prompt additions from skills and styles
   478→    const skillAdditions = this.skillSystem.getSystemPromptAdditions();
   479→    const styleAddition = this.styleManager.getSystemPromptAddition();
   480→    const thinkingAddition = this.extendedThinking.getSystemPromptAddition();
   481→
   482→    // Build UI callbacks
   483→    const callbacks: AgentCallbacks = {
   484→      onThinking: (thinking) => this.ui.thinking(thinking),
   485→      onToken: (token) => this.ui.streamingToken(token),
   486→      onToolCall: (name, args) => {
   487→        // Check .neuroignore for file paths
   488→        if (args.path && typeof args.path === 'string' && this.neuroIgnore.isIgnored(args.path)) {
   489→          this.ui.warning(`Ignored path: ${args.path} (matches .neuroignore rule)`);
   490→          return;
   491→        }
   492→        // Sandbox check before tool execution
   493→        if (this.sandbox.isEnabled() && !this.checkSandboxForTool(name, args)) {
   494→          return;
   495→        }
   496→        this.ui.toolCall(name, args);
   497→
   498→        // Record in undo/redo for file modification tools
   499→        if (['write_file', 'edit_file', 'apply_diff', 'delete_file'].includes(name)) {
   500→          // The undo/redo push will happen in the tool execution result handler
   501→        }
   502→      },
   503→      onToolResult: (name, result, isError) => this.ui.toolResult(name, result, isError),
   504→      onApprovalNeeded: async (name, args, risk) => {
   505→        return this.handleApproval(name, args, risk as 'low' | 'medium' | 'high');
   506→      },
   507→      onIteration: (i, max) => {
   508→        this.ui.info(`Iteration ${i}/${max}`);
   509→      },
   510→    };
   511→
   512→    let result;
   513→    const activeModel = routeDecision?.model || this.config.defaultModel;
   514→
   515→    if (mode === 'direct' && targetAgent) {
   516→      const agent = this.agents.get(targetAgent);
   517→      if (!agent) {
   518→        this.ui.error(`Agent not found: ${targetAgent}`);
   519→        return { content: '', usage: { inputTokens: 0, outputTokens: 0, cost: 0 } };
   520→      }
   521→      this.ui.startStreaming();
   522→      result = await agent.run(message, callbacks);
   523→      this.ui.endStreaming();
   524→    } else if (mode === 'agent') {
   525→      const orchestrateResult = await this.orchestrator.orchestrate(message, callbacks);
   526→      result = {
   527→        content: orchestrateResult.content,
   528→        toolCallsMade: 0,
   529→        iterations: orchestrateResult.execution.iterations,
   530→        usage: orchestrateResult.totalUsage,
   531→        execution: orchestrateResult.execution,
   532→      };
   533→    } else {
   534→      const complexity = routeDecision?.complexity || this.assessComplexity(message);
   535→      if (complexity === 'simple') {
   536→        const agent = this.agents.get('Coder');
   537→        if (agent) {
   538→          this.ui.startStreaming();
   539→          result = await agent.run(message, callbacks);
   540→          this.ui.endStreaming();
   541→        } else {
   542→          throw new Error('Coder agent not initialized');
   543→        }
   544→      } else {
   545→        this.ui.thinking('Analyzing task complexity... Using multi-agent orchestration');
   546→        const orchestrateResult = await this.orchestrator.orchestrate(message, callbacks);
   547→        result = {
   548→          content: orchestrateResult.content,
   549→          toolCallsMade: 0,
   550→          iterations: orchestrateResult.execution.iterations,
   551→          usage: orchestrateResult.totalUsage,
   552→          execution: orchestrateResult.execution,
   553→        };
   554→      }
   555→    }
   556→
   557→    // Parse extended thinking blocks from response
   558→    const thinkingResult = this.extendedThinking.parseResponse(result.content);
   559→    if (thinkingResult.hadThinking && this.extendedThinking.isDisplayEnabled()) {
   560→      // Thinking blocks were already displayed during streaming
   561→    }
   562→    // Use cleaned response (without thinking blocks) if thinking is hidden
   563→    if (thinkingResult.hadThinking && !this.extendedThinking.isDisplayEnabled()) {
   564→      result.content = thinkingResult.cleanedResponse;
   565→    }
   566→
   567→    // Record spending
   568→    const spendResult = this.spendingMonitor.record({
   569→      model: activeModel,
   570→      inputTokens: result.usage.inputTokens,
   571→      outputTokens: result.usage.outputTokens,
   572→      cost: result.usage.cost,
   573→      sessionId: this.sessionManager.getCurrent()?.id || 'unknown',
   574→    });
   575→    if (spendResult.warning) {
   576→      this.ui.warning(spendResult.warning);
   577→    }
   578→    if (!spendResult.allowed) {
   579→      this.ui.error('Spending limit reached. Operation completed but further requests may be blocked.');
   580→    }
   581→
   582→    // Cache the response if caching is enabled
   583→    if (this.promptCache && this.config.promptCache.enabled && session) {
   584→      this.promptCache.set(activeModel, session.messages, result.content, result.usage);
   585→    }
   586→
   587→    // Print usage
   588→    this.ui.tokenUsage(result.usage, activeModel);
   589→
   590→    // Update session
   591→    this.sessionManager.addMessage({
   592→      role: 'assistant',
   593→      content: result.content,
   594→      timestamp: Date.now(),
   595→    });
   596→    this.sessionManager.updateUsage(
   597→      result.usage.inputTokens,
   598→      result.usage.outputTokens,
   599→      result.usage.cost,
   600→    );
   601→    this.sessionManager.save();
   602→
   603→    return {
   604→      content: result.content,
   605→      usage: result.usage,
   606→      execution: result.execution,
   607→    };
   608→  }

   610→  /**
   611→   * Check sandbox permissions for a tool call
   612→   */
   613→  private checkSandboxForTool(toolName: string, args: Record<string, unknown>): boolean {
   614→    // File operations
   615→    if (['write_file', 'edit_file', 'apply_diff'].includes(toolName)) {
   616→      const path = args.path as string;
   617→      if (path && !this.sandbox.canWrite(path)) {
   618→        this.ui.warning(`Sandbox: Write access denied for ${path}`);
   619→        return false;
   620→      }
   621→      // Check neuroignore
   622→      if (path && this.neuroIgnore.isIgnored(path)) {
   623→        this.ui.warning(`Ignored: ${path} is in .neuroignore`);
   624→        return false;
   625→      }
   626→      // Backup file before modification
   627→      if (path) this.sandbox.backupFile(path);
   628→    }
   629→
   630→    if (['delete_file'].includes(toolName)) {
   631→      const path = args.path as string;
   632→      if (path && !this.sandbox.canDelete(path)) {
   633→        this.ui.warning(`Sandbox: Delete access denied for ${path}`);
   634→        return false;
   635→      }
   636→      if (path && this.neuroIgnore.isIgnored(path)) {
   637→        this.ui.warning(`Ignored: ${path} is in .neuroignore`);
   638→        return false;
   639→      }
   640→    }
   641→
   642→    if (['read_file', 'search_files', 'list_directory'].includes(toolName)) {
   643→      const path = (args.path || args.directory) as string;
   644→      if (path && !this.sandbox.canRead(path)) {
   645→        this.ui.warning(`Sandbox: Read access denied for ${path}`);
   646→        return false;
   647→      }
   648→      if (path && this.neuroIgnore.isIgnored(path)) {
   649→        this.ui.warning(`Ignored: ${path} is in .neuroignore`);
   650→        return false;
   651→      }
   652→    }
   653→
   654→    // Command execution
   655→    if (['run_command', 'bash'].includes(toolName)) {
   656→      const command = args.command as string;
   657→      if (command && !this.sandbox.canRunCommand(command)) {
   658→        this.ui.warning('Sandbox: Command execution denied');
   659→        return false;
   660→      }
   661→    }
   662→
   663→    // Network access
   664→    if (['web_search', 'web_fetch'].includes(toolName)) {
   665→      if (!this.sandbox.canAccessNetwork()) {
   666→        this.ui.warning('Sandbox: Network access denied');
   667→        return false;
   668→      }
   669→    }
   670→
   671→    return true;
   672→  }

   677→  private async handleApproval(
   678→    toolName: string,
   679→    args: Record<string, unknown>,
   680→    risk: 'low' | 'medium' | 'high',

   726→   * Register a custom agent
   727→   */
   728→  registerCustomAgent(name: string, config: { description: string; systemPrompt: string; tools?: string[]; maxIterations?: number }): void {
   729→    const agentConfig = {
   730→      name,
   731→      description: config.description,
   732→      systemPrompt: config.systemPrompt,
   733→      model: this.config.defaultModel,
   734→      temperature: 0.5,
   735→      maxTokens: 8192,
   736→      tools: config.tools || [],
   737→      maxIterations: config.maxIterations || 10,
   738→      isCustom: true,
   739→    };
   740→
   741→    const agent = new BaseAgent(
   742→      agentConfig,
   743→      this.client,
   744→      this.registry,
   745→      process.cwd(),
   746→      this.sessionManager.getCurrent()?.id || 'default',
   747→    );
   748→
   749→    this.agents.set(name, agent);
   750→    this.orchestrator.registerAgent(agent);
   751→
   752→    // Save to config
   753→    if (!this.config.customAgents) this.config.customAgents = {};
   754→    this.config.customAgents[name] = agentConfig;
   755→  }
