   416→  /**
   417→   * Register plugin tools with the tool registry
   418→   */
   419→  private registerPluginTools(): void {
   420→    const pluginTools = this.pluginManager.getToolDefinitions();
   421→    for (const toolDef of pluginTools) {
   422→      this.registry.register({
   423→        name: toolDef.name,
   424→        risk: toolDef.risk,
   425→        execute: async (args, context) => {
   426→          const result = await this.pluginManager.executeTool(toolDef.name, args, {
   427→            workingDirectory: context.workingDirectory,
   428→            sessionId: context.sessionId,
   429→            agentName: context.agentName,
   430→            onProgress: context.onProgress || (() => {}),
   431→            callTool: async (name, callArgs) => {
   432→              return this.registry.execute(name, callArgs, context);
   433→            },
   434→            memory: {
   435→              get: () => undefined,
   436→              set: () => {},
   437→              delete: () => {},
   438→              list: () => [],
   439→            },
   440→          });
   441→          return result.content;
   442→        },
   443→      });
   444→    }
   445→  }

   447→  /**
   448→   * Register custom tools from .neuro/tools/
   449→   */
   450→  private registerCustomTools(): void {
   451→    const customTools = this.customToolLoader.getAll();
   452→    for (const toolDef of customTools) {
   453→      const executor = this.customToolLoader.createExecutor(toolDef);
   454→      this.registry.register({
   455→        name: `custom_${toolDef.name}`,
   456→        risk: toolDef.risk || 'medium',
   457→        execute: async (args) => {
   458→          try {
   459→            const result = await executor(args);
   460→            return typeof result === 'string' ? result : JSON.stringify(result);
   461→          } catch (error) {
   462→            return `Custom tool error: ${error instanceof Error ? error.message : String(error)}`;
   463→          }
   464→        },
   465→      });
   466→    }
   467→  }

   469→  /**
   470→   * Load custom agents from .neuro/agents/

   472→  private loadCustomAgents(): void {
   473→    const customAgents = this.customAgentLoader.getAll();
   474→    const cwd = process.cwd();
   475→    const sessionId = this.sessionManager.getCurrent()?.id || 'default';
   476→
   477→    for (const def of customAgents) {
   478→      const agentConfig = this.customAgentLoader.toAgentConfig(def, this.config.defaultModel);
   479→      const agent = new BaseAgent(agentConfig, this.client, this.registry, cwd, sessionId);
   480→      this.agents.set(def.name, agent);
   481→
   482→      // Save to config custom agents
   483→      if (!this.config.customAgents) this.config.customAgents = {};
   484→      this.config.customAgents[def.name] = agentConfig;
   485→    }
   486→  }

   508→    // Custom agents from config
   509→    for (const [key, agentConfig] of Object.entries(this.config.customAgents || {})) {
   510→      const overrideConfig = { ...agentConfig, model: this.config.defaultModel, isCustom: true };
   511→      const agent = new BaseAgent(
   512→        overrideConfig,
   513→        this.client,
   514→        this.registry,
   515→        cwd,
   516→        sessionId,
   517→      );
   518→      this.agents.set(agentConfig.name, agent);
   519→    }
   520→  }

   522→  /**
   523→   * Process a user message
   524→   */
   525→  async processMessage(
   526→    message: string,
   527→    mode: 'auto' | 'agent' | 'direct' = 'auto',
   528→    targetAgent?: string,
   529→  ): Promise<{ content: string; usage: TokenUsage; execution?: AgentExecution }> {
   530→    // Check spending limit via spending monitor
   531→    const limitCheck = this.spendingMonitor.checkLimit();
   532→    if (!limitCheck.allowed) {
   533→      this.ui.error(`Spending limit reached (${limitCheck.limitReached}). Use /spending to check or /config to adjust.`);
   534→      return { content: 'Spending limit reached.', usage: { inputTokens: 0, outputTokens: 0, cost: 0 } };
   535→    }
   536→
   537→    // Auto-activate skills based on prompt
   538→    const activatedSkills = this.skillSystem.autoActivate(message);
   539→    if (activatedSkills.length > 0) {
   540→      for (const skill of activatedSkills) {
   541→        this.ui.info(`Skill activated: ${skill.skill.name} (${skill.activatedBy})`);
   542→      }
   543→    }
   544→
   545→    // Model routing (if auto mode)
   546→    let routeDecision: RoutingDecision | null = null;
   547→    if (mode === 'auto') {
   548→      routeDecision = this.modelRouter.route(message);
   549→      if (routeDecision.model !== this.config.defaultModel) {
   550→        this.ui.info(`Model router: ${routeDecision.complexity} task -> ${MODELS[routeDecision.model]?.name || routeDecision.model}`);
   551→      }
   552→    }
   553→
   554→    // Check prompt cache
   555→    if (this.promptCache && this.config.promptCache.enabled) {
   556→      const session = this.sessionManager.getCurrent();
   557→      if (session) {
   558→        const cached = this.promptCache.get(
   559→          routeDecision?.model || this.config.defaultModel,
   560→          session.messages,
   561→        );
   562→        if (cached) {
   563→          this.ui.info('Cache hit - using cached response');
   564→          return {
   565→            content: cached.response,
   566→            usage: { inputTokens: cached.inputTokens, outputTokens: cached.outputTokens, cost: 0 },
   567→          };
   568→        }
   569→      }
   570→    }
   571→
   572→    // Start or get session
   573→    let session = this.sessionManager.getCurrent();
   574→    if (!session) {
   575→      session = this.sessionManager.create(process.cwd(), this.config.defaultModel);
   576→    }
   577→
   578→    // Add user message to session
   579→    this.sessionManager.addMessage({
   580→      role: 'user',
   581→      content: message,
   582→      timestamp: Date.now(),
   583→    });
   584→
   585→    // Build system prompt additions from skills and styles
   586→    const skillAdditions = this.skillSystem.getSystemPromptAdditions();
   587→    const styleAddition = this.styleManager.getSystemPromptAddition();
   588→    const thinkingAddition = this.extendedThinking.getSystemPromptAddition();
   589→
   590→    // Build UI callbacks
   591→    const callbacks: AgentCallbacks = {
   592→      onThinking: (thinking) => this.ui.thinking(thinking),
   593→      onToken: (token) => this.ui.streamingToken(token),
   594→      onToolCall: (name, args) => {
   595→        // Check .neuroignore for file paths
   596→        if (args.path && typeof args.path === 'string' && this.neuroIgnore.isIgnored(args.path)) {
   597→          this.ui.warning(`Ignored path: ${args.path} (matches .neuroignore rule)`);
   598→          return;
   599→        }
   600→        // Sandbox check before tool execution
   601→        if (this.sandbox.isEnabled() && !this.checkSandboxForTool(name, args)) {
   602→          return;
   603→        }
   604→        this.ui.toolCall(name, args);
   605→
   606→        // Record in undo/redo for file modification tools
   607→        if (['write_file', 'edit_file', 'apply_diff', 'delete_file'].includes(name)) {
   608→          // The undo/redo push will happen in the tool execution result handler
   609→        }
   610→      },
   611→      onToolResult: (name, result, isError) => this.ui.toolResult(name, result, isError),
   612→      onApprovalNeeded: async (name, args, risk) => {
   613→        return this.handleApproval(name, args, risk as 'low' | 'medium' | 'high');
   614→      },
   615→      onIteration: (i, max) => {
   616→        this.ui.info(`Iteration ${i}/${max}`);
   617→      },
   618→    };
   619→
   620→    let result;
   621→    const activeModel = routeDecision?.model || this.config.defaultModel;
   622→
   623→    if (mode === 'direct' && targetAgent) {
   624→      const agent = this.agents.get(targetAgent);
   625→      if (!agent) {
   626→        this.ui.error(`Agent not found: ${targetAgent}`);
   627→        return { content: '', usage: { inputTokens: 0, outputTokens: 0, cost: 0 } };
   628→      }
   629→      // Override agent model if defaultModel differs from agent's model
   630→      const originalModel = agent.configModel;
   631→      if (this.config.defaultModel !== originalModel) {
   632→        agent.configModel = this.config.defaultModel;
   633→      }
   634→      this.ui.startStreaming();
   635→      result = await agent.run(message, callbacks);
   636→      this.ui.endStreaming();
   637→      // Restore original model
   638→      agent.configModel = originalModel;
   639→    } else if (mode === 'agent') {
   640→      const orchestrateResult = await this.orchestrator.orchestrate(message, callbacks);
   641→      result = {
   642→        content: orchestrateResult.content,
   643→        toolCallsMade: 0,
   644→        iterations: orchestrateResult.execution.iterations,
   645→        usage: orchestrateResult.totalUsage,
   646→        execution: orchestrateResult.execution,
   647→      };
   648→    } else {
   649→      const complexity = routeDecision?.complexity || this.assessComplexity(message);
   650→      if (complexity === 'simple') {
   651→        const agent = this.agents.get('Coder');
   652→        if (agent) {
   653→          this.ui.startStreaming();
   654→          result = await agent.run(message, callbacks);
   655→          this.ui.endStreaming();
   656→        } else {
   657→          throw new Error('Coder agent not initialized');
   658→        }
   659→      } else {
   660→        this.ui.thinking('Analyzing task complexity... Using multi-agent orchestration');
   661→        const orchestrateResult = await this.orchestrator.orchestrate(message, callbacks);
   662→        result = {
   663→          content: orchestrateResult.content,
   664→          toolCallsMade: 0,
   665→          iterations: orchestrateResult.execution.iterations,
   666→          usage: orchestrateResult.totalUsage,
   667→          execution: orchestrateResult.execution,
   668→        };
   669→      }
   670→    }
   671→
   672→    // Parse extended thinking blocks from response
   673→    const thinkingResult = this.extendedThinking.parseResponse(result.content);
   674→    if (thinkingResult.hadThinking && this.extendedThinking.isDisplayEnabled()) {
   675→      // Thinking blocks were already displayed during streaming
   676→    }
   677→    // Use cleaned response (without thinking blocks) if thinking is hidden
   678→    if (thinkingResult.hadThinking && !this.extendedThinking.isDisplayEnabled()) {
   679→      result.content = thinkingResult.cleanedResponse;
   680→    }
   681→
   682→    // Record spending
   683→    const spendResult = this.spendingMonitor.record({
   684→      model: activeModel,
   685→      inputTokens: result.usage.inputTokens,
   686→      outputTokens: result.usage.outputTokens,
   687→      cost: result.usage.cost,
   688→      sessionId: this.sessionManager.getCurrent()?.id || 'unknown',
   689→    });
   690→    if (spendResult.warning) {
   691→      this.ui.warning(spendResult.warning);
   692→    }
   693→    if (!spendResult.allowed) {
   694→      this.ui.error('Spending limit reached. Operation completed but further requests may be blocked.');
   695→    }
   696→
   697→    // Cache the response if caching is enabled
   698→    if (this.promptCache && this.config.promptCache.enabled && session) {
   699→      this.promptCache.set(activeModel, session.messages, result.content, result.usage);
   700→    }
   701→
   702→    // Print usage
   703→    this.ui.tokenUsage(result.usage, activeModel);
   704→
   705→    // Update session
   706→    this.sessionManager.addMessage({
   707→      role: 'assistant',
   708→      content: result.content,
   709→      timestamp: Date.now(),
   710→    });
   711→    this.sessionManager.updateUsage(
   712→      result.usage.inputTokens,
   713→      result.usage.outputTokens,
   714→      result.usage.cost,
   715→    );
   716→    this.sessionManager.save();
   717→
   718→    return {
   719→      content: result.content,
   720→      usage: result.usage,
   721→      execution: result.execution,
   722→    };
   723→  }

   725→  /**
   726→   * Check sandbox permissions for a tool call
   727→   */
   728→  private checkSandboxForTool(toolName: string, args: Record<string, unknown>): boolean {
   729→    // File operations
   730→    if (['write_file', 'edit_file', 'apply_diff'].includes(toolName)) {
   731→      const path = args.path as string;
   732→      if (path && !this.sandbox.canWrite(path)) {
   733→        this.ui.warning(`Sandbox: Write access denied for ${path}`);
   734→        return false;
   735→      }
   736→      // Check neuroignore
   737→      if (path && this.neuroIgnore.isIgnored(path)) {
   738→        this.ui.warning(`Ignored: ${path} is in .neuroignore`);
   739→        return false;
   740→      }
   741→      // Backup file before modification
   742→      if (path) this.sandbox.backupFile(path);
   743→    }
   744→
   745→    if (['delete_file'].includes(toolName)) {
   746→      const path = args.path as string;
   747→      if (path && !this.sandbox.canDelete(path)) {
   748→        this.ui.warning(`Sandbox: Delete access denied for ${path}`);
   749→        return false;
   750→      }
   751→      if (path && this.neuroIgnore.isIgnored(path)) {
   752→        this.ui.warning(`Ignored: ${path} is in .neuroignore`);
   753→        return false;
   754→      }
   755→    }
   756→
   757→    if (['read_file', 'search_files', 'list_directory'].includes(toolName)) {
   758→      const path = (args.path || args.directory) as string;
   759→      if (path && !this.sandbox.canRead(path)) {
   760→        this.ui.warning(`Sandbox: Read access denied for ${path}`);
   761→        return false;
   762→      }
   763→      if (path && this.neuroIgnore.isIgnored(path)) {
   764→        this.ui.warning(`Ignored: ${path} is in .neuroignore`);
   765→        return false;
   766→      }
   767→    }
   768→
   769→    // Command execution
   770→    if (['run_command', 'bash'].includes(toolName)) {
   771→      const command = args.command as string;
   772→      if (command && !this.sandbox.canRunCommand(command)) {
   773→        this.ui.warning('Sandbox: Command execution denied');
   774→        return false;
   775→      }
   776→    }
   777→
   778→    // Network access
   779→    if (['web_search', 'web_fetch'].includes(toolName)) {
   780→      if (!this.sandbox.canAccessNetwork()) {
   781→        this.ui.warning('Sandbox: Network access denied');
   782→        return false;
   783→      }
   784→    }
   785→
   786→    return true;
   787→  }

   789→  /**
   790→   * Handle tool approval using the enhanced ApprovalSystem
   791→   */
   792→  private async handleApproval(
   793→    toolName: string,
   794→    args: Record<string, unknown>,
   795→    risk: 'low' | 'medium' | 'high',
   796→  ): Promise<boolean> {
   797→    if (this.autoApproveSet.has(toolName)) return true;
   798→
   799→    const result = await this.approval.requestApproval(toolName, args, risk);
   800→    return result.approved;
   801→  }

   840→  /**
   841→   * Register a custom agent
   842→   */
   843→  registerCustomAgent(name: string, config: { description: string; systemPrompt: string; tools?: string[]; maxIterations?: number }): void {
   844→    const agentConfig = {
   845→      name,
   846→      description: config.description,
   847→      systemPrompt: config.systemPrompt,
   848→      model: this.config.defaultModel,
   849→      temperature: 0.5,
   850→      maxTokens: 8192,
   851→      tools: config.tools || [],
   852→      maxIterations: config.maxIterations || 10,
   853→      isCustom: true,
   854→    };
   855→
   856→    const agent = new BaseAgent(
   857→      agentConfig,
   858→      this.client,
   859→      this.registry,
   860→      process.cwd(),
   861→      this.sessionManager.getCurrent()?.id || 'default',
   862→    );
   863→
   864→    this.agents.set(name, agent);
   865→    this.orchestrator.registerAgent(agent);
   866→
   867→    // Save to config
   868→    if (!this.config.customAgents) this.config.customAgents = {};
   869→    this.config.customAgents[name] = agentConfig;
   870→  }
