   257→   257→// ---- Interactive Mode ----
   258→   258→async function startInteractive(options: any) {
   259→   259→  const config = initConfig(options.apiKey);
   260→   260→  if (options.model) config.defaultModel = options.model;
   261→   261→  if (options.theme) config.ui.theme = options.theme as 'dark' | 'light' | 'dracula' | 'nord';
   262→   262→  if (options.noStreaming) config.ui.streaming = false;
   263→   263→  if (options.permissionMode) config.permissionMode = options.permissionMode;
   264→   264→  if (options.diffPreview === true) config.diffPreview = true;
   265→   265→  if (options.diffPreview === false) config.diffPreview = false;
   266→   266→  if (options.cache === true) config.promptCache.enabled = true;
   267→   267→  if (options.cache === false) config.promptCache.enabled = false;
   268→   268→  if (options.spendingLimit) config.spendingLimit = options.spendingLimit;
   269→   269→  if (options.autoApprove) {
   270→   270→    config.tools.autoApprove = [...config.tools.autoApprove, ...config.tools.requireApproval];
   271→   271→    config.tools.requireApproval = [];
   272→   272→    config.permissionMode = 'yolo';
   273→   273→  }
   274→   274→
   275→   275→  // Check API key
   276→   276→  if (!config.apiKey && !options.ollama) {
   277→   277→    console.log(chalk.red('\nOpenRouter API key not configured!'));
   278→   278→    console.log(chalk.yellow('\nSet it with:'));
   279→   279→    console.log(`  ${chalk.cyan('neuro config --set-key YOUR_API_KEY')}`);
   280→   280→    console.log(`  ${chalk.cyan('export OPENROUTER_API_KEY=YOUR_API_KEY')}`);
   281→   281→    console.log(`  ${chalk.cyan('neuro -k YOUR_API_KEY')}`);
   282→   282→    console.log(chalk.gray('\nGet your key at: https://openrouter.ai/keys\n'));
   283→   283→    process.exit(1);
   284→   284→  }
   285→   285→
   286→   286→  // Initialize engine
   287→   287→  const engine = new NeuroEngine(config);
   288→   288→
   289→   289→  // Set effort level if specified
   290→   290→  if (options.effort) {
   291→   291→    engine.modelRouter.setEffort(options.effort);
   292→   292→  }
   293→   293→
   294→   294→  // Set output style if specified
   295→   295→  if (options.style) {
   296→   296→    engine.styleManager.setStyle(options.style);
   297→   297→  }
   298→   298→
   299→   299→  // Enable thinking if specified
   300→   300→  if (options.thinking) {
   301→   301→    engine.extendedThinking.setMode('full');
   302→   302→    engine.extendedThinking.toggleDisplay();
   303→   303→  }
   304→   304→
   305→   305→  // Enable sandbox if specified
   306→   306→  if (options.sandbox) {
   307→   307→    engine.sandbox.enable();
   308→   308→  }
   309→   309→
   310→   310→  // Enable prompt cache if specified
   311→   311→  if (options.cache) {
   312→   312→    config.promptCache.enabled = true;
   313→   313→  }
   314→   314→
   315→   315→  // Resume session if requested
   316→   316→  if (options.resume) {
   317→   317→    const session = engine.sessionManager.load(options.resume);
   318→   318→    if (session) {
   319→   319→      engine.ui.success(`Resumed session: ${session.id.slice(0, 20)}...`);
   320→   320→      if (options.fork) {
   321→   321→        const forked = engine.sessionManager.create(process.cwd(), config.defaultModel);
   322→   322→        forked.messages = [...session.messages];
   323→   323→        engine.sessionManager.save();
   324→   324→        engine.ui.success(`Forked to new session: ${forked.id.slice(0, 20)}...`);
   325→   325→      }
   326→   326→    } else {
   327→   327→      engine.ui.error(`Session not found: ${options.resume}`);
   328→   328→    }
   329→   329→  } else if (options.continue) {
   330→   330→    const sessions = engine.sessionManager.list();
   331→   331→    if (sessions.length > 0) {
   332→   332→      const session = engine.sessionManager.load(sessions[0].id);
   333→   333→      if (session) {
   334→   334→        engine.ui.success(`Continued session: ${session.id.slice(0, 20)}...`);
   335→   335→      }
   336→   336→    } else {
   337→   337→      engine.ui.warning('No sessions found to continue');
   338→   338→    }
   339→   339→  }
   340→   340→
   341→   341→  // Print banner
   342→   342→  engine.ui.banner();
   343→   343→  const theme = engine.ui.theme;
   344→   344→  const permMode = engine.approval.getMode();
   345→   345→  const permIcon = engine.approval.getModeIcon();
   346→   346→  const activeStyle = engine.styleManager.getStyle();
   347→   347→  console.log(theme.muted(`  Model: ${MODELS[config.defaultModel]?.name || config.defaultModel}`));
   348→   348→  console.log(theme.muted(`  Permission: ${permIcon} ${permMode}`));
   349→   349→  console.log(theme.muted(`  Style: ${activeStyle.name}`));
   350→   350→  console.log(theme.muted(`  Thinking: ${engine.extendedThinking.getMode()}`));
   351→   351→  console.log(theme.muted(`  Cache: ${config.promptCache.enabled ? 'on' : 'off'}`));
   352→   352→  console.log(theme.muted(`  Working Dir: ${process.cwd()}`));
   353→   353→  console.log(theme.muted(`  Type /help for commands, Tab for completion, Ctrl+C to exit\n`));
   354→   354→
   355→   355→  // Create readline with tab completion
   356→   356→  const completionEngine = new CompletionEngine(process.cwd());
   357→   357→  completionEngine.setAgentNames(Array.from(engine.agents.keys()));
   358→   358→
   359→   359→  const rl = createInterface({
   360→   360→    input: process.stdin,
   361→   361→    output: process.stdout,
   362→   362→    prompt: chalk.cyan('❯ '),
   363→   363→    historySize: 100,
   364→   364→    completer: completionEngine.complete,
   365→   365→  });
   366→   366→
   367→   367→  rl.prompt();
   368→   368→
   369→   369→  let currentMode: 'auto' | 'agent' | 'direct' = 'auto';
   370→   370→  let currentAgent: string | undefined;
   371→   371→
   372→   372→  rl.on('line', async (line) => {
   373→   373→    const input = line.trim();
   374→   374→    if (!input) { rl.prompt(); return; }
   375→   375→
   376→   376→    // Add to history
   377→   377→    completionEngine.addHistory(input);
   378→   378→
   379→   379→    // Handle slash commands
   380→   380→    if (input.startsWith('/')) {
   381→   381→      const [cmd, ...args] = input.slice(1).split(' ');
   382→   382→
   383→   383→      switch (cmd) {
   384→   384→        case 'help':
   385→   385→          printHelp(engine);
   386→   386→          break;
   387→   387→
   388→   388→        case 'model':
   389→   389→          if (args[0]) {
   390→   390→            engine.switchModel(args[0]);
   391→   391→            engine.fallback.addFallbackModel(config.defaultModel);
   392→   392→          } else {
   393→   393→            engine.ui.modelList(config.defaultModel);
   394→   394→          }
   395→   395→          break;
   396→   396→
   397→   397→        case 'agent':
   398→   398→          if (args[0]) {
   399→   399→            const agentNames = Array.from(engine.agents.keys());
   400→   400→            const found = agentNames.find(n => n.toLowerCase() === args[0].toLowerCase());
   401→   401→            if (found) { currentAgent = found; currentMode = 'direct'; engine.ui.success(`Switched to agent: ${found}`); }
   402→   402→            else { engine.ui.error(`Agent not found. Available: ${agentNames.join(', ')}`); }

   888→   888→
   889→   889→        case 'exit':
   890→   890→        case 'quit':
   891→   891→        case 'q':
   892→   892→          engine.mcpClient.disconnectAll().catch(() => {});
   893→   893→          engine.approval.close();
   894→   894→          engine.ui.info('Goodbye!');
   895→   895→          process.exit(0);
   896→   896→          break;
   897→   897→
   898→   898→        default:

   900→   900→      }
   901→   901→
   902→   902→      rl.prompt();
   903→   903→      return;
   904→   904→    }
   905→   905→
   906→   906→    // Process message with the engine
   907→   907→    try {

   909→   909→    } catch (error) {
   910→   910→      engine.ui.error(error instanceof Error ? error.message : String(error));
   911→   911→    }
   912→   912→
   913→   913→    rl.prompt();
   914→   914→  });
   915→   915→

   919→   919→    engine.ui.info('Goodbye!');

   921→   921→  });
