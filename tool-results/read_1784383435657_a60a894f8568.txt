   204→   204→  /**
   205→   205→   * Route a prompt to the best available model.
   206→   206→   */
   207→   207→  route(prompt: string, effort?: EffortLevel): RoutingDecision {
   208→   208→    const effectiveEffort = effort ?? this.currentEffort;
   209→   209→    const analysis = this.analyzePrompt(prompt);
   210→   210→    const estimatedTokens = this.estimateTokens(prompt);
   211→   211→    const model = this.selectModel(analysis.complexity, analysis.category, effectiveEffort);
   212→   212→    const alternatives = this.getAlternatives(analysis.complexity, analysis.category);
   213→   213→    const reasoning = this.buildReasoning(analysis, effectiveEffort, model);
   214→   214→
   215→   215→    return {
   216→   216→      model,
   217→   217→      complexity: analysis.complexity,
   218→   218→      category: analysis.category,
   219→   219→      effort: effectiveEffort,
   220→   220→      estimatedTokens,
   221→   221→      reasoning,
   222→   222→      alternatives,
   223→   223→    };
   224→   224→  }

   272→   272→
   273→   273→  /**
   274→   274→   * Estimate the number of tokens a prompt will consume.
   275→   275→   * Uses a character-based heuristic (~4 chars per token for English,
   276→   276→   * with adjustments for code-heavy content).
   277→   277→   */
   278→   278→  estimateTokens(prompt: string): number {
   279→   279→    if (prompt.length === 0) return 0;
   280→   280→
   281→   281→    // Base estimate: ~4 characters per token for typical English text
   282→   282→    let charPerToken = 4;
   283→   283→
   284→   284→    // Code-heavy prompts tend to have higher token density
   285→   285→    const codeIndicators = [
   286→   286→      /```[\s\S]*?```/,       // fenced code blocks
   287→   287→      /\bfunction\b/,          // function keyword
   288→   288→      /\bconst\b|\blet\b|\bvar\b/,  // variable declarations
   289→   289→      /[{}()\[\];]/,           // braces and brackets
   290→   290→      /=>/,                    // arrow functions
   291→   291→      /import\s+.+\s+from/,   // imports
   292→   292→    ];
   293→   293→
   294→   294→    const codeSignalCount = codeIndicators.filter(p => p.test(prompt)).length;
   295→   295→    if (codeSignalCount >= 3) {
   296→   296→      charPerToken = 3.2; // code is denser
   297→   297→    } else if (codeSignalCount >= 1) {
   298→   298→      charPerToken = 3.6;
   299→   299→    }
   300→   300→
   301→   301→    // Whitespace-heavy prompts have lower density
   302→   302→    const whitespaceRatio = (prompt.match(/\s/g) ?? []).length / prompt.length;
   303→   303→    if (whitespaceRatio > 0.4) {
   304→   304→      charPerToken = 4.5;
   305→   305→    }
   306→   306→
   307→   307→    const baseTokens = Math.ceil(prompt.length / charPerToken);
   308→   308→
   309→   309→    // Add overhead for special tokens (system, role markers, etc.)
   310→   310→    const overhead = 8;
   311→   311→
   312→   312→    return baseTokens + overhead;
   313→   313→  }

   343→   343→  // -----------------------------------------------------------------------
   344→   344→  // Private: prompt analysis
   345→   345→  // -----------------------------------------------------------------------
   346→   346→
   347→   347→  private analyzePrompt(prompt: string): PromptAnalysis {
   348→   348→    const indicators: string[] = [];
   349→   349→
   350→   350→    // --- Category detection (scored) ---
   351→   351→    const categoryScores = this.scoreCategories(prompt);
   352→   352→    const category = this.pickTopCategory(categoryScores, indicators);
   353→   353→
   354→   354→    // --- Complexity detection (scored) ---
   355→   355→    const complexity = this.scoreComplexity(prompt, indicators);
   356→   356→
   357→   357→    return { complexity, category, indicators };
   358→   358→  }
   359→   359→
   360→   360→  /**
   361→   361→   * Score each category based on how many pattern groups match.
   362→   362→   * Each group in CATEGORY_PATTERNS is worth 1 point when at least one
   363→   363→   * pattern in the group matches.
   364→   364→   */
   365→   365→  private scoreCategories(prompt: string): Map<TaskCategory, number> {
   366→   366→    const scores = new Map<TaskCategory, number>();
   367→   367→
   368→   368→    for (const [cat, patternGroups] of Object.entries(CATEGORY_PATTERNS) as [TaskCategory, RegExp[]][]) {
   369→   369→      let score = 0;
   370→   370→      for (const pattern of patternGroups) {
   371→   371→        if (pattern.test(prompt)) {
   372→   372→          score += 1;
   373→   373→        }
   374→   374→      }
   375→   375→      scores.set(cat, score);
   376→   376→    }
   377→   377→
   378→   378→    return scores;
   379→   379→  }
   380→   380→
   381→   381→  /**
   382→   382→   * Determine the winning category and record indicators.
   383→   383→   */
   384→   384→  private pickTopCategory(
   385→   385→    scores: Map<TaskCategory, number>,
   386→   386→    indicators: string[],
   387→   387→  ): TaskCategory {
   388→   388→    let bestCategory: TaskCategory = 'conversation';
   389→   389→    let bestScore = 0;
   390→   390→
   391→   391→    for (const [cat, score] of scores) {
   392→   392→      if (score > bestScore) {
   393→   393→        bestScore = score;
   394→   394→        bestCategory = cat;
   395→   395→      }
   396→   396→    }
   397→   397→
   398→   398→    // If nothing matched at all, default to conversation
   399→   399→    if (bestScore === 0) {
   400→   400→      indicators.push('no strong category signals -- defaulting to conversation');
   401→   401→      return 'conversation';
   402→   402→    }
   403→   403→
   404→   404→    indicators.push(`category "${bestCategory}" matched with score ${bestScore}`);
   405→   405→    return bestCategory;
   406→   406→  }
   407→   407→
   408→   408→  /**
   409→   409→   * Score complexity using a weighted combination of:
   410→   410→   *   1. Pattern match signals
   411→   411→   *   2. Prompt length
   412→   412→   *   3. Multi-step indicators
   413→   413→   */
   414→   414→  private scoreComplexity(prompt: string, indicators: string[]): Complexity {
   415→   415→    let simpleScore = 0;
   416→   416→    let moderateScore = 0;
   417→   417→    let complexScore = 0;
   418→   418→
   419→   419→    // --- Pattern-based scoring ---
   420→   420→
   421→   421→    for (const pattern of COMPLEXITY_SIGNALS.simple.patterns) {
   422→   422→      if (pattern.test(prompt)) {
   423→   423→        simpleScore += 2;
   424→   424→        indicators.push(`simple signal: "${this.truncateMatch(prompt, pattern)}"`);
   425→   425→      }
   426→   426→    }
   427→   427→
   428→   428→    for (const pattern of COMPLEXITY_SIGNALS.moderate.patterns) {
   429→   429→      if (pattern.test(prompt)) {
   430→   430→        moderateScore += 2;
   431→   431→        indicators.push(`moderate signal: "${this.truncateMatch(prompt, pattern)}"`);
   432→   432→      }
   433→   433→    }
   434→   434→
   435→   435→    for (const pattern of COMPLEXITY_SIGNALS.complex.patterns) {
   436→   436→      if (pattern.test(prompt)) {
   437→   437→        complexScore += 3;
   438→   438→        indicators.push(`complex signal: "${this.truncateMatch(prompt, pattern)}"`);
   439→   439→      }
   440→   440→    }
   441→   441→
   442→   442→    // --- Length-based scoring ---
   443→   443→    const length = prompt.length;
   444→   444→
   445→   445→    if (length <= COMPLEXITY_SIGNALS.simple.lengthThreshold) {
   446→   446→      simpleScore += 1;
   447→   447→      indicators.push(`short prompt (${length} chars)`);
   448→   448→    } else if (length <= COMPLEXITY_SIGNALS.moderate.lengthThreshold) {
   449→   449→      moderateScore += 1;
   450→   450→      indicators.push(`medium prompt (${length} chars)`);
   451→   451→    } else if (length <= COMPLEXITY_SIGNALS.complex.lengthThreshold) {
   452→   452→      moderateScore += 1;
   453→   453→      complexScore += 1;
   454→   454→      indicators.push(`long prompt (${length} chars)`);
   455→   455→    } else {
   456→   456→      complexScore += 2;
   457→   457→      indicators.push(`very long prompt (${length} chars)`);
   458→   458→    }
   459→   459→
   460→   460→    // --- Step-count scoring ---
   461→   461→    const stepCount = this.countSteps(prompt);
   462→   462→    if (stepCount <= COMPLEXITY_SIGNALS.simple.maxSteps) {
   463→   463→      simpleScore += 1;
   464→   464→    } else if (stepCount <= COMPLEXITY_SIGNALS.moderate.maxSteps) {
   465→   465→      moderateScore += 1;
   466→   466→      indicators.push(`multi-step request (${stepCount} steps detected)`);
   467→   467→    } else {
   468→   468→      complexScore += 2;
   469→   469→      indicators.push(`many-step request (${stepCount} steps detected)`);
   470→   470→    }
   471→   471→
   472→   472→    // --- Final decision ---
   473→   473→    const maxScore = Math.max(simpleScore, moderateScore, complexScore);
   474→   474→
   475→   475→    // Ties go to the higher complexity (safer default)
   476→   476→    if (complexScore === maxScore) return 'complex';
   477→   477→    if (moderateScore === maxScore) return 'moderate';
   478→   478→    return 'simple';
   479→   479→  }
   480→   480→
   481→   481→  /**
   482→   482→   * Count the approximate number of distinct steps or sub-tasks in a prompt.
   483→   483→   */
   484→   484→  private countSteps(prompt: string): number {
   485→   485→    let steps = 1; // at least one step
   486→   486→    for (const pattern of STEP_INDICATORS) {
   487→   487→      const matches = prompt.match(pattern);
   488→   488→      if (matches) {
   489→   489→        steps += matches.length;
   490→   490→      }
   491→   491→    }
   492→   492→    return steps;
   493→   493→  }
   494→   494→
   495→   495→  /**
   496→   496→   * Extract a short snippet from the prompt that matched a pattern,
   497→   497→   * for use as a human-readable indicator.
   498→   498→   */
   499→   499→  private truncateMatch(prompt: string, pattern: RegExp): string {
   500→   500→    const match = prompt.match(pattern);
   501→   501→    if (!match) return '(no match)';
   502→   502→    const text = match[0];
   503→   503→    return text.length > 40 ? text.slice(0, 37) + '...' : text;
   504→   504→  }

   506→   506→  // -----------------------------------------------------------------------
   507→   507→  // Private: model selection
   508→   508→  // -----------------------------------------------------------------------
   509→   509→
   510→   510→  private selectModel(complexity: Complexity, category: TaskCategory, effort: EffortLevel): string {
   511→   511→    // 1. User override takes absolute precedence
   512→   512→    if (this.forcedModel) {
   513→   513→      return this.forcedModel;
   514→   514→    }
   515→   515→
   516→   516→    // 2. Effort-level model mapping takes precedence when the effort is
   517→   517→    //    explicitly above "medium". This lets users escalate thinking
   518→   518→    //    without changing the prompt.
   519→   519→    if (effort === 'ultrathink' || effort === 'high') {
   520→   520→      const effortModel = this.config.effortModels[effort];
   521→   521→      if (effortModel && this.availableModels[effortModel]) {
   522→   522→        return effortModel;
   523→   523→      }
   524→   524→    }
   525→   525→
   526→   526→    // 3. Category-specific override
   527→   527→    const categoryModel = this.config.categoryOverrides[category];
   528→   528→    if (categoryModel && this.availableModels[categoryModel]) {
   529→   529→      return categoryModel;
   530→   530→    }
   531→   531→
   532→   532→    // 4. Effort-level mapping for low/medium
   533→   533→    if (effort === 'low' || effort === 'medium') {
   534→   534→      const effortModel = this.config.effortModels[effort];
   535→   535→      if (effortModel && this.availableModels[effortModel]) {
   536→   536→        return effortModel;
   537→   537→      }
   538→   538→    }
   539→   539→
   540→   540→    // 5. Complexity-based default
   541→   541→    switch (complexity) {
   542→   542→      case 'simple':
   543→   543→        return this.config.simpleModel;
   544→   544→      case 'moderate':
   545→   545→        return this.config.moderateModel;
   546→   546→      case 'complex':
   547→   547→        return this.config.complexModel;
   548→   548→    }
   549→   549→  }
   550→   550→
   551→   551→  /**
   552→   552→   * Return alternative models that could handle the same task but were not
   553→   553→   * selected as the primary choice.
   554→   554→   */
   555→   555→  private getAlternatives(complexity: Complexity, category: TaskCategory): string[] {
   556→   556→    const selected = this.selectModel(complexity, category, this.currentEffort);
   557→   557→    const candidates = new Set<string>();
   558→   558→
   559→   559→    // Add complexity-adjacent models
   560→   560→    candidates.add(this.config.simpleModel);
   561→   561→    candidates.add(this.config.moderateModel);
   562→   562→    candidates.add(this.config.complexModel);
   563→   563→
   564→   564→    // Add effort models
   565→   565→    for (const model of Object.values(this.config.effortModels)) {
   566→   566→      candidates.add(model);
   567→   567→    }
   568→   568→
   569→   569→    // Add category override if present
   570→   570→    const categoryModel = this.config.categoryOverrides[category];
   571→   571→    if (categoryModel) {
   572→   572→      candidates.add(categoryModel);
   573→   573→    }
   574→   574→
   575→   575→    // Remove the selected model and any models not in the available set
   576→   576→    candidates.delete(selected);
   577→   577→    for (const candidate of [...candidates]) {
   578→   578→      if (!this.availableModels[candidate]) {
   579→   579→        candidates.delete(candidate);
   580→   580→      }
   581→   581→    }
   582→   582→
   583→   583→    return [...candidates];
   584→   584→  }

   586→   586→  // -----------------------------------------------------------------------
   587→   587→  // Private: reasoning string
   588→   588→  // -----------------------------------------------------------------------
   589→   589→
   590→   590→  private buildReasoning(
   591→   591→    analysis: PromptAnalysis,
   592→   592→    effort: EffortLevel,
   593→   593→    model: string,
   594→   594→  ): string {
   595→   595→    const parts: string[] = [];
   596→   596→
   597→   597→    if (this.forcedModel) {
   598→   598→      parts.push(`User override forced model to "${this.forcedModel}".`);
   599→   599→      return parts.join(' ');
   600→   600→    }
   601→   601→
   602→   602→    parts.push(`Prompt classified as ${analysis.complexity} ${analysis.category} task.`);
   603→   603→
   604→   604→    switch (analysis.complexity) {
   605→   605→      case 'simple':
   606→   606→        parts.push('Routed to fast model for quick response.');
   607→   607→        break;
   608→   608→      case 'moderate':
   609→   609→        parts.push('Routed to balanced model for capable response.');
   610→   610→        break;
   611→   611→      case 'complex':
   612→   612→        parts.push('Routed to powerful model for thorough analysis.');
   613→   613→        break;
   614→   614→    }
   615→   615→
   616→   616→    if (effort === 'ultrathink') {
   617→   617→      parts.push('Ultrathink mode: using highest-capability model for deep reasoning.');
   618→   618→    } else if (effort === 'high') {
   619→   619→      parts.push('High effort: using advanced model for detailed output.');
   620→   620→    } else if (effort === 'low') {
   621→   621→      parts.push('Low effort: using fast model for concise response.');
   622→   622→    }
   623→   623→
   624→   624→    parts.push(`Selected "${model}".`);
   625→   625→
   626→   626→    return parts.join(' ');
   627→   627→  }

   630→   630→  // Private: config validation
   631→   631→  // -----------------------------------------------------------------------
   632→   632→
   633→   633→  private validateConfig(): void {
   634→   634→    const requiredModels: string[] = [
   635→   635→      this.config.defaultModel,
   636→   636→      this.config.simpleModel,
   637→   637→      this.config.moderateModel,
   638→   638→      this.config.complexModel,
   639→   639→    ];
   640→   640→
   641→   641→    for (const modelId of requiredModels) {
   642→   642→      if (!this.availableModels[modelId]) {
   643→   643→        throw new Error(
   644→   644→          `RouterConfig references model "${modelId}" which is not in the available models list. ` +
   645→   645→          `Available: ${Object.keys(this.availableModels).join(', ')}`,
   646→   646→        );
   647→   647→      }
   648→   648→    }
   649→   649→
   650→   650→    for (const [level, modelId] of Object.entries(this.config.effortModels)) {
   651→   651→      if (!this.availableModels[modelId]) {
   652→   652→        throw new Error(
   653→   653→          `RouterConfig effortModels.${level} references model "${modelId}" which is not in the available models list. ` +
   654→   654→          `Available: ${Object.keys(this.availableModels).join(', ')}`,
   655→   655→        );
   656→   656→      }
   657→   657→    }
   658→   658→
   659→   659→    for (const [category, modelId] of Object.entries(this.config.categoryOverrides)) {
   660→   660→      if (modelId && !this.availableModels[modelId]) {
   661→   661→        throw new Error(
   662→   662→          `RouterConfig categoryOverrides.${category} references model "${modelId}" which is not in the available models list. ` +
   663→   663→          `Available: ${Object.keys(this.availableModels).join(', ')}`,
   664→   664→        );
   665→   665→      }
   666→   666→    }
   667→   667→
   668→   668→    if (this.config.maxTokenBudget <= 0) {
   669→   669→      throw new Error('RouterConfig.maxTokenBudget must be a positive integer.');
   670→   670→    }
   671→   671→  }
   672→   672→}
